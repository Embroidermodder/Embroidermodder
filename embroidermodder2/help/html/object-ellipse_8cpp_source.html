263// Returns the closest snap point to the mouse point
264QPointF EllipseObject::mouseSnapPoint(const QPointF& mousePoint)
265{
266 QPointF center = objectCenter();
267 QPointF quad0 = objectQuadrant0();
268 QPointF quad90 = objectQuadrant90();
269 QPointF quad180 = objectQuadrant180();
270 QPointF quad270 = objectQuadrant270();
271
272 qreal cntrDist = QLineF(mousePoint, center).length();
273 qreal q0Dist = QLineF(mousePoint, quad0).length();
274 qreal q90Dist = QLineF(mousePoint, quad90).length();
275 qreal q180Dist = QLineF(mousePoint, quad180).length();
276 qreal q270Dist = QLineF(mousePoint, quad270).length();
277
278 qreal minDist = qMin(qMin(qMin(q0Dist, q90Dist), qMin(q180Dist, q270Dist)), cntrDist);
279
280 if (minDist == cntrDist) return center;
281 else if(minDist == q0Dist) return quad0;
282 else if(minDist == q90Dist) return quad90;
283 else if(minDist == q180Dist) return quad180;
284 else if(minDist == q270Dist) return quad270;
285
286 return scenePos();
287}
