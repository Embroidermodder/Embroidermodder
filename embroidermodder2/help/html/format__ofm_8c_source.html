190ofmReadExpanded(FILE* file, EmbPattern* p)
191{
192 int i, numberOfStitches = 0;
193
194 if (!file) { printf("ERROR: format-ofm.c ofmReadExpanded(), file argument is null\n"); return; }
195 if (!p) { printf("ERROR: format-ofm.c ofmReadExpanded(), p argument is null\n"); return; }
196
197 ofmReadBlockHeader(file);
198 embInt_read(file, "numberOfStitchs", &numberOfStitches, EMB_INT32_LITTLE);
199
200 for (i = 0; i < numberOfStitches; i++) {
201 unsigned char stitch[5];
202 fread(stitch, 1, 5, file); /* TODO: check return value */
203 if (stitch[0] == 0) {
204 embPattern_addStitchAbs(p, ofmDecode(stitch[1], stitch[2]) / 10.0, ofmDecode(stitch[3], stitch[4]) / 10.0, i == 0 ? JUMP : NORMAL, 1);
205 }
206 else if (stitch[0] == 32) {
207 embPattern_addStitchAbs(p, ofmDecode(stitch[1], stitch[2]) / 10.0, ofmDecode(stitch[3], stitch[4]) / 10.0, i == 0 ? TRIM : NORMAL, 1);
208 }

212char
213readOfm(EmbPattern* pattern, FILE* fileCompound)
214{
215 int unknownCount, key = 0, classNameLength;
216 char* s = 0;
217 FILE *file;
218 bcf_file* bcfFile = 0;
219
220 if (emb_verbose>1) {
221 puts("Overridden during development.");
222 return 0;
223 }
224
225 bcfFile = (bcf_file*)malloc(sizeof(bcf_file));
226 if (!bcfFile) {
227 printf("ERROR: format-ofm.c readOfm(), unable to allocate memory for bcfFile\n");
228 return 0;
229 }
230 bcfFile_read(fileCompound, bcfFile);
231 file = GetFile(bcfFile, fileCompound, "EdsIV Object");
232 bcf_file_free(bcfFile);
233 bcfFile = 0;
234 fseek(file, 0x1C6, SEEK_SET);
235 ofmReadThreads(file, pattern);
236 fseek(file, 0x110, SEEK_CUR);
237 fseek(file, 0x4, SEEK_CUR); /* EMB_INT32_LITTLE */
238 classNameLength = fread_int16(file);
239 s = (char*)malloc(sizeof(char) * classNameLength);
240 if (!s) {
241 printf("ERROR: format-ofm.c readOfm(), unable to allocate memory for s\n");
242 return 0;
243 }
244 fread((unsigned char*)s, 1, classNameLength, file); /* TODO: check return value */
245 unknownCount = fread_int16(file);
246 /* TODO: determine what unknown count represents */
247 if (emb_verbose>1) {
248 printf("unknownCount = %d\n", unknownCount);
249 }
250
251 fread_int16(file);
252 key = ofmReadClass(file);
253 while (1) {
254 if (key == 0xFEFF) {
255 break;
256 }
257 if (key == 0x809C) {
258 ofmReadExpanded(file, pattern);
259 }
260 else {
261 ofmReadColorChange(file, pattern);
262 }
263 key = fread_uint16(file);
264 if (key == 0xFFFF) {
265 ofmReadClass(file);
266 }
267 }
268
269 embPattern_flip(pattern, 1, 1);
270
271 return 1;
272}
