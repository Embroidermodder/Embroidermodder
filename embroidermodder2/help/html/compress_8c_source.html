73void
74huffman_build_table(huffman *h)
75{
76 int bit_length, i, max_length, size;
77 max_length = 0;
78 size = 1 << h->table_width;
79 for (i = 0; i < h->table_width; i++) {
80 if (h->lengths[i] > max_length) {
81 max_length = h->lengths[i];
82 }
83 }
84 for (bit_length=1; bit_length<=h->table_width; bit_length++) {
85 int j;
86 size /= 2;
87 for (j=0; j < h->nlengths; j++) {
88 if (h->lengths[j] == bit_length) {
89 int k;
90 for (k=0; k<size; k++) {
91 h->table[h->ntable+k] = j;
92 h->ntable++;
93 }
94 }
95 }
96 }
97}

143int
144compress_get_bits(compress *c, int length)
145{
146 int i, end_pos_in_bits, start_pos_in_bytes,
147 end_pos_in_bytes, value, mask_sample_bits,
148 unused_bits, original;
149
150 end_pos_in_bits = c->bit_position + length - 1;
151 start_pos_in_bytes = c->bit_position / 8;
152 end_pos_in_bytes = end_pos_in_bits / 8;
153 value = 0;
154
155 for (i=start_pos_in_bytes; i < end_pos_in_bytes+1; i++) {
156 value <<= 8;
157 if (i > c->input_length) {
158 break;
159 }
160 value |= (c->input_data[i]) & 0xFF;
161 }
162
163 unused_bits = (7 - end_pos_in_bits) % 8;
164 mask_sample_bits = (1<<length) - 1;
165 original = (value >> unused_bits) & mask_sample_bits;
166 return original;
167}
