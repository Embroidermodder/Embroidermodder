55{
56 qreal rot = radians(rotation());
57 qreal cosRot = qCos(rot);
58 qreal sinRot = qSin(rot);
59

64 qreal ptlYrot = ptlX*sinRot + ptlY*cosRot;

196// Returns the closest snap point to the mouse point
197QPointF RectObject::mouseSnapPoint(const QPointF& mousePoint)
198{
199 QPointF ptl = objectTopLeft(); //Top Left Corner QSnap
200 QPointF ptr = objectTopRight(); //Top Right Corner QSnap
201 QPointF pbl = objectBottomLeft(); //Bottom Left Corner QSnap
202 QPointF pbr = objectBottomRight(); //Bottom Right Corner QSnap
203
204 qreal ptlDist = QLineF(mousePoint, ptl).length();
205 qreal ptrDist = QLineF(mousePoint, ptr).length();
206 qreal pblDist = QLineF(mousePoint, pbl).length();
207 qreal pbrDist = QLineF(mousePoint, pbr).length();
208
209 qreal minDist = qMin(qMin(ptlDist, ptrDist), qMin(pblDist, pbrDist));
210
211 if (minDist == ptlDist) return ptl;
212 else if(minDist == ptrDist) return ptr;
213 else if(minDist == pblDist) return pbl;
214 else if(minDist == pbrDist) return pbr;
215
216 return scenePos();
217}
