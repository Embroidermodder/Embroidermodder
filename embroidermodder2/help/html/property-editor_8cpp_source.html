588 if(val) fieldNewText = fieldOnText;
589 else fieldNewText = fieldOffText;
590 }
591
592 if(fieldOldText.isEmpty())
593 {
594 if(yesOrNoText)
595 {
596 comboBox->addItem(fieldYesText, true);
597 comboBox->addItem(fieldNoText, false);
598 }
599 else
600 {
601 comboBox->addItem(fieldOnText, true);
602 comboBox->addItem(fieldOffText, false);
603 }
604 comboBox->setCurrentIndex(comboBox->findText(fieldNewText));
605 }
606 else if(fieldOldText != fieldNewText)
607 {
608 if(comboBox->findText(fieldVariesText) == -1) //Prevent multiple entries
609 comboBox->addItem(fieldVariesText);
610 comboBox->setCurrentIndex(comboBox->findText(fieldVariesText));
611 }
