96void CircleObject::updatePath()
97{
98 QPainterPath path;
99 QRectF r = rect();
100 //Add the center point
101 path.addRect(-0.00000001, -0.00000001, 0.00000002, 0.00000002);
102 //Add the circle
103 path.arcMoveTo(r, 0);
104 path.arcTo(r, 0, 360);
105 //NOTE: Reverse the path so that the inside area isn't considered part of the circle
106 path.arcTo(r, 0, -360);

126{
127 int rubberMode = objectRubberMode();
128 if(rubberMode == OBJ_RUBBER_CIRCLE_1P_RAD)
129 {
130 QPointF sceneCenterPoint = objectRubberPoint("CIRCLE_CENTER");
131 QPointF sceneQSnapPoint = objectRubberPoint("CIRCLE_RADIUS");

140 updatePath();
141 }
142 else if(rubberMode == OBJ_RUBBER_CIRCLE_1P_DIA)
143 {
144 QPointF sceneCenterPoint = objectRubberPoint("CIRCLE_CENTER");
145 QPointF sceneQSnapPoint = objectRubberPoint("CIRCLE_DIAMETER");
146 QPointF itemCenterPoint = mapFromScene(sceneCenterPoint);
147 QPointF itemQSnapPoint = mapFromScene(sceneQSnapPoint);
148 QLineF itemLine(itemCenterPoint, itemQSnapPoint);
149 setObjectCenter(sceneCenterPoint);
150 QLineF sceneLine(sceneCenterPoint, sceneQSnapPoint);
151 qreal diameter = sceneLine.length();
152 setObjectDiameter(diameter);
153 if(painter) drawRubberLine(itemLine, painter, VIEW_COLOR_CROSSHAIR);
154 updatePath();
155 }
156 else if(rubberMode == OBJ_RUBBER_CIRCLE_2P)
157 {
158 QPointF sceneTan1Point = objectRubberPoint("CIRCLE_TAN1");
159 QPointF sceneQSnapPoint = objectRubberPoint("CIRCLE_TAN2");
160 QLineF sceneLine(sceneTan1Point, sceneQSnapPoint);
161 setObjectCenter(sceneLine.pointAt(0.5));
162 qreal diameter = sceneLine.length();
163 setObjectDiameter(diameter);
164 updatePath();
165 }
166 else if(rubberMode == OBJ_RUBBER_CIRCLE_3P)
167 {
168 QPointF sceneTan1Point = objectRubberPoint("CIRCLE_TAN1");
169 QPointF sceneTan2Point = objectRubberPoint("CIRCLE_TAN2");
170 QPointF sceneTan3Point = objectRubberPoint("CIRCLE_TAN3");
171
172 double sceneCenterX;
173 double sceneCenterY;
174 getArcCenter(sceneTan1Point.x(), sceneTan1Point.y(),
175 sceneTan2Point.x(), sceneTan2Point.y(),
176 sceneTan3Point.x(), sceneTan3Point.y(),
177 &sceneCenterX, &sceneCenterY);
178 QPointF sceneCenterPoint(sceneCenterX, sceneCenterY);
179 QLineF sceneLine(sceneCenterPoint, sceneTan3Point);
180 setObjectCenter(sceneCenterPoint);
181 qreal radius = sceneLine.length();
182 setObjectRadius(radius);
183 updatePath();
184 }
185 else if(rubberMode == OBJ_RUBBER_GRIP)
186 {
187 if(painter)
188 {
189 QPointF gripPoint = objectRubberPoint("GRIP_POINT");
190 if(gripPoint == objectCenter())
191 {
192 painter->drawEllipse(rect().translated(mapFromScene(objectRubberPoint(QString()))-mapFromScene(gripPoint)));
193 }
194 else
195 {
196 qreal gripRadius = QLineF(objectCenter(), objectRubberPoint(QString())).length();
197 painter->drawEllipse(QPointF(), gripRadius, gripRadius);
198 }
199
200 QLineF rubLine(mapFromScene(gripPoint), mapFromScene(objectRubberPoint(QString())));
201 drawRubberLine(rubLine, painter, VIEW_COLOR_CROSSHAIR);
202 }

214// Returns the closest snap point to the mouse point
215QPointF CircleObject::mouseSnapPoint(const QPointF& mousePoint)
216{
217 QPointF center = objectCenter();
218 QPointF quad0 = objectQuadrant0();
219 QPointF quad90 = objectQuadrant90();
220 QPointF quad180 = objectQuadrant180();
221 QPointF quad270 = objectQuadrant270();
222
223 qreal cntrDist = QLineF(mousePoint, center).length();
224 qreal q0Dist = QLineF(mousePoint, quad0).length();
225 qreal q90Dist = QLineF(mousePoint, quad90).length();
226 qreal q180Dist = QLineF(mousePoint, quad180).length();
227 qreal q270Dist = QLineF(mousePoint, quad270).length();
228
229 qreal minDist = qMin(qMin(qMin(q0Dist, q90Dist), qMin(q180Dist, q270Dist)), cntrDist);
230
231 if (minDist == cntrDist) return center;
232 else if(minDist == q0Dist) return quad0;
233 else if(minDist == q90Dist) return quad90;
234 else if(minDist == q180Dist) return quad180;
235 else if(minDist == q270Dist) return quad270;
236
237 return scenePos();
238}
