95void UndoableRotateCommand::rotate(qreal x, qreal y, qreal rot)
96{
97 qreal rad = radians(rot);
98 qreal cosRot = qCos(rad);
99 qreal sinRot = qSin(rad);
100 qreal px = object->scenePos().x();
101 qreal py = object->scenePos().y();
102 px -= x;
103 py -= y;
104 qreal rotX = px*cosRot - py*sinRot;
105 qreal rotY = px*sinRot + py*cosRot;
106 rotX += x;
107 rotY += y;
108
109 object->setPos(rotX, rotY);
110 object->setRotation(object->rotation()+rot);
111}

117UndoableScaleCommand::UndoableScaleCommand(qreal x, qreal y, qreal scaleFactor, const QString& text, BaseObject* obj, View* v, QUndoCommand* parent) : QUndoCommand(parent)
118{
119 gview = v;
120 object = obj;
121 setText(text);
122
123 //Prevent division by zero and other wacky behavior
124 if(scaleFactor <= 0.0)
125 {
126 dx = 0.0;
127 dy = 0.0;
128 factor = 1.0;
129 QMessageBox::critical(0, QObject::tr("ScaleFactor Error"),

132 }
133 else
134 {
135 //Calculate the offset
136 qreal oldX = object->x();
137 qreal oldY = object->y();
138 QLineF scaleLine(x, y, oldX, oldY);
139 scaleLine.setLength(scaleLine.length()*scaleFactor);
140 qreal newX = scaleLine.x2();
141 qreal newY = scaleLine.y2();
142
143 dx = newX - oldX;
144 dy = newY - oldY;
145 factor = scaleFactor;
146 }
