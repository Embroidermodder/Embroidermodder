14void
15ksmEncode(unsigned char* b, char dx, char dy, int flags)
16{
17 if (!b) {
18 printf("ERROR: format-ksm.c ksmEncode(), b argument is null\n");
19 return;
20 }
21 /* TODO: How to encode JUMP stitches? JUMP must be handled.
22 Also check this for the EXP format since it appears to be similar */
23 if (flags == TRIM) {
24 b[0] = 128;
25 b[1] = 2;
26 b[2] = dx;
27 b[3] = dy;
28 } else if (flags == STOP) {
29 b[0] = 128;
30 b[1] = 1;
31 b[2] = dx;
32 b[3] = dy;
33 } else {
34 b[0] = dx;
35 b[1] = dy;
36 }
37}

39char
40readKsm(EmbPattern* pattern, FILE* file)
41{
42 int prevStitchType = NORMAL;
43 char b[3];
44 fseek(file, 0x200, SEEK_SET);
45 while (fread(b, 1, 3, file) == 3) {
46 int flags = NORMAL;
47
48 if (((prevStitchType & 0x08) == 0x08) && (b[2] & 0x08) == 0x08) {
49 flags = STOP;
50 } else if ((b[2] & 0x1F) != 0) {
51 flags = TRIM;
52 }
53 prevStitchType = b[2];
54 if (b[2] & 0x40) {
55 b[1] = -b[1];
56 }
57 if (b[2] & 0x20) {
58 b[0] = -b[0];
59 }
60 embPattern_addStitchRel(pattern, b[1] / 10.0, b[0] / 10.0, flags, 1);
61 }
62 return 1;
63}

65char
66writeKsm(EmbPattern* pattern, FILE* file)
67{
68 EmbVector pos;
69 int i;
70
71 fpad(file, 0, 0x200);
72 /* write stitches */
73 pos.x = 0.0;
74 pos.y = 0.0;
75 for (i = 0; i < pattern->stitch_list->count; i++) {
76 unsigned char b[4];
77 char dx, dy;
78 EmbStitch st = pattern->stitch_list->stitch[i];
79 dx = (char)(10.0*(st.x - pos.x));
80 dy = (char)(10.0*(st.y - pos.y));
81 pos.x += 0.1*dx;
82 pos.y += 0.1*dy;
83 ksmEncode(b, dx, dy, st.flags);
84 fprintf(file, "%c%c", b[0], b[1]);
85 }
86 fprintf(file, "\x1a");
87 return 1;
88}
