37int lindenmayer_system(L_system L, char *state, int iterations, int complete)
38{
39 /* We know that the full length of the curve has to fit within
40 * the number of stitches and we can cancel consecutive +-, -+
41 * etc.
42 *
43
44Potential reference:
45
46@book{Prusinkiewicz1996Mar,
47 author = {Prusinkiewicz, Przemyslaw and Lindenmayer, Aristid and Hanan, J. S. and Fracchia, F. D. and Fowler, D. R. and de Boer, M. J. M. and Mercer, L.},
48 title = {{The Algorithmic Beauty of Plants (The Virtual Laboratory)}},
49 year = {1996},
50 month = {Mar},
51 publisher = {Springer}
52}
53 */
54 char *new_state;
55 int j;
56
57 if (complete == 0) {
58 state[0] = L.axiom;
59 state[1] = 0;
60 lindenmayer_system(L, state, iterations, complete+1);
61 return 0;
62 }
63
64 new_state = state + MAX_STITCHES*5;
65
66 new_state[0] = 0;
67
68 /* replace letters using rules by copying to new_state */
69 for (j=0; j < (int)strlen(state); j++) {
70 if (state[j] >= 'A' && state[j] < 'F') {
71 strcat(new_state, L.rules[state[j]-'A']);
72 }
73 if (state[j] == 'F') {
74 strcat(new_state, "F");
75 }
76 if (state[j] == '+') {
77 strcat(new_state, "+");
78 }
79 if (state[j] == '-') {
80 strcat(new_state, "-");
81 }
82 }
83 memcpy(state, new_state, strlen(new_state)+1);
84
85 if (complete < iterations) {
86 lindenmayer_system(L, state, iterations, complete+1);
87 }
88 return 0;
89}

102static void
103join_short_stitches(int *points, int *n_points, int width, int tolerence)
104{
105 int found = 1;
106 while (found > 0) {
107 int i;
108 found = 0;
109 for (i=*n_points-2; i>=0; i--) {
110 int st1 = points[i+1]%width - points[i]%width;
111 int st2 = points[i+2]%width - points[i+1]%width;
112 int same_line = (points[i+1]/width == points[i]/width)
113 && (points[i+2]/width == points[i+1]/width);
114 if (st1 < tolerence && st2 < tolerence && same_line) {
115 found++;
116 break;
117 }
118 }
119 if (found) {
120 /* Remove the point. */
121 i++;
122 for (; i<*n_points; i++) {
123 points[i] = points[i+1];
124 }
125 (*n_points)--;
126 }
127 }
128}

142static int *
143threshold_method(EmbImage *image, int *n_points,
144 int subsample_width, int subsample_height, int threshold)
145{
146 int i, j;
147 int *points;
148 int height = image->height;
149 int width = image->width;
150 points = (int *)malloc((height/subsample_height)
151 *(width/subsample_width) * sizeof(int));
152 *n_points = 0;
153 for (i=0; i<height/subsample_height; i++)
154 for (j=0; j<width/subsample_width; j++) {
155 EmbColor color;
156 int index = subsample_height*i*width+subsample_width*j;
157 color.r = image->data[3*index+0];
158 color.g = image->data[3*index+1];
159 color.b = image->data[3*index+2];
160 if (color.r+color.g+color.b < threshold) {
161 points[*n_points] = index;
162 (*n_points)++;
163 }
164 }
165 return points;
166}

186greedy_algorithm(int *points, int n_points, int width, EmbReal bias)
187{
188 int i, j;
189 printf("points[0] = %d\n", points[0]);
190 printf("n_points = %d\n", n_points);
191 printf("width = %d\n", width);
192 printf("bias = %f\n", bias);
193
194 for (i=0; i<n_points-1; i++) {
195 int stor;
196 EmbReal shortest = 1.0e20;
197 int next = i+1;
198 /* Find nearest neighbour. */
199 int x1 = points[i]%width;
200 int y1 = points[i]/width;
201 for (j=i+1; j<n_points; j++) {
202 int x, y;
203 EmbReal distance;
204 x = x1 - (points[j]%width);
205 if (x*x > shortest) {
206 continue;
207 }
208 y = y1 - (points[j]/width);
209 distance = x*x + bias*y*y;
210 if (distance < shortest) {
211 next = j;
212 shortest = distance;
213 }
214 }
215 if (i%100 == 0) {
216 printf("%2.1f%%\n", (100.0*i)/(1.0*n_points));
217 }
218 /* swap points */
219 stor = points[next];
220 points[next] = points[i+1];
221 points[i+1] = stor;
222 }

339int
340hilbert_curve(EmbPattern *pattern, int iterations)
341{
342 char *state;
343 int i, position[2], direction;
344 EmbReal scale = 1.0;
345
346 /* Make the n-th iteration. */
347 state = malloc(MAX_STITCHES*10);
348 lindenmayer_system(hilbert_curve_l_system, state, iterations, 0);
349
350 /* Convert to an embroidery pattern. */
351 position[0] = 0;
352 position[1] = 0;
353 direction = 0;
354
355 for (i = 0; i < (int)strlen(state); i++) {
356 if (state[i] == '+') {
357 direction = (direction + 1) % 4;
358 continue;
359 }
360 if (state[i] == '-') {
361 direction = (direction + 3) % 4;
362 continue;
363 }
364 if (state[i] == 'F') {
365 int flags = NORMAL;
366 switch (direction) {
367 case 0:
368 default:
369 position[0]--;
370 break;
371 case 1:
372 position[1]++;
373 break;
374 case 2:
375 position[0]++;
376 break;
377 case 3:
378 position[1]--;
379 break;
380 }
381 embPattern_addStitchAbs(pattern, position[0]*scale, position[1]*scale, flags, 0);
382 }
383 }
384 free(state);
385 embPattern_end(pattern);
386 return 0;
387}

399{
400 int i, length;
401 if (iterations == 1) {
402 state[0] = 'R';
403 state[1] = 0;
404 return;
405 }
406 length = strlen(state);
407 for (i=length-1; i>=0; i--) {
408 state[2*i+1] = state[i];
409 if (i%2 == 0) {
410 state[2*i] = 'R';
411 } else {
412 state[2*i] = 'L';
413 }
414 }
415 state[2*length+1] = 0;
416 generate_dragon_curve(state, iterations-1);
417}

680Vertices CollinearSimplify(Vertices vertices, float collinearityTolerance)
681{
682 /* We can't simplify polygons under 3 vertices */

691

693 Vector2 current = vertices[i];
694 Vector2 next = vertices[nextId];

799EmbReal
800DistancePointLine(EmbVector p, EmbVector a, EmbVector b)
801{
802 /* if start == end, then use point-to-point distance */
803 if (a.X == b.X && a.Y == b.Y)
804 return DistancePointPoint(p, a);
805
806 /* otherwise use comp.graphics.algorithms Frequently Asked Questions method */
807 /* (1) AC dot AB
808 r = ---------
809 ||AB||^2
810
811 r has the following meaning:
812 r=0 Point = A
813 r=1 Point = B
814 r<0 Point is on the backward extension of AB
815 r>1 Point is on the forward extension of AB
816 0<r<1 Point is interior to AB
817 */
818
819 EmbReal r = ((p.X - a.X) * (b.X - a.X) + (p.Y - a.Y) * (b.Y - a.Y))
820 /
821 ((b.X - a.X) * (b.X - a.X) + (b.Y - a.Y) * (b.Y - a.Y));
822
823 if (r <= 0.0) return DistancePointPoint(p, a);
824 if (r >= 1.0) return DistancePointPoint(p, b);
825
826
827 /* (2)
828 (Ay-Cy)(Bx-Ax)-(Ax-Cx)(By-Ay)
829 s = -----------------------------
830 Curve^2
831
832 Then the distance from C to Point = |s|*Curve.
833 */
834
835 EmbReal s = ((a.Y - p.Y) * (b.X - a.X) - (a.X - p.X) * (b.Y - a.Y))
836 /
837 ((b.X - a.X) * (b.X - a.X) + (b.Y - a.Y) * (b.Y - a.Y));
838
839 return fabs(s) * sqrt(((b.X - a.X) * (b.X - a.X) + (b.Y - a.Y) * (b.Y - a.Y)));
840}

851public Vertices
852ReduceByArea(Vertices vertices, float areaTolerance)
853{
854 if (vertices.Count <= 3)
855 return vertices;
856
857 if (areaTolerance < 0)
858 {
859 throw new ArgumentOutOfRangeException("areaTolerance", "must be equal to or greater then zero.");
860 }
861
862 int result = new Vertices();
863 Vector2 v3;
864 Vector2 v1 = vertices[vertices.Count - 2];
865 Vector2 v2 = vertices[vertices.Count - 1];
866 areaTolerance *= 2;
867 for (int index = 0; index < vertices.Count; ++index, v2 = v3)
868 {
869 if (index == vertices.Count - 1)
870 {
871 if (result.Count == 0)
872 {
873 throw new ArgumentOutOfRangeException("areaTolerance", "The tolerance is too high!");
874 }
875 v3 = result[0];
876 }
877 else
878 {
879 v3 = vertices[index];
880 }
881 float old1, old2, new1;
882 MathUtils.Cross(ref v1, ref v2, out old1);
883 MathUtils.Cross(ref v2, ref v3, out old2);
884 MathUtils.Cross(ref v1, ref v3, out new1);
885 if (fabs(new1 - (old1 + old2)) > areaTolerance)
886 {
887 result.Add(v2);
888 v1 = v2;
889 }
890 }
891 return result;
892}

903void
904MergeParallelEdges(EmbArray *vertices, float tolerance)
905{
906 int i;
907 if (vertices.Count <= 3) {
908 /* Can't do anything useful here to a triangle. */
909 return;
910 }
911
912 int mergeMe = new bool[vertices.Count];
913 int newNVertices = vertices.Count;
914
915 /* Gather points to process */
916 for (i = 0; i < vertices->count; i++) {
917 EmbVector delta0, delta1;
918 int lower = (i == 0) ? (vertices.Count - 1) : (i - 1);
919 int upper = (i == vertices.Count - 1) ? (0) : (i + 1);
920
921 delta0 = embVector_subtract(vertices[i], vertices[lower]);
922 delta1 = embVector_subtract(vertices[upper], vertices[i]);
923 float norm0 = embVector_length(delta0);
924 float norm1 = embVector_length(delta0);
925

927 /* Merge identical points */
928 mergeMe[i] = 1;
929 newNVertices--;
930 }

933 embVector_normalize(delta1, &delta1);

935 float dot = embVector_dot(delta0, delta1);

938 mergeMe[i] = 1;
939 newNVertices--;
940 }
941 else {
942 mergeMe[i] = 0;
943 }
944 }

947 return;

952 int oldVertices = new Vertices(vertices);
953 vertices.Clear();

958
959 vertices.Add(oldVertices[i]);
