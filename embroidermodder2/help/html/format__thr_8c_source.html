15/*
16 * bit definitions for attributes of stitch
17 * 0-3 stitch color
18 * 4-14 form pointer
19 * 15-18 spares
20 * 19 not a form stitch
21 * 20 center walk stitch
22 * 21 edge walk stitch
23 * 22 underlay stitch
24 * 23 knot stitch
25 * 24 feather stitch
26 * 25-27 layer
27 * 28 spare
28 * 29-30 stitch type
29 * 00=not a form stitch,
30 * 01=form fill,
31 * 10=form border fill,
32 * 11=applique stitches
33 * 31 set for user edited stitches
34 */

36char
37readThr(EmbPattern* pattern, FILE* file)
38{
39 ThredHeader header;
40 EmbColor background;
41 int currentColor;
42 int i;
43
44 embInt_read(file, "sig version", &(header.sigVersion), EMB_INT32_LITTLE);
45 embInt_read(file, "length", &(header.length), EMB_INT32_LITTLE);
46 embInt_read(file, "number of stitches", &(header.numStiches), EMB_INT16_LITTLE);
47 header.hoopSize = fread_uint16(file);
48 for (i=0; i<7; i++) {
49 header.reserved[i] = fread_uint16(file);
50 }
51
52 if ((header.sigVersion & 0xffffff) == 0x746872) {
53 unsigned int verVar = (header.sigVersion & 0xff000000) >> 24;
54 switch (verVar) {
55 case 0:
56 break;
57 case 1:
58 case 2:
59 /* skip the file header extension */
60 fseek(file, 144, SEEK_CUR);
61 break;
62 default:
63 return 0; /* unsupported version */
64 }
65 }
66 currentColor = -1;
67 for (i = 0; i < header.numStiches; i++) {
68 int type = NORMAL;
69 float x, y;
70 unsigned int color;
71 embInt_read(file, "x", &x, EMB_INT32_LITTLE);
72 embInt_read(file, "y", &y, EMB_INT32_LITTLE);
73 embInt_read(file, "color", &color, EMB_INT32_LITTLE);
74 x /= 10.0;
75 y /= 10.0;
76
77 if ((int)(color & 0xF) != currentColor) {
78 currentColor = (int)color & 0xF;
79 embPattern_changeColor(pattern, currentColor);
80 type = STOP | TRIM;
81 }
82 embPattern_addStitchAbs(pattern, x, y, type, 0);
83 }
84 fseek(file, 16, SEEK_CUR); /* skip bitmap name (16 chars) */
85
86 embColor_read(file, &background, 4);
87 if (emb_verbose>1) {
88 printf("background: %c %c %c\n", background.r, background.g, background.b);
89 }
90 for (i = 0; i < 16; i++) {
91 EmbThread thread;
92 strcpy(thread.description, "NULL");
93 strcpy(thread.catalogNumber, "NULL");
94 embColor_read(file, &(thread.color), 4);
95 embPattern_addThread(pattern, thread);
96 }
97 /* 64 bytes of rgbx(4 bytes) colors (16 custom colors) */
98 /* 16 bytes of thread size (ascii representation ie. '4') */
99 return 1;
100}

102char
103writeThr(EmbPattern* pattern, FILE* file)
104{
105 int i, stitchCount;
106 unsigned char version = 0;
107 ThredHeader header;
108 ThredExtension extension;
109 char bitmapName[16];
110
111 stitchCount = pattern->stitch_list->count;
112
113 memset(&header, 0, sizeof(ThredHeader));
114 header.sigVersion = 0x746872 | (version << 24);
115 header.length = stitchCount * 12 + 16;
116 if (version == 1 || version == 2) {
117 header.length = header.length + sizeof(ThredHeader);
118 }
119 header.numStiches = (unsigned short)stitchCount; /* number of stitches in design */
120 header.hoopSize = 5;
121
122 binaryWriteUInt(file, header.sigVersion);
123 binaryWriteUInt(file, header.length);
124 binaryWriteUShort(file, header.numStiches);
125 binaryWriteUShort(file, header.hoopSize);
126 for (i=0; i<7; i++) {
127 binaryWriteUShort(file, header.reserved[i]);
128 }
129
130 if (version == 1 || version == 2) {
131 memset(&extension, 0, sizeof(ThredExtension));
132 extension.auxFormat = 1;
133 extension.hoopX = 640;
134 extension.hoopY = 640;
135
136 embInt_write(file, ".", &(extension.hoopX), EMB_INT32_LITTLE);
137 embInt_write(file, ".", &(extension.hoopY), EMB_INT32_LITTLE);
138 embInt_write(file, ".", &(extension.stitchGranularity), EMB_INT32_LITTLE);
139 fwrite(extension.creatorName, 1, 50, file);
140 fwrite(extension.modifierName, 1, 50, file);
141 fputc(extension.auxFormat, file);
142 fwrite(extension.reserved, 1, 31, file);
143 }
144
145 /* write stitches */
146 for (i = 0; i < pattern->stitch_list->count; i++) {
147 unsigned int NOTFRM = 0x00080000;
148 EmbStitch st = pattern->stitch_list->stitch[i];
149 float x, y;
150 x = (float)(st.x * 10.0);
151 y = (float)(st.y * 10.0);
152 embInt_write(file, ".", &x, EMB_INT32_LITTLE);
153 embInt_write(file, ".", &y, EMB_INT32_LITTLE);
154 binaryWriteUInt(file, NOTFRM | (st.color & 0x0F));
155 }
156 fwrite(bitmapName, 1, 16, file);
157 /* background color */
158 fwrite("\xFF\xFF\xFF\x00", 1, 4, file);
159
160 for (i = 0; i < pattern->thread_list->count; i++) {
161 EmbColor c = pattern->thread_list->thread[i].color;
162 embColor_write(file, c, 4);
163 if (i >= 16) break;
164 }
165
166 /* write remaining colors if not yet 16 */
167 fpad(file, 0, 4*(16-i));
168
169 fpad(file, '4', 16);
170 return 1;
171}
