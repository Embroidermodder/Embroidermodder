65void TextSingleObject::setObjectText(const QString& str)
66{
67 objText = str;
68 QPainterPath textPath;
69 QFont font;
70 font.setFamily(objTextFont);
71 font.setPointSizeF(objTextSize);
72 font.setBold(objTextBold);
73 font.setItalic(objTextItalic);
74 font.setUnderline(objTextUnderline);
75 font.setStrikeOut(objTextStrikeOut);
76 font.setOverline(objTextOverline);
77 textPath.addText(0, 0, font, str);
78
79 //Translate the path based on the justification
80 QRectF jRect = textPath.boundingRect();
81 if (objTextJustify == "Left") { textPath.translate(-jRect.left(), 0); }
82 else if(objTextJustify == "Center") { textPath.translate(-jRect.center().x(), 0); }
83 else if(objTextJustify == "Right") { textPath.translate(-jRect.right(), 0); }
84 else if(objTextJustify == "Aligned") { } //TODO: TextSingleObject Aligned Justification
85 else if(objTextJustify == "Middle") { textPath.translate(-jRect.center()); }
86 else if(objTextJustify == "Fit") { } //TODO: TextSingleObject Fit Justification
87 else if(objTextJustify == "Top Left") { textPath.translate(-jRect.topLeft()); }
88 else if(objTextJustify == "Top Center") { textPath.translate(-jRect.center().x(), -jRect.top()); }
89 else if(objTextJustify == "Top Right") { textPath.translate(-jRect.topRight()); }
90 else if(objTextJustify == "Middle Left") { textPath.translate(-jRect.left(), -jRect.top()/2.0); }
91 else if(objTextJustify == "Middle Center") { textPath.translate(-jRect.center().x(), -jRect.top()/2.0); }
92 else if(objTextJustify == "Middle Right") { textPath.translate(-jRect.right(), -jRect.top()/2.0); }
93 else if(objTextJustify == "Bottom Left") { textPath.translate(-jRect.bottomLeft()); }
94 else if(objTextJustify == "Bottom Center") { textPath.translate(-jRect.center().x(), -jRect.bottom()); }
95 else if(objTextJustify == "Bottom Right") { textPath.translate(-jRect.bottomRight()); }
96
97 //Backward or Upside Down
98 if(objTextBackward || objTextUpsideDown)
99 {
100 qreal horiz = 1.0;
101 qreal vert = 1.0;
102 if(objTextBackward) horiz = -1.0;
103 if(objTextUpsideDown) vert = -1.0;
104
105 QPainterPath flippedPath;
106
107 QPainterPath::Element element;
108 QPainterPath::Element P2;
109 QPainterPath::Element P3;
110 QPainterPath::Element P4;
111 for(int i = 0; i < textPath.elementCount(); ++i)
112 {
113 element = textPath.elementAt(i);
114 if(element.isMoveTo())
115 {
116 flippedPath.moveTo(horiz * element.x, vert * element.y);
117 }
118 else if(element.isLineTo())
119 {
120 flippedPath.lineTo(horiz * element.x, vert * element.y);
121 }
122 else if(element.isCurveTo())
123 {
124 // start point P1 is not needed
125 P2 = textPath.elementAt(i); // control point
126 P3 = textPath.elementAt(i+1); // control point
127 P4 = textPath.elementAt(i+2); // end point
128
129 flippedPath.cubicTo(horiz * P2.x, vert * P2.y,
130 horiz * P3.x, vert * P3.y,
131 horiz * P4.x, vert * P4.y);
132 }
133 }
134 objTextPath = flippedPath;
135 }
136 else
137 objTextPath = textPath;
138
139 //Add the grip point to the shape path
140 QPainterPath gripPath = objTextPath;
141 gripPath.connectPath(objTextPath);
142 gripPath.addRect(-0.00000001, -0.00000001, 0.00000002, 0.00000002);
143 setObjectPath(gripPath);
144}

248{
249 int rubberMode = objectRubberMode();

302QList<QPainterPath> TextSingleObject::subPathList() const
303{
304 qreal s = scale();
305 QTransform trans;
306 trans.rotate(rotation());
307 trans.scale(s,s);
308
309 QList<QPainterPath> pathList;
310
311 QPainterPath path = objTextPath;
312
313 QPainterPath::Element element;
314 QList<int> pathMoves;
315 int numMoves = 0;
316
317 for(int i = 0; i < path.elementCount(); i++)
318 {
319 element = path.elementAt(i);
320 if(element.isMoveTo())
321 {
322 pathMoves << i;
323 numMoves++;
324 }
325 }
326
327 pathMoves << path.elementCount();
328
329 for(int p = 0; p < pathMoves.size()-1 && p < numMoves; p++)
330 {
331 QPainterPath subPath;
332 for(int i = pathMoves.value(p); i < pathMoves.value(p+1); i++)
333 {
334 element = path.elementAt(i);
335 if(element.isMoveTo())
336 {
337 subPath.moveTo(element.x, element.y);
338 }
339 else if(element.isLineTo())
340 {
341 subPath.lineTo(element.x, element.y);
342 }
343 else if(element.isCurveTo())
344 {
345 subPath.cubicTo(path.elementAt(i ).x, path.elementAt(i ).y, //control point 1
346 path.elementAt(i+1).x, path.elementAt(i+1).y, //control point 2
347 path.elementAt(i+2).x, path.elementAt(i+2).y); //end point
348 }
349 }
350 pathList.append(trans.map(subPath));
351 }
352
353 return pathList;
354}
