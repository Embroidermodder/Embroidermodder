302 //Ensure the list only has max amount of entries
303 while(settings_opensave_recent_list_of_files.size() > settings_opensave_recent_max_files)

305 settings_opensave_recent_list_of_files.removeLast();

411void MainWindow::openFilesSelected(const QStringList& filesToOpen)
412{
413 bool doOnce = true;
414
415 if(filesToOpen.count())
416 {
417 for(int i = 0; i < filesToOpen.count(); i++)
418 {
419 if(!validFileFormat(filesToOpen[i]))
420 continue;
421
422 QMdiSubWindow* existing = findMdiWindow(filesToOpen[i]);
423 if(existing)
424 {
425 mdiArea->setActiveSubWindow(existing);
426 continue;
427 }
428
429 //The docIndex doesn't need increased as it is only used for unnamed files
430 numOfDocs++;
431 MdiWindow* mdiWin = new MdiWindow(docIndex, mainWin, mdiArea, Qt::SubWindow);
432 connect(mdiWin, SIGNAL(sendCloseMdiWin(MdiWindow*)), this, SLOT(onCloseMdiWin(MdiWindow*)));
433 connect(mdiArea, SIGNAL(subWindowActivated(QMdiSubWindow*)), this, SLOT(onWindowActivated(QMdiSubWindow*)));
434
435 //Make sure the toolbars/etc... are shown before doing their zoomExtents
436 if(doOnce) { updateMenuToolbarStatusbar(); doOnce = false; }
437
438 if(mdiWin->loadFile(filesToOpen.at(i)))
439 {
440 statusbar->showMessage(tr("File(s) loaded"), 2000);
441 mdiWin->show();
442 mdiWin->showMaximized();
443 //Prevent duplicate entries in the recent files list
444 if(!settings_opensave_recent_list_of_files.contains(filesToOpen.at(i), Qt::CaseInsensitive))
445 {
446 settings_opensave_recent_list_of_files.prepend(filesToOpen.at(i));
447 }
448 //Move the recent file to the top of the list
449 else
450 {
451 settings_opensave_recent_list_of_files.removeAll(filesToOpen.at(i));
452 settings_opensave_recent_list_of_files.prepend(filesToOpen.at(i));
453 }
454 settings_opensave_recent_directory = QFileInfo(filesToOpen.at(i)).absolutePath();
455
456 View* v = mdiWin->getView();
457 if(v)
458 {
459 v->recalculateLimits();
460 v->zoomExtents();
461 }
462 }
463 else
464 {
465 mdiWin->close();
466 }
467 }
468 }
469
470 windowMenuAboutToShow();
471}

506{

509
510 foreach(QMdiSubWindow* subWindow, mdiArea->subWindowList())
511 {
512 MdiWindow* mdiWin = qobject_cast<MdiWindow*>(subWindow);
513 if(mdiWin)
