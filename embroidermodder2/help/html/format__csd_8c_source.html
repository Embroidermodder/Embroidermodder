41void
42BuildDecryptionTable(int seed) {
43 int i;
44 const int mul1 = 0x41C64E6D;
45 const int add1 = 0x3039;
46
47 for (i = 0; i < CsdSubMaskSize; i++) {
48 seed *= mul1;
49 seed += add1;
50 _subMask[i] = (char) ((seed >> 16) & 0xFF);
51 }
52 for (i = 0; i < CsdXorMaskSize; i++) {
53 seed *= mul1;
54 seed += add1;
55 _xorMask[i] = (char) ((seed >> 16) & 0xFF);
56 }
57}

59unsigned char DecodeCsdByte(long fileOffset,
60 unsigned char val, int type) {
61 int newOffset;
62
63 fileOffset = fileOffset - 1;
64 if (type != 0) {
65 int final;
66 int fileOffsetHigh = (int) (fileOffset & 0xFFFFFF00);
67 int fileOffsetLow = (int) (fileOffset & 0xFF);
68
69 newOffset = fileOffsetLow;
70 fileOffsetLow = fileOffsetHigh;
71 final = fileOffsetLow%0x300;
72 if (final != 0x100 && final != 0x200) {
73 newOffset = csd_decryptArray[newOffset] | fileOffsetHigh;
74 } else if (final != 0x100 && final == 0x200) {
75 if (newOffset == 0) {
76 fileOffsetHigh = fileOffsetHigh - 0x100;
77 }
78 newOffset = csd_decryptArray[newOffset] | fileOffsetHigh;
79 } else if (newOffset != 1 && newOffset != 0) {
80 newOffset = csd_decryptArray[newOffset] | fileOffsetHigh;
81 } else {
82 fileOffsetHigh = fileOffsetHigh - 0x100;
83 newOffset = csd_decryptArray[newOffset] | fileOffsetHigh;
84 }
85 } else {
86 newOffset = (int) fileOffset;
87 }
88 return ((unsigned char) ((unsigned char)
89 (val ^ _xorMask[newOffset%CsdXorMaskSize]) -
90 _subMask[newOffset%CsdSubMaskSize]));
91}

93char
94readCsd(EmbPattern* pattern, FILE* file) {
95 int i, type = 0;
96 unsigned char identifier[8];
97 unsigned char unknown1, unknown2;
98 char dx = 0, dy = 0;
99 int colorChange = -1;
100 int flags;
101 unsigned char colorOrder[14];
102
103 fread(identifier, 1, 8, file);
104 /* TODO: check return value */
105
106 if (identifier[0] != 0x7C && identifier[2] != 0xC3) {
107 type = 1;
108 }
109 if (type == 0) {
110 BuildDecryptionTable(0xC);
111 } else {
112 BuildDecryptionTable(identifier[0]);
113 }
114
115 for (i = 0; i < 16; i++) {
116 EmbThread thread;
117 thread.color.r = DecodeCsdByte(ftell(file), (unsigned char)fgetc(file), type);
118 thread.color.g = DecodeCsdByte(ftell(file), (unsigned char)fgetc(file), type);
119 thread.color.b = DecodeCsdByte(ftell(file), (unsigned char)fgetc(file), type);
120 strcpy(thread.catalogNumber, "");
121 strcpy(thread.description, "");
122 embPattern_addThread(pattern, thread);
123 }
124 unknown1 = DecodeCsdByte(ftell(file), (unsigned char)fgetc(file), type);
125 unknown2 = DecodeCsdByte(ftell(file), (unsigned char)fgetc(file), type);
126 if (emb_verbose>1) {
127 printf("unknown bytes to decode: %c %c", unknown1, unknown2);
128 }
129
130 for (i = 0; i < 14; i++) {
131 colorOrder[i] = (unsigned char) DecodeCsdByte(ftell(file), (unsigned char)fgetc(file), type);
132 }
133 for (i = 0; !feof(file); i++) {
134 char negativeX, negativeY;
135 unsigned char b0 = DecodeCsdByte(ftell(file), (unsigned char)fgetc(file), type);
136 unsigned char b1 = DecodeCsdByte(ftell(file), (unsigned char)fgetc(file), type);
137 unsigned char b2 = DecodeCsdByte(ftell(file), (unsigned char)fgetc(file), type);
138
139 if (b0 == 0xF8 || b0 == 0x87 || b0 == 0x91) {
140 break;
141 }
142 negativeX = ((b0 & 0x20) > 0);
143 negativeY = ((b0 & 0x40) > 0);
144 b0 = (unsigned char)(b0 & (0xFF ^ 0xE0));
145
146 if ((b0 & 0x1F) == 0) {
147 flags = NORMAL;
148 } else if ((b0 & 0x0C) > 0) {
149 flags = STOP;
150 if (colorChange >= 14) {
151 printf("Invalid color change detected\n");
152 }
153 embPattern_changeColor(pattern, colorOrder[colorChange % 14]);
154 colorChange += 1;
155 } else if ((b0 & 0x1F) > 0) {
156 flags = TRIM;
157 } else {
158 flags = NORMAL;
159 }
160 dx = (char) b2;
161 dy = (char) b1;
162 if (negativeX) {
163 dx = (char) -dx;
164 }
165 if (negativeY) {
166 dy = (char) -dy;
167 }
168 if (flags == STOP) {
169 embPattern_addStitchRel(pattern, 0, 0, flags, 1);
170 } else {
171 embPattern_addStitchRel(pattern, dx / 10.0, dy / 10.0, flags, 1);
172 }
173 }
174 return 1;
175}
