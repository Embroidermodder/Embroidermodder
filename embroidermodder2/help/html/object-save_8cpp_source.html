47
48 bool writeSuccessful = false;
49 int i;
50
51 formatType = embFormat_typeFromName(qPrintable(fileName));

95 else if(objType == OBJ_TYPE_TEXTMULTI) { addTextMulti(pattern, item); }
96 else if(objType == OBJ_TYPE_TEXTSINGLE) { addTextSingle(pattern, item); }
97 }
98
99 //TODO: handle EMBFORMAT_STCHANDOBJ also
100 if(formatType == EMBFORMAT_STITCHONLY)
101 embPattern_movePolylinesToStitchList(pattern); //TODO: handle all objects like this
102

107 //TODO: check the embLog for errors and if any exist, report them.

368void SaveObject::toPolyline(EmbPattern* pattern, const QPointF& objPos, const QPainterPath& objPath, const QString& layer, const QColor& color, const QString& lineType, const QString& lineWeight)
369{
370 qreal startX = objPos.x();
371 qreal startY = objPos.y();
372 EmbPointList* pointList = 0;
373 EmbPointList* lastPoint = 0;
374 QPainterPath::Element element;
375 for(int i = 0; i < objPath.elementCount(); ++i)
376 {
377 element = objPath.elementAt(i);
378 if(!pointList)
379 {
380 pointList = lastPoint = embPointList_create(element.x + startX, -(element.y + startY));
381 }
382 else
383 {
384 lastPoint = embPointList_add(lastPoint, embPoint_make(element.x + startX, -(element.y + startY)));
385 }
386 }
387
388 EmbPolylineObject* polyObject = embPolylineObject_create(pointList, embColor_make(color.red(), color.green(), color.blue()), 1); //TODO: proper lineType
