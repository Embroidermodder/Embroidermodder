25char
26readSew(EmbPattern* pattern, FILE* file)
27{
28 int i, flags, numberOfColors, fileLength;
29 char dx, dy, thisStitchIsJump = 0;
30
31 fseek(file, 0x00, SEEK_END);
32 fileLength = ftell(file);
33 fseek(file, 0x00, SEEK_SET);
34 numberOfColors = fgetc(file);
35 numberOfColors += (fgetc(file) << 8);
36
37
38 for (i = 0; i < numberOfColors; i++) {
39 int color = fread_int16(file);
40 embPattern_addThread(pattern, jefThreads[color%78]);
41 }
42 fseek(file, 0x1D78, SEEK_SET);
43
44 for (i = 0; ftell(file) < fileLength; i++) {
45 unsigned char b[2];
46 fread(b, 1, 2, file);
47
48 flags = NORMAL;
49 if (thisStitchIsJump) {
50 flags = TRIM;
51 thisStitchIsJump = 0;
52 }
53 if (b[0] == 0x80) {
54 if (b[1] == 1) {
55 fread(b, 1, 2, file);
56 flags = STOP;
57 } else if ((b[1] == 0x02) || (b[1] == 0x04)) {
58 thisStitchIsJump = 1;
59 fread(b, 1, 2, file);
60 flags = TRIM;
61 } else if (b[1] == 0x10) {
62 break;
63 }
64 }
65 dx = sewDecode(b[0]);
66 dy = sewDecode(b[1]);
67 if (abs(dx) == 127 || abs(dy) == 127) {
68 thisStitchIsJump = 1;
69 flags = TRIM;
70 }
71 embPattern_addStitchRel(pattern, dx / 10.0, dy / 10.0, flags, 1);
72 }
73 printf("current position: %ld\n", ftell(file));
74 return 1;
75}

77char
78writeSew(EmbPattern* pattern, FILE* file)
79{
80 int i;
81 EmbReal xx = 0.0, yy = 0.0;
82 binaryWriteShort(file, pattern->thread_list->count);
83
84 if (emb_verbose>1) {
85 printf("Debugging Information\n");
86 printf("number of colors = %d\n", pattern->thread_list->count);
87 printf("number of stitches = %d\n", pattern->stitch_list->count);
88 }
89
90 for (i = 0; i < pattern->thread_list->count; i++) {
91 short thr;
92 EmbColor col;
93 col = pattern->thread_list->thread[i].color;
94 thr = embThread_findNearestThread(col, (EmbThread *)jefThreads, 79);
95 binaryWriteShort(file, thr);
96 }
97 fpad(file, 0, 0x1D78 - 2 - pattern->thread_list->count * 2);
98
99 for (i = 0; i < pattern->stitch_list->count; i++) {
100 EmbStitch st;
101 unsigned char b[4];
102 char dx, dy;
103 st = pattern->stitch_list->stitch[i];
104 dx = (char)emb_round(10.0*(st.x - xx));
105 dy = (char)emb_round(10.0*(st.y - yy));
106 xx += 0.1*dx;
107 yy += 0.1*dy;
108 if (st.flags & STOP) {
109 b[0] = 0x80;
110 b[1] = 0x01;
111 b[2] = dx;
112 b[3] = dy;
113 fwrite(b, 1, 4, file);
114 } else if (st.flags & END) {
115 b[0] = 0x80;
116 b[1] = 0x10;
117 b[2] = 0;
118 b[3] = 0;
119 fwrite(b, 1, 4, file);
120 } else if ((st.flags & TRIM) || (st.flags & JUMP)) {
121 b[0] = 0x80;
122 b[1] = 2;
123 b[2] = dx;
124 b[3] = dy;
125 fwrite(b, 1, 4, file);
126 } else {
127 b[0] = dx;
128 b[1] = dy;
129 fwrite(b, 1, 2, file);
130 }
131 }
132 return 1;
133}
