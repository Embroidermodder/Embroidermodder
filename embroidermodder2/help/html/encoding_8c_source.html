71void
72reverse_byte_order(void *b, int bytes)
73{
74 char swap;
75 if (bytes == 2) {
76 swap = *((char*)b+0);
77 *((char*)b+0) = *((char*)b+1);
78 *((char*)b+1) = swap;
79 }
80 else {
81 swap = *((char*)b+0);
82 *((char*)b+0) = *((char*)b+3);
83 *((char*)b+3) = swap;
84 swap = *((char*)b+1);
85 *((char*)b+1) = *((char*)b+2);
86 *((char*)b+2) = swap;
87 }
88}

99int
100decode_t01_record(unsigned char b[3], int *x, int *y, int *flags)
101{
102 decode_tajima_ternary(b, x, y);
103
104 if (b[2] == 0xF3) {
105 *flags = END;
106 }
107 else {
108 switch (b[2] & 0xC3) {
109 case 0x03:
110 *flags = NORMAL;
111 break;
112 case 0x83:
113 *flags = TRIM;
114 break;
115 case 0xC3:
116 *flags = STOP;
117 break;
118 default:
119 *flags = NORMAL;
120 break;
121 }
122 }
123 return 1;
124}

134void
135encode_t01_record(unsigned char b[3], int x, int y, int flags)
136{
137 if (!encode_tajima_ternary(b, x, y)) {
138 return;
139 }
140
141 b[2] |= (unsigned char)3;
142 if (flags & END) {
143 b[0] = 0;
144 b[1] = 0;
145 b[2] = 0xF3;
146 }
147 if (flags & (JUMP | TRIM)) {
148 b[2] = (unsigned char)(b[2] | 0x83);
149 }
150 if (flags & STOP) {
151 b[2] = (unsigned char)(b[2] | 0xC3);
152 }
153}

163int
164encode_tajima_ternary(unsigned char b[3], int x, int y)
165{
166 b[0] = 0;
167 b[1] = 0;
168 b[2] = 0;
169
170 /* cannot encode values > +121 or < -121. */
171 if (x > 121 || x < -121) {
172 printf("ERROR: format-t01.c encode_record(), ");
173 printf("x is not in valid range [-121,121] , x = %d\n", x);
174 return 0;
175 }
176 if (y > 121 || y < -121) {
177 printf("ERROR: format-t01.c encode_record(), ");
178 printf("y is not in valid range [-121,121] , y = %d\n", y);
179 return 0;
180 }
181
182 if (x >= +41) {
183 b[2] |= 0x04;
184 x -= 81;
185 }
186 if (x <= -41) {
187 b[2] |= 0x08;
188 x += 81;
189 }
190 if (x >= +14) {
191 b[1] |= 0x04;
192 x -= 27;
193 }
194 if (x <= -14) {
195 b[1] |= 0x08;
196 x += 27;
197 }
198 if (x >= +5) {
199 b[0] |= 0x04;
200 x -= 9;
201 }
202 if (x <= -5) {
203 b[0] |= 0x08;
204 x += 9;
205 }
206 if (x >= +2) {
207 b[1] |= 0x01;
208 x -= 3;
209 }
210 if (x <= -2) {
211 b[1] |= 0x02;
212 x += 3;
213 }
214 if (x >= +1) {
215 b[0] |= 0x01;
216 x -= 1;
217 }
218 if (x <= -1) {
219 b[0] |= 0x02;
220 x += 1;
221 }
222 if (x != 0) {
223 printf("ERROR: format-dst.c encode_record(), ");
224 printf("x should be zero yet x = %d\n", x);
225 return 0;
226 }
227 if (y >= +41) {
228 b[2] |= 0x20;
229 y -= 81;
230 }
231 if (y <= -41) {
232 b[2] |= 0x10;
233 y += 81;
234 }
235 if (y >= +14) {
236 b[1] |= 0x20;
237 y -= 27;
238 }
239 if (y <= -14) {
240 b[1] |= 0x10;
241 y += 27;
242 }
243 if (y >= +5) {
244 b[0] |= 0x20;
245 y -= 9;
246 }
247 if (y <= -5) {
248 b[0] |= 0x10;
249 y += 9;
250 }
251 if (y >= +2) {
252 b[1] |= 0x80;
253 y -= 3;
254 }
255 if (y <= -2) {
256 b[1] |= 0x40;
257 y += 3;
258 }
259 if (y >= +1) {
260 b[0] |= 0x80;
261 y -= 1;
262 }
263 if (y <= -1) {
264 b[0] |= 0x40;
265 y += 1;
266 }
267 if (y != 0) {
268 printf("ERROR: format-dst.c encode_record(), ");
269 printf("y should be zero yet y = %d\n", y);
270 return 0;
271 }
272 return 1;
273}

284{
285 *x = 0;

288 *x += 1;

357void pfaffEncode(FILE* file, int dx, int dy, int flags)
358{
359 unsigned char flagsToWrite = 0;
360
361 if (!file) {
362 printf("ERROR: format-pcs.c pcsEncode(), file argument is null\n");
363 return;
364 }
365
366 write_24bit(file, dx);
367 write_24bit(file, dy);
368
369 if (flags & STOP)
370 {
371 flagsToWrite |= 0x01;
372 }
373 if (flags & TRIM)
374 {
375 flagsToWrite |= 0x04;
376 }
377 fwrite(&flagsToWrite, 1, 1, file);
378}

388EmbReal
389pfaffDecode(unsigned char a1, unsigned char a2, unsigned char a3)
390{
391 int res = a1 + (a2 << 8) + (a3 << 16);
392 if (res > 0x7FFFFF) {
393 return (-((~(res) & 0x7FFFFF) - 1));
394 }
395 return res;
396}

456void
457embInt_read(FILE* f, char *label, void *b, int mode)
458{
459 int endian = mode & 0x01;
460 int length = mode - endian;
461 fread(b, 1, length, f);
462 if (endian != ENDIAN_HOST) {
463 reverse_byte_order(b, length);
464 }
465
466 if (emb_verbose>1) {
467 switch (mode) {
468 case EMB_INT16_LITTLE:
469 printf("read int16_le %s: %d\n", label, *((short*)b));
470 break;
471 case EMB_INT16_BIG:
472 printf("read int16_be %s: %d\n", label, *((short*)b));
473 break;
474 case EMB_INT32_LITTLE:
475 printf("read int32_le %s: %d\n", label, *((int*)b));
476 break;
477 case EMB_INT32_BIG:
478 printf("read int32_be %s: %d\n", label, *((int*)b));
479 break;
480 default:
481 puts("ERROR: the mode supplied to fread_int is invalid.");
482 break;
483 }
484 }
