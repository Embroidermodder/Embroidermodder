444void
445set_style(char font_file[200])
446{
447 ImGuiIO& io = ImGui::GetIO();
448 ImFontConfig config;
449 config.OversampleH = 2;
450 config.OversampleV = 2;
451 FILE *f = fopen(font_file, "r");
452 if (f) {
453 fclose(f);
454 font = io.Fonts->AddFontFromFileTTF(font_file, 16, &config);
455 header_font = io.Fonts->AddFontFromFileTTF(font_file, 24, &config);
456 }
457 else {
458 log_debug("ERROR: Failed to load font file.");
459 }
460
461 ImGuiStyle& style = ImGui::GetStyle();
462 style.Colors[ImGuiCol_Text] = ImVec4(0.00f, 0.00f, 0.00f, 1.00f);
463 style.Colors[ImGuiCol_FrameBgActive] = ImVec4(0.75f, 0.75f, 0.75f, 1.00f);
464 style.Colors[ImGuiCol_PopupBg] = ImVec4(0.85f, 0.85f, 0.85f, 1.00f);
465 style.Colors[ImGuiCol_WindowBg] = ImVec4(0.65f, 0.65f, 0.65f, 1.00f);
466 style.Colors[ImGuiCol_MenuBarBg] = ImVec4(1.0f, 1.0f, 1.0f, 1.00f);
467 style.Colors[ImGuiCol_ChildBg] = ImVec4(1.0f, 1.0f, 1.0f, 1.00f);
468}

530int
531load_textures(Dictionary *texture_list)
532{
533 for (int i=0; i<texture_list->length; i++) {
534 DictionaryEntry d = texture_list->data[i];
535 textures[d.key] = load_texture(d.value);
536 if (!textures[d.key]) {
537 char error[200];
538 sprintf(error, "ERROR: Failed to open %s.", d.value);
539 log_debug(error);
540 /* return 0; */
541 }
542 }
543
544 return 1;
545}
