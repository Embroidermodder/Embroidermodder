89QPointF DimLeaderObject::objectEndPoint2() const
90{
91 QLineF lyne = line();
92 qreal rot = radians(rotation());
93 qreal cosRot = qCos(rot);
94 qreal sinRot = qSin(rot);
95 qreal x2 = lyne.x2()*scale();
96 qreal y2 = lyne.y2()*scale();
97 qreal rotEnd2X = x2*cosRot - y2*sinRot;
98 qreal rotEnd2Y = x2*sinRot + y2*cosRot;
99
100 return (scenePos() + QPointF(rotEnd2X, rotEnd2Y));

126void DimLeaderObject::updateLeader()
127{
128 int arrowStyle = Closed; //TODO: Make this customizable
129 qreal arrowStyleAngle = 15.0; //TODO: Make this customizable
130 qreal arrowStyleLength = 1.0; //TODO: Make this customizable
131 qreal lineStyleAngle = 45.0; //TODO: Make this customizable
132 qreal lineStyleLength = 1.0; //TODO: Make this customizable
133
134 QLineF lyne = line();
135 qreal angle = lyne.angle();
136 QPointF ap0 = lyne.p1();
137 QPointF lp0 = lyne.p2();
138
139 //Arrow
140 QLineF lynePerp(lyne.pointAt(arrowStyleLength/lyne.length()) ,lp0);
141 lynePerp.setAngle(angle + 90);
142 QLineF lyne1(ap0, lp0);
143 QLineF lyne2(ap0, lp0);
144 lyne1.setAngle(angle + arrowStyleAngle);
145 lyne2.setAngle(angle - arrowStyleAngle);
146 QPointF ap1;
147 QPointF ap2;
148 lynePerp.intersects(lyne1, &ap1);
149 lynePerp.intersects(lyne2, &ap2);
150
151 //Math Diagram
152 // .(ap1) .(lp1)
153 // /| /|
154 // / | / |
155 // / | / |
156 // / | / |
157 // / | / |
158 // / | / |
159 // / | / |
160 // / | / |
161 // /+(aSA) | /+(lSA) |
162 // (ap0)./__(aSL)__|__________(lp0)./__(lSL)__|
163 // \ -(aSA) | \ -(lSA) |
164 // \ | \ |
165 // \ | \ |
166 // \ | \ |
167 // \ | \ |
168 // \ | \ |
169 // \ | \ |
170 // \ | \ |
171 // \ | \ |
172 // \| \|
173 // .(ap2) .(lp2)
174
175 if(arrowStyle == Open)
176 {
177 arrowStylePath = QPainterPath();
178 arrowStylePath.moveTo(ap1);
179 arrowStylePath.lineTo(ap0);
180 arrowStylePath.lineTo(ap2);
181 arrowStylePath.lineTo(ap0);
182 arrowStylePath.lineTo(ap1);
183 }
184 else if(arrowStyle == Closed)
185 {
186 arrowStylePath = QPainterPath();
187 arrowStylePath.moveTo(ap1);
188 arrowStylePath.lineTo(ap0);
189 arrowStylePath.lineTo(ap2);
190 arrowStylePath.lineTo(ap1);
191 }
192 else if(arrowStyle == Dot)
193 {
194 arrowStylePath = QPainterPath();
195 arrowStylePath.addEllipse(ap0, arrowStyleLength, arrowStyleLength);
196 }
197 else if(arrowStyle == Box)
198 {
199 arrowStylePath = QPainterPath();
200 qreal side = QLineF(ap1, ap2).length();
201 QRectF ar0(0, 0, side, side);
202 ar0.moveCenter(ap0);
203 arrowStylePath.addRect(ar0);
204 }
205 else if(arrowStyle == Tick)
206 {
207 }
208
209 lineStylePath = QPainterPath();
210 lineStylePath.moveTo(ap0);
211 lineStylePath.lineTo(lp0);
212}

233void DimLeaderObject::updateRubber(QPainter* painter)
234{
235 int rubberMode = objectRubberMode();
236 if(rubberMode == OBJ_RUBBER_DIMLEADER_LINE)
237 {
238 QPointF sceneStartPoint = objectRubberPoint("DIMLEADER_LINE_START");
239 QPointF sceneQSnapPoint = objectRubberPoint("DIMLEADER_LINE_END");
240
241 setObjectEndPoint1(sceneStartPoint);
242 setObjectEndPoint2(sceneQSnapPoint);
243 }
244 else if(rubberMode == OBJ_RUBBER_GRIP)
245 {
246 if(painter)
247 {
248 QPointF gripPoint = objectRubberPoint("GRIP_POINT");
249 if (gripPoint == objectEndPoint1()) painter->drawLine(line().p2(), mapFromScene(objectRubberPoint(QString())));
250 else if(gripPoint == objectEndPoint2()) painter->drawLine(line().p1(), mapFromScene(objectRubberPoint(QString())));
251 else if(gripPoint == objectMidPoint()) painter->drawLine(line().translated(mapFromScene(objectRubberPoint(QString()))-mapFromScene(gripPoint)));
252 }
253 }

264// Returns the closest snap point to the mouse point
265QPointF DimLeaderObject::mouseSnapPoint(const QPointF& mousePoint)
266{
267 QPointF endPoint1 = objectEndPoint1();
268 QPointF endPoint2 = objectEndPoint2();
269 QPointF midPoint = objectMidPoint();
270
271 qreal end1Dist = QLineF(mousePoint, endPoint1).length();
272 qreal end2Dist = QLineF(mousePoint, endPoint2).length();
273 qreal midDist = QLineF(mousePoint, midPoint).length();
274
275 qreal minDist = qMin(end1Dist, end2Dist);
276
277 if(curved)
278 minDist = qMin(minDist, midDist);
279
280 if (minDist == end1Dist) return endPoint1;
281 else if(minDist == end2Dist) return endPoint2;
282 else if(minDist == midDist) return midPoint;
283
284 return scenePos();
285}
