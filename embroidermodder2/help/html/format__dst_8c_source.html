24int decode_record_flags(unsigned char b2)
25{
26 if (b2 == 0xF3) {
27 return END;
28 }
29 if (b2 & 0x40) {
30 return STOP;
31 }
32 if (b2 & 0x80) {
33 return JUMP;
34 }
35 return NORMAL;
36}

86void
87encode_record(FILE* file, int x, int y, int flags)
88{
89 unsigned char b[3];
90 encode_tajima_ternary(b, x, y);
91
92 b[2] |= 0x03;
93
94 if (flags & END) {
95 /* 0b10001101 = -13 = 0x8D */
96 b[0] = 0;
97 b[1] = 0;
98 b[2] = 0x8D;
99 }
100
101 /* make the variable v configurable for the pattern */
102 if (flags & TRIM) {
103 int dx, dy, i, v = 5;
104 dx = x/v;
105 dy = y/v;
106 for (i = 1; i < v; i++) {
107 encode_record(file, dx, dy, JUMP);
108 }
109 encode_record(file, x - (dx * (v - 1)), y - (dy * (v - 1)), JUMP);
110 return;
111 }
112 if (flags & (JUMP | TRIM)) {
113 b[2] = (char) (b[2] | 0x83);
114 }
115 if (flags & STOP) {
116 b[2] = (char) (b[2] | 0xC3);
117 }
118
119 fwrite(b, 1, 3, file);
120}

189/*
190 * The header seems to contain information about the design.
191 * Seems to be ASCII text delimited by 0x0D (carriage returns).
192 * This must be in the file for most new software or hardware
193 * to consider it a good file! This is much more important
194 * than I originally believed. The header is 125 bytes in
195 * length and padded out by 0x20 to 512 bytes total.
196 * All entries in the header seem to be 2 ASCII characters
197 * followed by a colon, then it's value trailed by a carriage return.
198 *
199 * char LA[16+1]; First is the 'LA' entry, which is the design name with no
200 * path or extension information. The blank is 16 characters
201 * in total, but the name must not be longer that 8 characters
202 * and padded out with 0x20.
203 *
204 * char ST[7+1]; Next is the stitch count ST, this is a 7 digit number
205 * padded by leading zeros. This is the total stitch count
206 * including color changes, jumps, nups, and special records.
207 *
208 * char CO[3+1]; Next, is CO or colors, a 3 digit number padded by leading
209 * zeros. This is the number of color change records in the file.
210 *
211 * char POSX[5+1]; Next is +X or the positive X extent in centimeters, a 5
212 * digit non-decimal number padded by leading zeros.
213 *
214 * char NEGX[5+1]; Following is the -X or the negative X extent in millimeters,
215 * a 5 digit non-decimal number padded by leading zeros.
216 *
217 * char POSY[5+1]; Again, the +Y extents.
218 *
219 * char NEGY[5+1]; Again, the -Y extents.
220 *
221 * char AX[6+1]; AX and AY should express the relative coordinates of the
222 * char AY[6+1]; last point from the start point in 0.1 mm. If the start
223 * and last points are the same, the coordinates are (0,0).
224 *
225 * char MX[6+1]; MX and MY should express coordinates of the last point of
226 * char MY[6+1]; the previous file for a multi-volume design. A multi-
227 * volume design means a design consisted of two or more files.
228 * This was used for huge designs that can not be stored in a
229 * single paper tape roll. It is not used so much (almost
230 * never) nowadays.
231 *
232 * char PD[9+1]; PD is also storing some information for multi-volume design.
233 */
234char
235readDst(EmbPattern* pattern, FILE* file) {
236 char var[3]; /* temporary storage variable name */
237 char val[512]; /* temporary storage variable value */
238 int valpos;
239 unsigned char b[3];
240 char header[512 + 1];
241 int i = 0, flags;
242
243 /* TODO: review commented code below
244 pattern->clear();
245 pattern->set_variable("file_name",filename);
246 */
247
248 fread(header, 1, 512, file);
249
250 /*TODO:It would probably be a good idea to
251 validate file before accepting it. */
252 /* fill variables from header fields */
253 for (i = 0; i < 512; i++) {
254 if (header[i] == ':' && i > 1) {
255 var[0] = header[i - 2];
256 var[1] = header[i - 1];
257 var[2] = '\0';
258 valpos = i + 1;
259 for (i++; i < 512; i++)
260 {
261 /* don't accept : without CR because there's a bug below: i-valpos must be > 0 which is not the case if the : is before the third character. */
262 if (header[i] == 13/*||header[i]==':'*/) { /* 0x0d = carriage return */
263 /* : indicates another variable, CR was missing! */
264 if (header[i] == ':') {
265 i -= 2;
266 }
267 strncpy(val, &header[valpos], (size_t)(i - valpos));
268 val[i - valpos] = '\0';
269 set_dst_variable(pattern, var, val);
270 break;
271 }
272 }
273 }
274 }
275
276 while (fread(b, 1, 3, file) == 3) {
277 int x, y;
278 decode_tajima_ternary(b, &x, &y);
279 flags = decode_record_flags(b[2]);
280 if (flags == END) {
281 break;
282 }
283 embPattern_addStitchRel(pattern, x / 10.0, y / 10.0, flags, 1);
284 }
285
286 /* combineJumpStitches(pattern, 5); */
287 return 1;
288}
