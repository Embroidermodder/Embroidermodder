82QPointF LineObject::objectEndPoint2() const
83{
84 QLineF lyne = line();
85 qreal rot = radians(rotation());
86 qreal cosRot = qCos(rot);
87 qreal sinRot = qSin(rot);
88 qreal x2 = lyne.x2()*scale();
89 qreal y2 = lyne.y2()*scale();
90 qreal rotEnd2X = x2*cosRot - y2*sinRot;
91 qreal rotEnd2Y = x2*sinRot + y2*cosRot;
92
93 return (scenePos() + QPointF(rotEnd2X, rotEnd2Y));

136void LineObject::updateRubber(QPainter* painter)
137{
138 int rubberMode = objectRubberMode();
139 if(rubberMode == OBJ_RUBBER_LINE)
140 {
141 QPointF sceneStartPoint = objectRubberPoint("LINE_START");
142 QPointF sceneQSnapPoint = objectRubberPoint("LINE_END");
143
144 setObjectEndPoint1(sceneStartPoint);
145 setObjectEndPoint2(sceneQSnapPoint);
146
147 drawRubberLine(line(), painter, VIEW_COLOR_CROSSHAIR);
148 }
149 else if(rubberMode == OBJ_RUBBER_GRIP)
150 {
151 if(painter)
152 {
153 QPointF gripPoint = objectRubberPoint("GRIP_POINT");
154 if (gripPoint == objectEndPoint1()) painter->drawLine(line().p2(), mapFromScene(objectRubberPoint(QString())));
155 else if(gripPoint == objectEndPoint2()) painter->drawLine(line().p1(), mapFromScene(objectRubberPoint(QString())));
156 else if(gripPoint == objectMidPoint()) painter->drawLine(line().translated(mapFromScene(objectRubberPoint(QString()))-mapFromScene(gripPoint)));
157
158 QLineF rubLine(mapFromScene(gripPoint), mapFromScene(objectRubberPoint(QString())));
159 drawRubberLine(rubLine, painter, VIEW_COLOR_CROSSHAIR);
160 }
161 }

172// Returns the closest snap point to the mouse point
173QPointF LineObject::mouseSnapPoint(const QPointF& mousePoint)
174{
175 QPointF endPoint1 = objectEndPoint1();
176 QPointF endPoint2 = objectEndPoint2();
177 QPointF midPoint = objectMidPoint();
178
179 qreal end1Dist = QLineF(mousePoint, endPoint1).length();
180 qreal end2Dist = QLineF(mousePoint, endPoint2).length();
181 qreal midDist = QLineF(mousePoint, midPoint).length();
182
183 qreal minDist = qMin(qMin(end1Dist, end2Dist), midDist);
184
185 if (minDist == end1Dist) return endPoint1;
186 else if(minDist == end2Dist) return endPoint2;
187 else if(minDist == midDist) return midPoint;
188
189 return scenePos();
190}
