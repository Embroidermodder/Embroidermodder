34 /* READ STITCH RECORDS */
35 for (i = 0; i < st; i++) {
36 int flags;
37 unsigned char b[9];

39 if (fread(b, 1, 9, file) != 9) {
40 break;
41 }
42 if (b[8] & 0x01) {
43 flags = STOP;
44 } else if (b[8] & 0x04) {
45 flags = TRIM;
46 } else if (b[8] != 0) {
47 /* TODO: ONLY INTERESTED IN THIS CASE TO LEARN MORE ABOUT THE FORMAT */
48 }
49 dx = pfaffDecode(b[2], b[1], b[0]);
50 dy = pfaffDecode(b[6], b[5], b[4]);
51 embPattern_addStitchAbs(pattern, dx / 10.0, dy / 10.0, flags, 1);
52 }
53 return 1;
54}
