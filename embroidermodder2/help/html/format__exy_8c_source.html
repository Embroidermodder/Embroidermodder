14int decode_exy_flags(unsigned char b2)
15{
16 int returnCode = 0;
17 if (b2 == 0xF3) {
18 return (END);
19 }
20 if ((b2 & 0xC3) == 0xC3) {
21 return TRIM | STOP;
22 }
23 if (b2 & 0x80) {
24 returnCode |= TRIM;
25 }
26 if (b2 & 0x40) {
27 returnCode |= STOP;
28 }
29 return returnCode;
30}
