14char
15read10o(EmbPattern* pattern, FILE* file)
16{
17 unsigned char b[3];
18 while (fread(b, 1, 3, file) == 3) {
19 int x, y;
20 int stitchType = NORMAL;
21 unsigned char ctrl = b[0];
22 y = b[1];
23 x = b[2];
24 if (ctrl & 0x20) {
25 x = -x;
26 }
27 if (ctrl & 0x40) {
28 y = -y;
29 }
30 if (ctrl & 0x01) {
31 stitchType = TRIM;
32 }
33 if ((ctrl & 0x5) == 5) {
34 stitchType = STOP;
35 }
36 if (ctrl == 0xF8 || ctrl == 0x91 || ctrl == 0x87) {
37 embPattern_addStitchRel(pattern, 0, 0, END, 1);
38 break;
39 }
40 embPattern_addStitchRel(pattern, x / 10.0, y / 10.0, stitchType, 1);
41 }
42 return 1;
43}

45char
46write10o(EmbPattern* pattern, FILE* file)
47{
48 int i;
49 for (i=0; i<pattern->stitch_list->count; i++) {
50 unsigned char b[3];
51 EmbStitch st = pattern->stitch_list->stitch[i];
52 b[0] = 0;
53 b[1] = 0;
54 b[2] = 0;
55 if (st.x < 0) {
56 b[2] |= 0x20;
57 b[0] = -st.x;
58 }
59 else {
60 b[0] = st.x;
61 }
62 if (st.y < 0) {
63 b[2] |= 0x40;
64 b[1] = -st.y;
65 }
66 else {
67 b[1] = st.y;
68 }
69 if (st.flags == TRIM) {
70 b[2] |= 1;
71 }
72 if (st.flags == STOP) {
73 b[2] |= 5;
74 }
75 if (st.flags == END) {
76 b[2] = 0xF8;
77 }
78 fwrite(b, 1, 3, file);
79 }
80 return 1;
81}
