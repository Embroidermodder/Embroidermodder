24EmbArray*
25embArray_create(int type)
26{
27 EmbArray *a;
28 a = (EmbArray*)malloc(sizeof(EmbArray));
29 a->type = type;
30 a->length = CHUNK_SIZE;
31 a->count = 0;
32 switch (type) {
33 case EMB_STITCH:
34 a->stitch = (EmbStitch*)malloc(CHUNK_SIZE*sizeof(EmbStitch));
35 break;
36 case EMB_THREAD:
37 a->thread = (EmbThread*)malloc(CHUNK_SIZE*sizeof(EmbThread));
38 break;
39 default:
40 a->geometry = (EmbGeometry*)malloc(CHUNK_SIZE*sizeof(EmbGeometry));
41 break;
42 }
43 return a;
44}

52int
53embArray_resize(EmbArray *a)
54{
55 if (a->count < a->length) {
56 return 1;
57 }
58 a->length += CHUNK_SIZE;
59 switch (a->type) {
60 case EMB_STITCH:
61 a->stitch = (EmbStitch*)realloc(a->stitch, a->length*sizeof(EmbStitch));
62 if (!a->stitch) {
63 /* TODO: Error reporting */
64 return 0;
65 }
66 break;
67 case EMB_THREAD:
68 a->thread = (EmbThread*)realloc(a->thread, a->length*sizeof(EmbThread));
69 if (!a->thread) {
70 /* TODO: Error reporting */
71 return 0;
72 }
73 break;
74 default:
75 a->geometry = (EmbGeometry *)realloc(a->geometry, a->length*sizeof(EmbGeometry));
76 if (!a->geometry) {
77 /* TODO: Error reporting */
78 return 0;
79 }
80 break;
81 }
82
83 return 1;
84}

92void
93embArray_copy(EmbArray *dst, EmbArray *src)
94{
95 dst = embArray_create(src->type);
96 dst->length = src->length;
97 dst->count = src->count;
98 embArray_resize(dst);
99 /* BUG: Potential failure to copy path memory, only copy pointers? */
100
101 switch (dst->type) {
102 case EMB_STITCH:
103 memcpy(dst->stitch, src->stitch, sizeof(EmbStitch)*src->count);
104 break;
105 case EMB_THREAD:
106 memcpy(dst->thread, src->thread, sizeof(EmbThread)*src->count);
107 break;
108 default:
109 memcpy(dst->geometry, src->geometry, sizeof(EmbGeometry)*src->count);
110 break;
111 }
112}

346void
347embArray_free(EmbArray* a)
348{
349 int i;
350 if (!a) {
351 return;
352 }
353 switch (a->type) {
354 case EMB_STITCH:
355 free(a->stitch);
356 break;
357 case EMB_THREAD:
358 free(a->thread);
359 break;
360 default:
361 for (i = 0; i < a->count; i++) {
362 EmbGeometry g = a->geometry[i];
363 switch (a->geometry[i].type) {
364 case EMB_PATH: {
365 embArray_free(g.object.path.pointList);
366 break;
367 }
368 case EMB_POLYGON: {
369 embArray_free(g.object.polygon.pointList);
370 break;
371 }
372 case EMB_POLYLINE: {
373 embArray_free(g.object.polyline.pointList);
374 break;
375 }
376 default:
377 break;
378 }
379 }
380 free(a->geometry);
381 break;
382 }
383 free(a);
384}
