302QString CmdPromptHistory::applyFormatting(const QString& txt, int prefixLength)
303{
304 QString prefix = txt.left(prefixLength);
305 QString usrtxt = txt.right(txt.length()-prefixLength);
306
307 int start = -1;
308 int stop = -1;
309
310 //Bold Prefix
311 prefix.prepend("<b>");
312 prefix.append("</b>");
313
314 //Keywords
315 start = prefix.indexOf('[');
316 stop = prefix.lastIndexOf(']');
317 if(start != -1 && stop != -1 && start < stop)
318 {
319 for(int i = stop; i >= start; i--)
320 {
321 if(prefix.at(i) == ']')
322 prefix.insert(i, "</font>");
323 if(prefix.at(i) == '[')
324 prefix.insert(i+1, "<font color=\"#0095FF\">");
325 if(prefix.at(i) == '/')
326 {
327 prefix.insert(i+1, "<font color=\"#0095FF\">");
328 prefix.insert(i, "</font>");
329 }
330 }
331 }
332
333 //Default Values
334 start = prefix.indexOf('{');
335 stop = prefix.lastIndexOf('}');
336 if(start != -1 && stop != -1 && start < stop)
337 {
338 for(int i = stop; i >= start; i--)
339 {
340 if(prefix.at(i) == '}')
341 prefix.insert(i, "</font>");
342 if(prefix.at(i) == '{')
343 prefix.insert(i+1, "<font color=\"#00AA00\">");
344 }
345 }
346
347 return prefix + usrtxt;
348}

445void CmdPromptInput::processInput(const QChar& rapidChar)
446{
447 qDebug("CmdPromptInput::processInput");
448
449 updateCurrentText(curText);
450
451 QString cmdtxt(curText);
452 cmdtxt.replace(0, prefix.length(), "");
453 if(!rapidFireEnabled) cmdtxt = cmdtxt.toLower();
454
455 if(cmdActive)
456 {
457 if(rapidFireEnabled)
458 {
481 }
482 else
483 {
484 emit appendHistory(curText, prefix.length());
485 emit runCommand(curCmd, cmdtxt);
486 }
487 }
488 else
489 {
490 if(aliasHash->contains(cmdtxt))
491 {
492 cmdActive = true;
493 lastCmd = curCmd;
494 curCmd = aliasHash->value(cmdtxt);
495 emit appendHistory(curText, prefix.length());
496 emit startCommand(curCmd);
497 }
498 else if(cmdtxt.isEmpty())
499 {
500 cmdActive = true;
501 emit appendHistory(curText, prefix.length());
502 //Rerun the last successful command
503 emit startCommand(lastCmd);
504 }
505 else
506 {
507 emit appendHistory(curText + "<br/><font color=\"red\">Unknown command \"" + cmdtxt + "\". Press F1 for help.</font>", prefix.length());
508 }
509 }
510
511 if(!rapidFireEnabled)
512 clear();
513}

551void CmdPromptInput::applyFormatting()
552{
553 int prefixLength = prefix.length();
554
555 int start = -1;
556 int stop = -1;
557
558 QList<QTextLayout::FormatRange> formats;
559
560 //Bold Prefix
561 QTextCharFormat formatPrefix;
562 formatPrefix.setFontWeight(QFont::Bold);
563 QTextLayout::FormatRange rangePrefix;
564 rangePrefix.start = 0;
565 rangePrefix.length = prefixLength;
566 rangePrefix.format = formatPrefix;
567 formats.append(rangePrefix);
568
569 //Keywords
570 start = prefix.indexOf('[');
571 stop = prefix.lastIndexOf(']');
572 if(start != -1 && stop != -1 && start < stop)
573 {
574 QTextCharFormat formatKeyword;
575 formatKeyword.setFontWeight(QFont::Bold);
576 formatKeyword.setForeground(QColor("#0095FF"));
577
578 int rangeStart = -1;
579 int rangeStop = -1;
580 for(int i = stop; i >= start; i--)
581 {
582 if(prefix.at(i) == ']')
583 {
584 rangeStop = i;
585 }
586 if(prefix.at(i) == '[' || prefix.at(i) == '/')
587 {
588 rangeStart = i;
589
590 QTextLayout::FormatRange rangeKeyword;
591 rangeKeyword.start = rangeStart+1;
592 rangeKeyword.length = rangeStop-rangeStart-1;
593 rangeKeyword.format = formatKeyword;
594 formats.append(rangeKeyword);
595
596 rangeStop = i;
597 }
598 }
599 }
600
601 //Default Values
602 start = prefix.indexOf('{');
603 stop = prefix.lastIndexOf('}');
604 if(start != -1 && stop != -1 && start < stop)
605 {
606 QTextCharFormat formatKeyword;
607 formatKeyword.setFontWeight(QFont::Bold);
608 formatKeyword.setForeground(QColor("#00AA00"));
609
610 int rangeStart = -1;
611 int rangeStop = -1;
612 for(int i = stop; i >= start; i--)
613 {
614 if(prefix.at(i) == '}')
615 {
616 rangeStop = i;
617 }
618 if(prefix.at(i) == '{')
619 {
620 rangeStart = i;
621
622 QTextLayout::FormatRange rangeKeyword;
623 rangeKeyword.start = rangeStart+1;
624 rangeKeyword.length = rangeStop-rangeStart-1;
625 rangeKeyword.format = formatKeyword;
626 formats.append(rangeKeyword);
627
628 rangeStop = i;
629 }
630 }
631 }
632
633 changeFormatting(formats);
634}

707bool CmdPromptInput::eventFilter(QObject* obj, QEvent* event)
708{
709 if(event->type() == QEvent::KeyPress)
710 {
711 if(isBlinking) emit stopBlinking();
712
713 QKeyEvent* pressedKey = (QKeyEvent*)event;
714
715 //NOTE: These shortcuts need to be caught since QLineEdit uses them
716 if(pressedKey->matches(QKeySequence::Cut))
717 {
718 pressedKey->accept();
719 emit cutPressed();
720 return true;
721 }
722 else if(pressedKey->matches(QKeySequence::Copy))
723 {
724 pressedKey->accept();
725 emit copyPressed();
726 return true;
727 }
728 else if(pressedKey->matches(QKeySequence::Paste))
729 {
730 pressedKey->accept();
731 emit pastePressed();
732 return true;
733 }
734 else if(pressedKey->matches(QKeySequence::SelectAll))
735 {
736 pressedKey->accept();
737 emit selectAllPressed();
738 return true;
739 }
740 else if(pressedKey->matches(QKeySequence::Undo))
741 {
742 pressedKey->accept();
743 emit undoPressed();
744 return true;
745 }
746 else if(pressedKey->matches(QKeySequence::Redo))
747 {
748 pressedKey->accept();
749 emit redoPressed();
750 return true;
751 }
752
753 int key = pressedKey->key();
754 switch(key)
755 {
756 case Qt::Key_Enter:
757 case Qt::Key_Return:
758 case Qt::Key_Space:
759 pressedKey->accept();
760 processInput(QChar(key));
761 return true;
762 break;
763 case Qt::Key_Delete:
764 pressedKey->accept();
765 del();
766 emit deletePressed();
767 return true;
768 break;
769 case Qt::Key_Tab:
770 pressedKey->accept();
771 emit tabPressed();
772 return true;
773 break;
774 case Qt::Key_Escape:
775 pressedKey->accept();
776 prefix = defaultPrefix;
777 clear();
778 emit appendHistory(curText + tr("*Cancel*"), prefix.length());
779 emit escapePressed();
780 return true;
781 break;
782 case Qt::Key_Up:
783 pressedKey->accept();
784 emit upPressed();
785 return true;
786 break;
787 case Qt::Key_Down:
788 pressedKey->accept();
789 emit downPressed();
790 return true;
791 break;
792 case Qt::Key_F1:
793 pressedKey->accept();
794 emit F1Pressed();
795 return true;
796 break;
797 case Qt::Key_F2:
798 pressedKey->accept();
799 emit F2Pressed();
800 return true;
801 break;
802 case Qt::Key_F3:
803 pressedKey->accept();
804 emit F3Pressed();
805 return true;
806 break;
807 case Qt::Key_F4:
808 pressedKey->accept();
809 emit F4Pressed();
810 return true;
811 break;
812 case Qt::Key_F5:
813 emit F5Pressed();
814 pressedKey->accept();
815 return true;
816 break;
817 case Qt::Key_F6:
818 pressedKey->accept();
819 emit F6Pressed();
820 return true;
821 break;
822 case Qt::Key_F7:
