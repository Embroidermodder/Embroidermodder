16char* csvStitchFlagToStr(int flags)
17{
18 switch (flags) {
19 case NORMAL:
20 return "STITCH";
21 break;
22 case JUMP:
23 return "JUMP";
24 break;
25 case TRIM:
26 return "TRIM";
27 break;
28 case STOP:
29 return "COLOR";
30 break;
31 case END:
32 return "END";
33 break;
34 default:
35 return "UNKNOWN";
36 break;
37 }
38}

40int csvStrToStitchFlag(const char* str)
41{
42 if (!str) {
43 printf("ERROR: format-csv.c csvStrToStitchFlag(), str argument is null\n");
44 return -1;
45 }
46 if (!strcmp(str, "STITCH")) {
47 return NORMAL;
48 } else if (!strcmp(str, "JUMP")) {
49 return JUMP;
50 } else if (!strcmp(str, "TRIM")) {
51 return TRIM;
52 } else if (!strcmp(str, "COLOR")) {
53 return STOP;
54 } else if (!strcmp(str, "END")) {
55 return END;
56 } else if (!strcmp(str, "UNKNOWN")) {
57 return STOP;
58 }
59 return -1;
60}

62char
63readCsv(EmbPattern* pattern, FILE* file) {
64 int numColorChanges = 0;
65 int pos = 0;
66 int c = 0;
67 int cellNum = 0;
68 int process = 0;
69 int csvMode = CSV_MODE_NULL;
70 int expect = CSV_EXPECT_QUOTE1;
71 int flags = 0;
72 EmbReal xx = 0.0;
73 EmbReal yy = 0.0;
74 unsigned char r = 0, g = 0, b = 0;
75 char buff[100];
76
77 pos = 0;
78 while (!feof(file)) {
79 c = fgetc(file);
80 switch(c) {
81 case '"':
82 if (expect == CSV_EXPECT_QUOTE1) {
83 expect = CSV_EXPECT_QUOTE2;
84 } else if (expect == CSV_EXPECT_QUOTE2) {
85 expect = CSV_EXPECT_COMMA;
86 }
87 break;
88 case ',':
89 if (expect == CSV_EXPECT_COMMA) {
90 process = 1;
91 }
92 break;
93 case '\n':
94 if (expect == CSV_EXPECT_COMMA) {
95 process = 1;
96 } else if (expect == CSV_EXPECT_QUOTE1) {
97 /* Do Nothing. We encountered a blank line. */
98 } else {
99 printf("ERROR: format-csv.c readCsv(), premature newline\n");
100 return 0;
101 }
102 break;
103 default:
104 break;
105 }
106
107 if (process) {
108 buff[pos] = 0;
109 pos = 0;
110 process = 0;
111 cellNum++;
112 expect = CSV_EXPECT_QUOTE1;
113 if (csvMode == CSV_MODE_NULL) {
114 if (!strcmp(buff, "#")) {
115 csvMode = CSV_MODE_COMMENT;
116 } else if (!strcmp(buff, ">")) {
117 csvMode = CSV_MODE_VARIABLE;
118 } else if (!strcmp(buff, "$")) {
119 csvMode = CSV_MODE_THREAD;
120 } else if (!strcmp(buff, "*")) {
121 csvMode = CSV_MODE_STITCH;
122 } else {/* TODO: error */
123 return 0;
124 }
125 } else if (csvMode == CSV_MODE_COMMENT) {
126 /* Do Nothing */
127 } else if (csvMode == CSV_MODE_VARIABLE) {
128 /* Do Nothing */
129 } else if (csvMode == CSV_MODE_THREAD) {
130 if (cellNum == 2) {
131 /* Do Nothing. Ignore Thread Number */
132 } else if (cellNum == 3) {
133 r = (unsigned char)atoi(buff);
134 } else if (cellNum == 4) {
135 g = (unsigned char)atoi(buff);
136 } else if (cellNum == 5) {
137 b = (unsigned char)atoi(buff);
138 } else if (cellNum == 6) {
139 /* TODO: Thread Description */
140 } else if (cellNum == 7) {
141 /* TODO: Thread Catalog Number */
142 EmbThread t;
143 t.color.r = r;
144 t.color.g = g;
145 t.color.b = b;
146 strcpy(t.description, "TODO:DESCRIPTION");
147 strcpy(t.catalogNumber, "TODO:CATALOG_NUMBER");
148 embPattern_addThread(pattern, t);
149 csvMode = CSV_MODE_NULL;
150 cellNum = 0;
151 } else {
152 /* TODO: error */
153 return 0;
154 }
155 } else if (csvMode == CSV_MODE_STITCH) {
156 /* printf("%s\n", buff); */
157 if (cellNum == 2) {
158 flags = csvStrToStitchFlag(buff);
159 if (flags == STOP) {
160 numColorChanges++;
161 }
162 } else if (cellNum == 3) {
163 xx = atof(buff);
164 } else if (cellNum == 4) {
165 yy = atof(buff);
166 /* printf("%d %d %f %f\n",
167 pattern->stitch_list->count, flags, xx, yy); */
168 embPattern_addStitchAbs(pattern, xx, yy, flags, 1);
169 csvMode = CSV_MODE_NULL;
170 cellNum = 0;
171 } else {
172 /* TODO: error */
173 return 0;
174 }
175 }
176 if (c == '\n') {
177 csvMode = CSV_MODE_NULL;
178 cellNum = 0;
179 }
180 } else {
181 if (expect == CSV_EXPECT_QUOTE2 && c != '"') {
182 buff[pos++] = (char)c;
183 }
184 }
185 }
186
187 return 1;
188}

190char
191writeCsv(EmbPattern* pattern, FILE* file) {
192 EmbRect boundingRect;
193 int i;
194
195 boundingRect = embPattern_calcBoundingBox(pattern);
196
197#if LIBEMBROIDERY_EMBEDDED_VERSION
198 /* write header */
199 FILE *csv_header;
200 csv_header = fopen("assets/header.csv", "r");
201 while (!feof(csv_header)) {
202 char c;
203 fread(&c, 1, 1, csv_header);
204 fwrite(&c, 1, 1, file);
205 }
206 fclose(csv_header);
207#else
208 fprintf(file, "\"#\",\"Embroidermodder 2 CSV Embroidery File\"\n");
209 fprintf(file, "\"#\",\"http://embroidermodder.github.io\"\n");
210 fprintf(file, "\"#\",\" \"\n");
211 fprintf(file, "\"#\",\"General Notes:\"\n");
212 fprintf(file, "\"#\",\"This file can be read by Excel or LibreOffice as CSV (Comma Separated Value) or with a text editor.\"\n");
213 fprintf(file, "\"#\",\"Lines beginning with # are comments.\"\n");
214 fprintf(file, "\"#\",\"Lines beginning with > are variables: [VAR_NAME], [VAR_VALUE]\"\n");
215 fprintf(file, "\"#\",\"Lines beginning with $ are threads: [THREAD_NUMBER], [RED], [GREEN], [BLUE], [DESCRIPTION], [CATALOG_NUMBER]\"\n");
216 fprintf(file, "\"#\",\"Lines beginning with * are stitch entries: [STITCH_TYPE], [X], [Y]\"\n");
217 fprintf(file, "\"#\",\" \"\n");
218 fprintf(file, "\"#\",\"Stitch Entry Notes:\"\n");
219 fprintf(file, "\"#\",\"STITCH instructs the machine to move to the position [X][Y] and then make a stitch.\"\n");
220 fprintf(file, "\"#\",\"JUMP instructs the machine to move to the position [X][Y] without making a stitch.\"\n");
221 fprintf(file, "\"#\",\"TRIM instructs the machine to cut the thread before moving to the position [X][Y] without making a stitch.\"\n");
222 fprintf(file, "\"#\",\"COLOR instructs the machine to stop temporarily so that the user can change to a different color thread before resuming.\"\n");
223 fprintf(file, "\"#\",\"END instructs the machine that the design is completed and there are no further instructions.\"\n");
224 fprintf(file, "\"#\",\"UNKNOWN encompasses instructions that may not be supported currently.\"\n");
225 fprintf(file, "\"#\",\"[X] and [Y] are absolute coordinates in millimeters (mm).\"\n");
226 fprintf(file, "\"#\",\" \"\n");
227#endif
228
229 /* write variables */
230 fprintf(file,"\"#\",\"[VAR_NAME]\",\"[VAR_VALUE]\"\n");
231 fprintf(file, "\">\",\"STITCH_COUNT:\",\"%u\"\n", (unsigned int)pattern->stitch_list->count);
232 fprintf(file, "\">\",\"THREAD_COUNT:\",\"%u\"\n", (unsigned int)pattern->thread_list->count);
233 fprintf(file, "\">\",\"EXTENTS_LEFT:\",\"%f\"\n", boundingRect.left);
234 fprintf(file, "\">\",\"EXTENTS_TOP:\",\"%f\"\n", boundingRect.top);
235 fprintf(file, "\">\",\"EXTENTS_RIGHT:\",\"%f\"\n", boundingRect.right);
236 fprintf(file, "\">\",\"EXTENTS_BOTTOM:\",\"%f\"\n", boundingRect.bottom);
237 fprintf(file, "\">\",\"EXTENTS_WIDTH:\",\"%f\"\n", boundingRect.right - boundingRect.left);
238 fprintf(file, "\">\",\"EXTENTS_HEIGHT:\",\"%f\"\n", boundingRect.bottom - boundingRect.top);
239 fprintf(file,"\n");
240
241 /* write colors */
242 fprintf(file, "\"#\",\"[THREAD_NUMBER]\",\"[RED]\",\"[GREEN]\",");
243 fprintf(file, "\"[BLUE]\",\"[DESCRIPTION]\",\"[CATALOG_NUMBER]\"\n");
244
245 for (i = 0; i < pattern->thread_list->count; i++) {
246 EmbThread thr = pattern->thread_list->thread[i];
247 /* TODO: fix segfault that backtraces here when
248 libembroidery-convert from dst to csv. */
249 fprintf(file, "\"$\",\"%d\",\"%d\",\"%d\",\"%d\",\"%s\",\"%s\"\n",
250 i+1,
251 (int)thr.color.r,
252 (int)thr.color.g,
253 (int)thr.color.b,
254 thr.description,
255 thr.catalogNumber);
256 }
257 fprintf(file, "\n");
258
259 /* write stitches */
260 fprintf(file, "\"#\",\"[STITCH_TYPE]\",\"[X]\",\"[Y]\"\n");
261 for (i = 0; i < pattern->stitch_list->count; i++) {
262 EmbStitch s = pattern->stitch_list->stitch[i];
263 fprintf(file, "\"*\",\"%s\",\"%f\",\"%f\"\n",
264 csvStitchFlagToStr(s.flags), s.x, s.y);
265 }
266 return 1;
267}
