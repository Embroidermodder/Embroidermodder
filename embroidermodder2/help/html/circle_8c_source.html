46/****************************************************************
47 * Calculates the intersection points of two overlapping circles.
48 * Returns true if the circles intersect.
49 * Returns false if the circles do not intersect.
50 ****************************************************************/
51int
52getCircleCircleIntersections(EmbCircle c0, EmbCircle c1,
53 EmbVector *p0, EmbVector *p1)
54{
55 EmbVector delta;
56 EmbReal a, h, px2, py2, mx, my, d;
57 /* Distance between centers */
58 delta = embVector_subtract(c1.center, c0.center);
59 d = embVector_length(delta);
60
61 /*Circles share centers. This results in division by zero,
62 infinite solutions or one circle being contained within the other. */
63 if (d == 0.0) {
64 return 0;
65 } else if (d > (c0.radius + c1.radius)) {
66 /* Circles do not touch each other */
67 return 0;
68 } else if (d < (c0.radius - c1.radius)) {
69 /* One circle is contained within the other */
70 return 0;
71 }
72 /*
73 * Considering the two right triangles p0p2p3 and p1p2p3 we can write:
74 * a^2 + h^2 = r0^2 and b^2 + h^2 = r1^2
75 *
76 * BEGIN PROOF
77 *
78 * Remove h^2 from the equation by setting them equal to themselves:
79 * r0^2 - a^2 = r1^2 - b^2
80 * Substitute b with (d - a) since it is proven that d = a + b:
81 * r0^2 - a^2 = r1^2 - (d - a)^2
82 * Complete the square:
83 * r0^2 - a^2 = r1^2 - (d^2 -2da + a^2)
84 * Subtract r1^2 from both sides:
85 * r0^2 - r1^2 - a^2 = -(d^2 -2da + a^2)
86 * Invert the signs:
87 * r0^2 - r1^2 - a^2 = -d^2 + 2da - a^2
88 * Adding a^2 to each side cancels them out:
89 * r0^2 - r1^2 = -d^2 + 2da
90 * Add d^2 to both sides to shift it to the other side:
91 * r0^2 - r1^2 + d^2 = 2da
92 * Divide by 2d to finally solve for a:
93 * a = (r0^2 - r1^2 + d^2)/ (2d)
94 *
95 * END PROOF
96 */
97
98 a = ((c0.radius*c0.radius) - (c1.radius*c1.radius) + (d*d)) / (2.0 * d);
99 /* Solve for h by substituting a into a^2 + h^2 = r0^2 */
100 h = sqrt((c0.radius*c0.radius) - (a*a));
101
102 /*Find point p2 by adding the a offset in relation to line d to point p0 */
103 px2 = c0.center.x + (delta.x * a/d);
104 py2 = c0.center.y + (delta.y * a/d);
105
106 /* Tangent circles have only one intersection
107
108 TODO: using == in floating point arithmetic
109 doesn't account for the machine accuracy, having
110 a stated (EmbReal) tolerance value would help.
111 */
112 if (d == (c0.radius + c1.radius)) {
113 p0->x = px2;
114 p0->y = py2;
115 p1->x = px2;
116 p1->y = py2;
117 return 1;
118 }
119
120 /* Get the perpendicular slope by multiplying by the negative reciprocal
121 * Then multiply by the h offset in relation to d to get the actual offsets */
122 mx = -(delta.y * h/d);
123 my = (delta.x * h/d);
124
125 /* Add the offsets to point p2 to obtain the intersection points */
126 p0->x = px2 + mx;
127 p0->y = py2 + my;
128 p1->x = px2 - mx;
129 p1->y = py2 - my;
130
131 return 1;
132}

134/****************************************************************
135 * Calculates the tangent points on circle from a given point.
136 * Returns true if the given point lies outside the circle.
137 * Returns false if the given point is inside the circle.
138 ****************************************************************/
139int
140getCircleTangentPoints(EmbCircle c, EmbVector point, EmbVector *t0, EmbVector *t1)
141{
142 EmbCircle p;
143 EmbReal hyp;
144 /* Distance to center of circle */
145 hyp = embVector_distance(point, c.center);
146 /* Point is inside the circle */
147 if (hyp < c.radius) {
148 return 0;
149 } else if (hyp == c.center.y) {
150 /* Point is lies on the circle, so there is only one tangent point */
151 *t0 = point;
152 *t1 = point;
153 return 1;
154 }
155
156 /* Since the tangent lines are always perpendicular to the radius, so
157 * we can use the Pythagorean theorem to solve for the missing side */
158 p.center = point;
159 p.radius = sqrt((hyp*hyp) - (c.radius*c.radius));
160 return getCircleCircleIntersections(c, p, t0, t1);
161}
