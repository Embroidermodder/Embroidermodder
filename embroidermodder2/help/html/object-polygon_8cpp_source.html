81void PolygonObject::updateRubber(QPainter* painter)
82{
83 int rubberMode = objectRubberMode();
84 if(rubberMode == OBJ_RUBBER_POLYGON)
85 {
86 setObjectPos(objectRubberPoint("POLYGON_POINT_0"));
87
88 bool ok = false;
89 QString numStr = objectRubberText("POLYGON_NUM_POINTS");
90 if(numStr.isNull()) return;
91 int num = numStr.toInt(&ok);
92 if(!ok) return;
93
94 QString appendStr;
95 QPainterPath rubberPath;
96 rubberPath.moveTo(mapFromScene(objectRubberPoint("POLYGON_POINT_0")));
97 for(int i = 1; i <= num; i++)
98 {
99 appendStr = "POLYGON_POINT_" + QString().setNum(i);
100 QPointF appendPoint = mapFromScene(objectRubberPoint(appendStr));
101 rubberPath.lineTo(appendPoint);
102 }
103 //rubberPath.lineTo(0,0);
104 updatePath(rubberPath);
105
106 //Ensure the path isn't updated until the number of points is changed again
107 setObjectRubberText("POLYGON_NUM_POINTS", QString());
108 }
109 else if(rubberMode == OBJ_RUBBER_POLYGON_INSCRIBE)
110 {
111 setObjectPos(objectRubberPoint("POLYGON_CENTER"));
112
113 quint16 numSides = objectRubberPoint("POLYGON_NUM_SIDES").x();
114
115 QPointF inscribePoint = mapFromScene(objectRubberPoint("POLYGON_INSCRIBE_POINT"));
116 QLineF inscribeLine = QLineF(QPointF(0,0), inscribePoint);
117 qreal inscribeAngle = inscribeLine.angle();
118 qreal inscribeInc = 360.0/numSides;
119
120 if(painter) drawRubberLine(inscribeLine, painter, VIEW_COLOR_CROSSHAIR);
121
122 QPainterPath inscribePath;
123 //First Point
124 inscribePath.moveTo(inscribePoint);
125 //Remaining Points
126 for(int i = 1; i < numSides; i++)
127 {
128 inscribeLine.setAngle(inscribeAngle + inscribeInc*i);
129 inscribePath.lineTo(inscribeLine.p2());
130 }
131 updatePath(inscribePath);
132 }
133 else if(rubberMode == OBJ_RUBBER_POLYGON_CIRCUMSCRIBE)
134 {
135 setObjectPos(objectRubberPoint("POLYGON_CENTER"));
136
137 quint16 numSides = objectRubberPoint("POLYGON_NUM_SIDES").x();
138
139 QPointF circumscribePoint = mapFromScene(objectRubberPoint("POLYGON_CIRCUMSCRIBE_POINT"));
140 QLineF circumscribeLine = QLineF(QPointF(0,0), circumscribePoint);
141 qreal circumscribeAngle = circumscribeLine.angle();
142 qreal circumscribeInc = 360.0/numSides;
143
144 if(painter) drawRubberLine(circumscribeLine, painter, VIEW_COLOR_CROSSHAIR);
145
146 QPainterPath circumscribePath;
147 //First Point
148 QLineF prev(circumscribeLine.p2(), QPointF(0,0));
149 prev = prev.normalVector();
150 circumscribeLine.setAngle(circumscribeAngle + circumscribeInc);
151 QLineF perp(circumscribeLine.p2(), QPointF(0,0));
152 perp = perp.normalVector();
153 QPointF iPoint;
154 perp.intersects(prev, &iPoint);
155 circumscribePath.moveTo(iPoint);
156 //Remaining Points
157 for(int i = 2; i <= numSides; i++)
158 {
159 prev = perp;
160 circumscribeLine.setAngle(circumscribeAngle + circumscribeInc*i);
161 perp = QLineF(circumscribeLine.p2(), QPointF(0,0));
162 perp = perp.normalVector();
163 perp.intersects(prev, &iPoint);
164 circumscribePath.lineTo(iPoint);
165 }
166 updatePath(circumscribePath);
167 }
168 else if(rubberMode == OBJ_RUBBER_GRIP)
169 {
170 if(painter)
171 {
172 int elemCount = normalPath.elementCount();
173 QPointF gripPoint = objectRubberPoint("GRIP_POINT");
174 if(gripIndex == -1) gripIndex = findIndex(gripPoint);
175 if(gripIndex == -1) return;
176
177 int m = 0;
178 int n = 0;
179
180 if(!gripIndex) { m = elemCount-1; n = 1; }
181 else if(gripIndex == elemCount-1) { m = elemCount-2; n = 0; }
182 else { m = gripIndex-1; n = gripIndex+1; }
183 QPainterPath::Element em = normalPath.elementAt(m);
184 QPainterPath::Element en = normalPath.elementAt(n);
185 QPointF emPoint = QPointF(em.x, em.y);
186 QPointF enPoint = QPointF(en.x, en.y);
187 painter->drawLine(emPoint, mapFromScene(objectRubberPoint(QString())));
188 painter->drawLine(enPoint, mapFromScene(objectRubberPoint(QString())));
189
190 QLineF rubLine(mapFromScene(gripPoint), mapFromScene(objectRubberPoint(QString())));
191 drawRubberLine(rubLine, painter, VIEW_COLOR_CROSSHAIR);
192 }
193 }

207// Returns the closest snap point to the mouse point
208QPointF PolygonObject::mouseSnapPoint(const QPointF& mousePoint)
209{
210 QPainterPath::Element element = normalPath.elementAt(0);
211 QPointF closestPoint = mapToScene(QPointF(element.x, element.y));
212 qreal closestDist = QLineF(mousePoint, closestPoint).length();
213 int elemCount = normalPath.elementCount();
214 for(int i = 0; i < elemCount; ++i)
215 {
216 element = normalPath.elementAt(i);
217 QPointF elemPoint = mapToScene(element.x, element.y);
218 qreal elemDist = QLineF(mousePoint, elemPoint).length();
219 if(elemDist < closestDist)
220 {
221 closestPoint = elemPoint;
222 closestDist = elemDist;
223 }
224 }
225 return closestPoint;
226}

240int PolygonObject::findIndex(const QPointF& point)
241{
242 int i = 0;
243 int elemCount = normalPath.elementCount();
244 //NOTE: Points here are in item coordinates
245 QPointF itemPoint = mapFromScene(point);
246 for(i = 0; i < elemCount; i++)
247 {
248 QPainterPath::Element e = normalPath.elementAt(i);
249 QPointF elemPoint = QPointF(e.x, e.y);
250 if(itemPoint == elemPoint) return i;
251 }
252 return -1;
