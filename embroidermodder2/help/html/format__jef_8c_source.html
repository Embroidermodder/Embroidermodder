14int
15jefGetHoopSize(int width, int height)
16{
17 if (width < 50 && height < 50) {
18 return HOOP_50X50;
19 }
20 if (width < 110 && height < 110) {
21 return HOOP_110X110;
22 }
23 if (width < 140 && height < 200) {
24 return HOOP_140X200;
25 }
26 return ((int) HOOP_110X110);
27}

30jefDecode(unsigned char inputByte) {
31 if (inputByte >= 0x80) {
32 return (char) ((-~inputByte) - 1);

37void
38jefSetHoopFromId(EmbPattern* pattern, int hoopCode)
39{
40 if (!pattern) {
41 printf("ERROR: format-jef.c jefSetHoopFromId(), pattern argument is null\n");
42 return;
43 }
44 switch (hoopCode) {
45 case HOOP_126X110:
46 pattern->hoop_height = 126.0;
47 pattern->hoop_width = 110.0;
48 break;
49 case HOOP_50X50:
50 pattern->hoop_height = 50.0;
51 pattern->hoop_width = 50.0;
52 break;
53 case HOOP_110X110:
54 pattern->hoop_height = 110.0;
55 pattern->hoop_width = 110.0;
56 break;
57 case HOOP_140X200:
58 pattern->hoop_height = 140.0;
59 pattern->hoop_width = 200.0;
60 break;
61 case HOOP_230X200:
62 pattern->hoop_height = 230.0;
63 pattern->hoop_width = 200.0;
64 break;
65 }
66}

88char
89readJef(EmbPattern* pattern, FILE* file) {
90 int stitchOffset, formatFlags, numberOfColors, numberOfStitchs;
91 int hoopSize, i, stitchCount;
92 struct hoop_padding bounds, rectFrom110x110;
93 struct hoop_padding rectFrom50x50, rectFrom200x140, rect_from_custom;
94 char date[8], time[8];
95
96 embInt_read(file, "stitchOffset", &stitchOffset, EMB_INT32_LITTLE);
97 embInt_read(file, "formatFlags", &formatFlags, EMB_INT32_LITTLE); /* TODO: find out what this means */
98
99 fread((unsigned char*) date, 1, 8, file); /* TODO: check return value */
100 fread((unsigned char*) time, 1, 8, file); /* TODO: check return value */
101 embInt_read(file, "numberOfColors", &numberOfColors, EMB_INT32_LITTLE);
102 embInt_read(file, "numberOfStitchs", &numberOfStitchs, EMB_INT32_LITTLE);
103 embInt_read(file, "hoopSize", &hoopSize, EMB_INT32_LITTLE);
104 jefSetHoopFromId(pattern, hoopSize);
105 if (numberOfStitchs > MAX_STITCHES) {
106 numberOfStitchs = MAX_STITCHES;
107 puts("ERROR: this file is corrupted or has too many stitches.");
108 return 0;
109 }
110 if (emb_verbose>1) {
111 printf("format flags = %d\n", formatFlags);
112 printf("number of colors = %d\n", numberOfColors);
113 printf("number of stitches = %d\n", numberOfStitchs);
114 }
115
116 read_hoop(file, &bounds, "bounds");
117 read_hoop(file, &rectFrom110x110, "rectFrom110x110");
118 read_hoop(file, &rectFrom50x50, "rectFrom50x50");
119 read_hoop(file, &rectFrom200x140, "rectFrom200x140");
120 read_hoop(file, &rect_from_custom, "rect_from_custom");
121
122 for (i = 0; i < numberOfColors; i++) {
123 int thread_num;
124 embInt_read(file, "thread_num", &thread_num, EMB_INT32_LITTLE);
125 embPattern_addThread(pattern, jefThreads[thread_num % 79]);
126 }
127 fseek(file, stitchOffset, SEEK_SET);
128 stitchCount = 0;
129 while (stitchCount < numberOfStitchs + 100) {
130 unsigned char b[2];
131 char dx = 0, dy = 0;
132 int flags = NORMAL;
133 if (fread(b, 1, 2, file) != 2) {
134 break;
135 }
136
137 if (b[0] == 0x80) {
138 if (b[1] & 1) {
139 if (fread(b, 1, 2, file) != 2) {
140 break;
141 }
142 flags = STOP;
143 }
144 else if ((b[1] == 2) || (b[1] == 4) || b[1] == 6) {
145 if (fread(b, 1, 2, file) != 2) {
146 break;
147 }
148 flags = TRIM;
149 }
150 else if (b[1] == 0x10) {
151 embPattern_addStitchRel(pattern, 0.0, 0.0, END, 1);
152 break;
153 }
154 }
155 dx = jefDecode(b[0]);
156 dy = jefDecode(b[1]);
157 embPattern_addStitchRel(pattern, dx / 10.0, dy / 10.0, flags, 1);
158 stitchCount++;
159 }
160 return 1;
161}

163void
164jefEncode(unsigned char* b, char dx, char dy, int flags) {
165 if (!b) {
166 printf("ERROR: format-jef.c expEncode(), b argument is null\n");
167 return;
168 }
169 if (flags == STOP) {
170 b[0] = 0x80;
171 b[1] = 1;
172 b[2] = dx;
173 b[3] = dy;
174 } else if (flags == END) {
175 b[0] = 0x80;
176 b[1] = 0x10;
177 b[2] = 0;
178 b[3] = 0;
179 } else if (flags == TRIM || flags == JUMP) {
180 b[0] = 0x80;
181 b[1] = 2;
182 b[2] = dx;
183 b[3] = dy;
184 } else {
185 b[0] = dx;
186 b[1] = dy;
187 }
188}

190char
191writeJef(EmbPattern* pattern, FILE* file)
192{
193 int colorlistSize, minColors, designWidth, designHeight, i;
194 EmbRect boundingRect;
195 EmbTime time;
196 EmbReal width, height;
197 int data;
198 EmbVector pos;
199
200 embPattern_correctForMaxStitchLength(pattern, 12.7, 12.7);
201
202 colorlistSize = pattern->thread_list->count;
203 minColors = EMB_MAX(colorlistSize, 6);
204 binaryWriteInt(file, 0x74 + (minColors * 4));
205 binaryWriteInt(file, 0x0A);
206
207 embTime_initNow(&time);
208
209 fprintf(file, "%04d%02d%02d%02d%02d%02d", (int)(time.year + 1900),
210 (int)(time.month + 1), (int)(time.day), (int)(time.hour),
211 (int)(time.minute), (int)(time.second));
212 fpad(file, 0, 2);
213 embInt_write(file, ".", &(pattern->thread_list->count), EMB_INT32_LITTLE);
214 data = pattern->stitch_list->count + EMB_MAX(0, (6 - colorlistSize) * 2) + 1;
215 embInt_write(file, ".", &data, EMB_INT32_LITTLE);
216
217 boundingRect = embPattern_calcBoundingBox(pattern);
218 width = boundingRect.right - boundingRect.left;
219 height = boundingRect.bottom - boundingRect.top;
220 designWidth = (int)(width * 10.0);
221 designHeight = (int)(height * 10.0);
222
223 binaryWriteInt(file, jefGetHoopSize(designWidth, designHeight));
224
225 /* Distance from center of Hoop */
226 binaryWriteInt(file, (int) (designWidth / 2)); /* left */
227 binaryWriteInt(file, (int) (designHeight / 2)); /* top */
228 binaryWriteInt(file, (int) (designWidth / 2)); /* right */
229 binaryWriteInt(file, (int) (designHeight / 2)); /* bottom */
230
231 /* Distance from default 110 x 110 Hoop */
232 if (EMB_MIN(550 - designWidth / 2, 550 - designHeight / 2) >= 0) {
233 binaryWriteInt(file, EMB_MAX(-1, 550 - designWidth / 2)); /* left */
234 binaryWriteInt(file, EMB_MAX(-1, 550 - designHeight / 2)); /* top */
235 binaryWriteInt(file, EMB_MAX(-1, 550 - designWidth / 2)); /* right */
236 binaryWriteInt(file, EMB_MAX(-1, 550 - designHeight / 2)); /* bottom */
237 } else {
238 binaryWriteInt(file, -1);
239 binaryWriteInt(file, -1);
240 binaryWriteInt(file, -1);
241 binaryWriteInt(file, -1);
242 }
243
244 /* Distance from default 50 x 50 Hoop */
245 if (EMB_MIN(250 - designWidth / 2, 250 - designHeight / 2) >= 0) {
246 binaryWriteInt(file, (int) EMB_MAX(-1, 250 - designWidth / 2)); /* left */
247 binaryWriteInt(file, (int) EMB_MAX(-1, 250 - designHeight / 2)); /* top */
248 binaryWriteInt(file, (int) EMB_MAX(-1, 250 - designWidth / 2)); /* right */
249 binaryWriteInt(file, (int) EMB_MAX(-1, 250 - designHeight / 2)); /* bottom */
250 } else {
251 binaryWriteInt(file, -1);
252 binaryWriteInt(file, -1);
253 binaryWriteInt(file, -1);
254 binaryWriteInt(file, -1);
255 }
256
257 /* Distance from default 140 x 200 Hoop */
258 binaryWriteInt(file, (int) (700 - designWidth / 2)); /* left */
259 binaryWriteInt(file, (int) (1000 - designHeight / 2)); /* top */
260 binaryWriteInt(file, (int) (700 - designWidth / 2)); /* right */
261 binaryWriteInt(file, (int) (1000 - designHeight / 2)); /* bottom */
262
263 /* repeated Distance from default 140 x 200 Hoop */
264 /* TODO: Actually should be distance to custom hoop */
265 binaryWriteInt(file, (int) (630 - designWidth / 2)); /* left */
266 binaryWriteInt(file, (int) (550 - designHeight / 2)); /* top */
267 binaryWriteInt(file, (int) (630 - designWidth / 2)); /* right */
268 binaryWriteInt(file, (int) (550 - designHeight / 2)); /* bottom */
269
270 for (i = 0; i < pattern->thread_list->count; i++) {
271 int j = embThread_findNearestThread(pattern->thread_list->thread[i].color, (EmbThread *)jefThreads, 79);
272 binaryWriteInt(file, j);
273 }
274
275 for (i = 0; i < (minColors - colorlistSize); i++) {
276 int a = 0x0D;
277 binaryWriteInt(file, a);
278 }
279
280 pos.x = 0.0;
281 pos.y = 0.0;
282 for (i = 0; i < pattern->stitch_list->count; i++) {
283 unsigned char b[4];
284 EmbStitch st;
285 char dx, dy;
286 b[0] = 0;
287 b[1] = 0;
288 b[2] = 0;
289 b[3] = 0;
290 st = pattern->stitch_list->stitch[i];
291 dx = (char)emb_round(10.0*(st.x - pos.x));
292 dy = (char)emb_round(10.0*(st.y - pos.y));
293 pos.x += 0.1*dx;
294 pos.y += 0.1*dy;
295 jefEncode(b, dx, dy, st.flags);
296 if ((b[0] == 0x80) && ((b[1] == 1) || (b[1] == 2) || (b[1] == 4) || (b[1] == 0x10))) {
297 fwrite(b, 1, 4, file);
298 } else {
299 fwrite(b, 1, 2, file);
300 }
301 }
302 return 1;
303}
