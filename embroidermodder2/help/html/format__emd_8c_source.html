46
47 while (!feof(file)) {
48 char dx, dy;
49 int flags = NORMAL;
50 fread(b, 1, 2, file);
51
52 if (b[0] == 0x80) {
53 if (b[1] == 0x2A) {
54 embPattern_addStitchRel(pattern, 0, 0, STOP, 1);
55 continue;
56 } else if (b[1] == 0x80) {
57 fread(b, 1, 2, file);
58 flags = TRIM;
59 } else if (b[1] == 0xFD) {
60 embPattern_addStitchRel(pattern, 0, 0, END, 1);
61 break;
62 } else {
63 continue;
64 }
65 }
66 dx = emdDecode(b[0]);
67 dy = emdDecode(b[1]);
68 embPattern_addStitchRel(pattern, dx / 10.0, dy / 10.0, flags, 1);
69 }
70 return 1;
71}
