14unsigned char*
15vp3ReadString(FILE* file)
16{
17 short stringLength;
18 unsigned char* charString = 0;
19 if (!file) { printf("ERROR: format-vp3.c vp3ReadString(), file argument is null\n"); return 0; }
20 embInt_read(file, ".", &stringLength, EMB_INT16_BIG);
21 charString = (unsigned char*)malloc(stringLength);
22 if (!charString) { printf("ERROR: format-vp3.c vp3ReadString(), cannot allocate memory for charString\n"); return 0; }
23 fread(charString, 1, stringLength, file); /* TODO: check return value */
24 return charString;
25}

27int
28vp3Decode(unsigned char inputByte)
29{
30 if (inputByte > 0x80) {
31 return (int)-((unsigned char)((~inputByte) + 1));
32 }
33 return ((int)inputByte);
34}

36short
37vp3DecodeInt16(unsigned short inputByte)
38{
39 if (inputByte > 0x8000) {
40 return -((short) ((~inputByte) + 1));
41 }
42 return ((short)inputByte);

109char
110readVp3(EmbPattern* pattern, FILE* file)
111{
112 unsigned char magicString[5];
113 unsigned char some;
114 unsigned char* softwareVendorString = 0;
115 /* unsigned char v2, ..., v18; */
116 unsigned char* anotherSoftwareVendorString = 0;
117 int numberOfColors;
118 long colorSectionOffset;
119 unsigned char magicCode[6];
120 short someShort;
121 unsigned char someByte;
122 int bytesRemainingInFile, hoopConfigurationOffset;
123 unsigned char* fileCommentString = 0; /* some software writes used settings here */
124 unsigned char* anotherCommentString = 0;
125 int i;
126
127 fread(magicString, 1, 5, file); /* %vsm% */ /* TODO: check return value */
128 some = (char)fgetc(file); /* 0 */
129 softwareVendorString = vp3ReadString(file);
130 embInt_read(file, "someShort", &someShort, EMB_INT16_LITTLE);
131 someByte = (char)fgetc(file);
132 embInt_read(file, ".", &bytesRemainingInFile, EMB_INT32_LITTLE);
133 fileCommentString = vp3ReadString(file);
134 hoopConfigurationOffset = (int)ftell(file);
135
136 vp3ReadHoopSection(file);
137
138 anotherCommentString = vp3ReadString(file);
139
140 /* TODO: review v1 thru v18 variables and use emb_unused() if needed */
141 for (i = 0; i < 18; i++) {
142 unsigned char v1;
143 v1 = (char)fgetc(file);
144 if (emb_verbose>1) {
145 printf("v%d = %d\n", i, v1);
146 }
147 }
148
149 fread(magicCode, 1, 6, file); /* 0x78 0x78 0x55 0x55 0x01 0x00 */ /* TODO: check return value */
150
151 anotherSoftwareVendorString = vp3ReadString(file);
152
153 embInt_read(file, ".", &numberOfColors, EMB_INT16_BIG);
154 colorSectionOffset = (int)ftell(file);
155 if (emb_verbose>1) {
156 puts("LIBEMBROIDERY DEBUG");
157 puts("Format vp3");
158 printf(" some: %d\n", some);
159 printf(" softwareVendorString: %s\n", softwareVendorString);
160 printf(" someByte: %d\n", someByte);
161 printf(" bytesRemainingInFile: %d\n", bytesRemainingInFile);
162 printf(" hoop config offset: %d\n", hoopConfigurationOffset);
163 printf(" another comment string: %s\n", anotherCommentString);
164 printf(" another software vendor string: %s\n", anotherSoftwareVendorString);
165 printf(" number of colors: %d\n", numberOfColors);
166 printf(" color section offset: %ld\n", colorSectionOffset);
167 }
168
169 for (i = 0; i < numberOfColors; i++) {
170 EmbThread t;
171 char tableSize;
172 int startX, startY, offsetToNextColorX, offsetToNextColorY;
173 unsigned char* threadColorNumber, *colorName, *threadVendor;
174 int unknownThreadString, numberOfBytesInColor;
175
176 strcpy(t.catalogNumber, "");
177 strcpy(t.description, "");
178 fseek(file, colorSectionOffset, SEEK_SET);
179 printf("ERROR: format-vp3.c Color Check Byte #1: 0 == %d\n", (char)fgetc(file));
180 printf("ERROR: format-vp3.c Color Check Byte #2: 5 == %d\n", (char)fgetc(file));
181 printf("ERROR: format-vp3.c Color Check Byte #3: 0 == %d\n", (char)fgetc(file));
182 embInt_read(file, ".", &colorSectionOffset, EMB_INT32_BIG);
183 colorSectionOffset += ftell(file);
184 embInt_read(file, ".", &startX, EMB_INT32_BIG);
185 embInt_read(file, ".", &startY, EMB_INT32_BIG);
186 embPattern_addStitchAbs(pattern, startX / 1000.0, -startY / 1000.0, JUMP, 1);
187
188 tableSize = (char)fgetc(file);
189 fseek(file, 1, SEEK_CUR);
190 embColor_read(file, &(t.color), 3);
191 embPattern_addThread(pattern, t);
192 fseek(file, 6*tableSize - 1, SEEK_CUR);
193
194 threadColorNumber = vp3ReadString(file);
195 colorName = vp3ReadString(file);
196 threadVendor = vp3ReadString(file);
197
198 offsetToNextColorX = fread_int32_be(file);
199 offsetToNextColorY = fread_int32_be(file);
200
201 embInt_read(file, "unknown thread string", &unknownThreadString, EMB_INT16_BIG);
202 fseek(file, unknownThreadString, SEEK_CUR);
203 numberOfBytesInColor = fread_int32_be(file);
204 fseek(file, 0x3, SEEK_CUR);
205
206 if (emb_verbose>1) {
207 printf("number of bytes in color: %d\n", numberOfBytesInColor);
208 printf("thread color number: %s\n", threadColorNumber);
209 printf("offset to next color x: %d\n", offsetToNextColorX);
210 printf("offset to next color y: %d\n", offsetToNextColorY);
211 printf("color name: %s\n", colorName);
212 printf("thread vendor: %s\n", threadVendor);
213 printf("fileCommentString: %s\n", fileCommentString);
214 }
215
216 while (ftell(file) < colorSectionOffset - 1) {
217 int lastFilePosition = ftell(file);
218 int x = vp3Decode((char)fgetc(file));
219 int y = vp3Decode((char)fgetc(file));
220 short readIn;
221 if (x == 0x80) {
222 switch (y) {
223 case 0x00:
224 case 0x03:
225 break;
226 case 0x01:
227 embInt_read(file, ".", &readIn, EMB_INT16_BIG);
228 x = vp3DecodeInt16(readIn);
229 embInt_read(file, ".", &readIn, EMB_INT16_BIG);
230 y = vp3DecodeInt16(readIn);
231 fseek(file, 2, SEEK_CUR);
232 embPattern_addStitchRel(pattern, x/ 10.0, y / 10.0, TRIM, 1);
233 break;
234 default:
235 break;
236 }
237 } else {
238 embPattern_addStitchRel(pattern, x / 10.0, y / 10.0, NORMAL, 1);
239 }
240
241 if (ftell(file) == lastFilePosition) {
242 printf("ERROR: format-vp3.c could not read stitch block in entirety\n");
243 return 0;
244 }
245 }
246 if (i + 1 < numberOfColors) {
247 embPattern_addStitchRel(pattern, 0, 0, STOP, 1);
248 }
249 }
250 embPattern_flipVertical(pattern);
251 return 1;
252}

279{

283 int colorSectionStitchBytes, first = 1, i, numberOfColors;
284 EmbColor color;
285 color.r = 0xFE;
286 color.g = 0xFE;
287 color.b = 0xFE;
288
289 bounds = embPattern_calcBoundingBox(pattern);
290
291 embPattern_correctForMaxStitchLength(pattern, 3200.0, 3200.0); /* VP3 can encode signed 16bit deltas */
292
293 embPattern_flipVertical(pattern);
294
295 fwrite("%vsm%\0", 1, 6, file);
296 vp3WriteString(file, "Embroidermodder");
297 fwrite("\x00\x02\x00", 1, 3, file);
298
299 remainingBytesPos = ftell(file);
300 binaryWriteInt(file, 0); /* placeholder */
301 vp3WriteString(file, "");
302 binaryWriteIntBE(file, bounds.right * 1000);
303 binaryWriteIntBE(file, bounds.bottom * 1000);
304 binaryWriteIntBE(file, bounds.left * 1000);
305 binaryWriteIntBE(file, bounds.top * 1000);
306 binaryWriteInt(file, 0); /* this would be some (unknown) function of thread length */
307 fputc(0, file);
308
309 numberOfColors = embPattern_color_count(pattern, color);
310 fputc(numberOfColors, file);
311 fwrite("\x0C\x00\x01\x00\x03\x00", 1, 6, file);
312
313 remainingBytesPos2 = ftell(file);
314 binaryWriteInt(file, 0); /* placeholder */
315
316 binaryWriteIntBE(file, 0); /* origin X */
317 binaryWriteIntBE(file, 0); /* origin Y */
318 fpad(file, 0, 3);
319
320 binaryWriteIntBE(file, bounds.right * 1000);
321 binaryWriteIntBE(file, bounds.bottom * 1000);
322 binaryWriteIntBE(file, bounds.left * 1000);
323 binaryWriteIntBE(file, bounds.top * 1000);
324
325 binaryWriteIntBE(file, (bounds.right - bounds.left) * 1000);
326 binaryWriteIntBE(file, (bounds.bottom - bounds.top) * 1000);
327
328 vp3WriteString(file, "");
329 a = 25700;
330 embInt_write(file, ".", &a, EMB_INT16_BIG);
331 a_int = 4096;
332 embInt_write(file, ".", &a_int, EMB_INT32_BIG);
333 a_int = 0;
334 embInt_write(file, ".", &a_int, EMB_INT32_BIG);
335 embInt_write(file, ".", &a_int, EMB_INT32_BIG);
336 a_int = 4096;
337 embInt_write(file, ".", &a_int, EMB_INT32_BIG);
338
339 fwrite("xxPP\x01\0", 1, 6, file);
340 vp3WriteString(file, "");
341 embInt_write(file, ".", &numberOfColors, EMB_INT16_BIG);
342
343 for (i=0; i<pattern->stitch_list->count; i++) {
344 char colorName[8] = { 0 };
345 EmbReal lastX, lastY;
346 int colorSectionLengthPos, j;
347 EmbStitch s;
348 int lastColor;
349
350 j = 0;
351 s.x = 0.0;
352 s.y = 0.0;
353 s.color = 0;
354 s.flags = 0;
355
356 if (!first) {
357 fputc(0, file);
358 }
359 fputc(0, file);
360 fputc(5, file);
361 fputc(0, file);
362
363 colorSectionLengthPos = ftell(file);
364 binaryWriteInt(file, 0); /* placeholder */
365
366 /*
367 pointer = mainPointer;
368 color = pattern->thread_list->thread[pointer->stitch.color].color;
369
370 if (first && pointer->stitch.flags & JUMP && pointer->next->stitch.flags & JUMP) {
371 pointer = pointer->next;
372 }
373
374 s = pointer->stitch;
375 */
376 if (emb_verbose>1) {
377 printf("%d\n", j);
378 printf("format-vp3.c DEBUG %d, %f, %f\n", s.flags, s.x, s.y);
379 }
380 binaryWriteIntBE(file, s.x * 1000);
381 binaryWriteIntBE(file, -s.y * 1000);
382 /* pointer = pointer->next; */
383
384 first = 0;
385
386 lastX = s.x;
387 lastY = s.y;
388 lastColor = s.color;
389 if (emb_verbose>1) {
390 printf("last %f %f %d\n", lastX, lastY, lastColor);
391 }
392
393 fwrite("\x01\x00", 1, 2, file);
394
395 printf("format-vp3.c writeVp3(), switching to color (%d, %d, %d)\n", color.r, color.g, color.b);
396 embColor_write(file, color, 4);
397
398 fwrite("\x00\x00\x05", 1, 3, file);
399 fputc(40, file);
400
401 vp3WriteString(file, "");
402
403 sprintf(colorName, "#%02x%02x%02x", color.b, color.g, color.r);
404
405 vp3WriteString(file, colorName);
406 vp3WriteString(file, "");
407
408 binaryWriteIntBE(file, 0);
409 binaryWriteIntBE(file, 0);
410
411 vp3WriteStringLen(file, "\0", 1);
412
413 colorSectionStitchBytes = ftell(file);
414 binaryWriteInt(file, 0); /* placeholder */
415
416 fputc(10, file);
417 fputc(246, file);
418 fputc(0, file);
419
420 /*
421 for (j=i; j<pattern->stitch_list->count; j++) {
422 while (pointer) {
423 short dx, dy;
424
425 EmbStitch s = pointer->stitch;
426 if (s.color != lastColor) {
427 break;
428 }
429 if (s.flags & END || s.flags & STOP) {
430 break;
431 }
432 dx = (s.x - lastX) * 10;
433 dy = (s.y - lastY) * 10;
434 // output is in ints, ensure rounding errors do not sum up.
435 lastX = lastX + dx / 10.0;
436 lastY = lastY + dy / 10.0;
437
438 if (dx < -127 || dx > 127 || dy < -127 || dy > 127) {
439 fputc(128, file);
440 fputc(1, file);
441 embInt_write(file, ".", &dx, EMB_INT16_BIG);
442 embInt_write(file, ".", &dy, EMB_INT16_BIG);
443 fputc(128, file);
444 fputc(2, file);
445 }
446 else {
447 char b[2];
448 b[0] = dx;
449 b[1] = dy;
450 fwrite(b, 1, 2, file);
451 }
452
453 pointer = pointer->next;
454 }
455 */
456
457 vp3PatchByteCount(file, colorSectionStitchBytes, -4);
458 vp3PatchByteCount(file, colorSectionLengthPos, -3);
459
460 /* mainPointer = pointer; */
461 }
462
463 vp3PatchByteCount(file, remainingBytesPos2, -4);
464 vp3PatchByteCount(file, remainingBytesPos, -4);
465
466 embPattern_flipVertical(pattern);
467 return 0;
468}
