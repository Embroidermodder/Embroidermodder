24char
25readXxx(EmbPattern* pattern, FILE* file)
26{
27 int dx = 0, dy = 0, numberOfColors, paletteOffset, i;
28 char thisStitchJump = 0;
29
30 if (emb_verbose>1) {
31 puts("readXxx has been overridden.");
32 return 0;
33 }
34
35 fseek(file, 0x27, SEEK_SET);
36 embInt_read(file, ".", &numberOfColors, EMB_INT16_LITTLE);
37 fseek(file, 0xFC, SEEK_SET);
38 embInt_read(file, ".", &paletteOffset, EMB_INT32_LITTLE);
39 fseek(file, paletteOffset + 6, SEEK_SET);
40
41 for (i = 0; i < numberOfColors; i++) {
42 EmbThread thread;
43 strcpy(thread.catalogNumber, "NULL");
44 strcpy(thread.description, "NULL");
45 fseek(file, 1, SEEK_CUR);
46 embColor_read(file, &(thread.color), 3);
47 embPattern_addThread(pattern, thread);
48 }
49 fseek(file, 0x100, SEEK_SET);
50
51 for (i = 0; !feof(file) && ftell(file) < paletteOffset; i++) {
52 unsigned char b0, b1;
53 int flags;
54 flags = NORMAL;
55 if (thisStitchJump) flags = TRIM;
56 thisStitchJump = 0;
57 b0 = (char)fgetc(file);
58 b1 = (char)fgetc(file);
59 /* TODO: ARE THERE OTHER BIG JUMP CODES? */
60 if (b0 == 0x7E || b0 == 0x7D) {
61 dx = b1 + ((char)fgetc(file) << 8);
62 dx = ((short) dx);
63 dy = fread_int16(file);
64 flags = TRIM;
65 } else if (b0 == 0x7F) {
66 /* TODO: LOOKS LIKE THESE CODES ARE IN THE HEADER */
67 if (b1 != 0x17 && b1 != 0x46 && b1 >= 8) {
68 b0 = 0;
69 b1 = 0;
70 thisStitchJump = 1;
71 flags = STOP;
72 } else if (b1 == 1) {
73 flags = TRIM;
74 b0 = (char)fgetc(file);
75 b1 = (char)fgetc(file);
76 } else {
77 continue;
78 }
79 dx = xxxDecodeByte(b0);
80 dy = xxxDecodeByte(b1);
81 } else {
82 dx = xxxDecodeByte(b0);
83 dy = xxxDecodeByte(b1);
84 }
85 embPattern_addStitchRel(pattern, dx / 10.0, dy / 10.0, flags, 1);
86 }
87 /* TODO: check end of pattern for multiple STOPs */
88 return 1;
89}

98void
99xxxEncodeStitch(FILE* file, EmbReal deltaX, EmbReal deltaY, int flags)
100{
101 if ((flags & (JUMP | TRIM)) && (fabs(deltaX) > 124 || fabs(deltaY) > 124)) {
102 fputc(0x7E, file);
103 /* Does this cast work right? */
104 binaryWriteShort(file, (short)deltaX);
105 binaryWriteShort(file, (short)deltaY);
106 } else {
107 /* TODO: Verify this works after changing this to unsigned char */
108 fputc((unsigned char)emb_round(deltaX), file);
109 fputc((unsigned char)emb_round(deltaY), file);
110 }

113void
114xxxEncodeDesign(FILE* file, EmbPattern* p)
115{
116 int i;
117 EmbReal thisX = 0.0f;
118 EmbReal thisY = 0.0f;
119
120 if (p->stitch_list->count > 0) {
121 thisX = (float)p->stitch_list->stitch[0].x;
122 thisY = (float)p->stitch_list->stitch[0].y;
123 }
124 for (i = 0; i < p->stitch_list->count; i++) {
125 EmbStitch s = p->stitch_list->stitch[i];
126 EmbReal deltaX, deltaY;
127 EmbReal previousX = thisX;
128 EmbReal previousY = thisY;
129 thisX = s.x;
130 thisY = s.y;
131 deltaX = thisX - previousX;
132 deltaY = thisY - previousY;
133 if (s.flags & STOP) {
134 xxxEncodeStop(file, s);
135 } else if (s.flags & END) {
136 } else {
137 xxxEncodeStitch(file, deltaX * 10.0f, deltaY * 10.0f, s.flags);
138 }
139 }
