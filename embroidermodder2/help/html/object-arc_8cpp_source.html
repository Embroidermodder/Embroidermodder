180{
181 qreal rot = radians(rotation());
182 qreal cosRot = qCos(rot);
183 qreal sinRot = qSin(rot);
184 qreal x = arcStartPoint.x()*scale();

186 qreal rotX = x*cosRot - y*sinRot;

266qreal ArcObject::objectIncludedAngle() const
267{
268 qreal chord = objectChord();
269 qreal rad = objectRadius();
270 if(chord <= 0 || rad <= 0) return 0; //Prevents division by zero and non-existant circles
271
272 //NOTE: Due to floating point rounding errors, we need to clamp the quotient so it is in the range [-1, 1]
273 // If the quotient is out of that range, then the result of asin() will be NaN.
274 qreal quotient = chord/(2.0*rad);
275 if(quotient > 1.0) quotient = 1.0;
276 if(quotient < 0.0) quotient = 0.0; //NOTE: 0 rather than -1 since we are enforcing a positive chord and radius
277 return degrees(2.0*asin(quotient)); //Properties of a Circle - Get the Included Angle - Reference: ASD9
278}

288void ArcObject::updatePath()
289{
290 qreal startAngle = (objectStartAngle() + rotation());
291 qreal spanAngle = objectIncludedAngle();
292
293 if(objectClockwise())
294 spanAngle = -spanAngle;
295
296 QPainterPath path;
297 path.arcMoveTo(rect(), startAngle);
298 path.arcTo(rect(), startAngle, spanAngle);
299 //NOTE: Reverse the path so that the inside area isn't considered part of the arc
300 path.arcTo(rect(), startAngle+spanAngle, -spanAngle);
301 setObjectPath(path);

305{
306 QGraphicsScene* objScene = scene();

308

343// Returns the closest snap point to the mouse point
344QPointF ArcObject::mouseSnapPoint(const QPointF& mousePoint)
345{
346 QPointF center = objectCenter();
347 QPointF start = objectStartPoint();
348 QPointF mid = objectMidPoint();
349 QPointF end = objectEndPoint();
350
351 qreal cntrDist = QLineF(mousePoint, center).length();
352 qreal startDist = QLineF(mousePoint, start).length();
353 qreal midDist = QLineF(mousePoint, mid).length();
354 qreal endDist = QLineF(mousePoint, end).length();
355
356 qreal minDist = qMin(qMin(cntrDist, startDist), qMin(midDist, endDist));
357
358 if (minDist == cntrDist) return center;
359 else if(minDist == startDist) return start;
360 else if(minDist == midDist) return mid;
361 else if(minDist == endDist) return end;
362
363 return scenePos();
364}
