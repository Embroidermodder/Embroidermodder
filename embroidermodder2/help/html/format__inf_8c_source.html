19char
20readInf(EmbPattern* pattern, FILE* file)
21{
22 int nColors, i;
23 char colorType[50];
24 char colorDescription[50];
25 EmbThread t;
26
27 fseek(file, 12, SEEK_CUR);
28 embInt_read(file, "nColors", &nColors, EMB_INT32_BIG);
29
30 pattern->thread_list->count = 0;
31
32 for (i = 0; i < nColors; i++) {
33 fseek(file, 4, SEEK_CUR);
34 embColor_read(file, &(t.color), 3);
35 strcpy(t.catalogNumber, "");
36 strcpy(t.description, "");
37 embPattern_addThread(pattern, t);
38 fseek(file, 2, SEEK_CUR);
39 binaryReadString(file, colorType, 50);
40 binaryReadString(file, colorDescription, 50);
41 }
42 return 1;
43}

45char
46writeInf(EmbPattern* pattern, FILE* file)
47{
48 int i, bytesRemaining;
49
50 binaryWriteUIntBE(file, 0x01);
51 binaryWriteUIntBE(file, 0x08);
52 /* write place holder offset */
53 binaryWriteUIntBE(file, 0x00);
54 binaryWriteUIntBE(file, pattern->thread_list->count);
55
56 for (i = 0; i < pattern->thread_list->count; i++) {
57 char buffer[50];
58 unsigned short record_length, record_number, needle_number;
59 EmbColor c;
60 c = pattern->thread_list->thread[i].color;
61 sprintf(buffer, "RGB(%d,%d,%d)", (int)c.r, (int)c.g, (int)c.b);
62 record_length = 14 + strlen(buffer);
63 record_number = i;
64 needle_number = i;
65 embInt_write(file, "record length", &record_length, EMB_INT16_BIG);
66 embInt_write(file, "record number", &record_number, EMB_INT16_BIG);
67 embColor_write(file, c, 3);
68 embInt_write(file, "needle number", &needle_number, EMB_INT16_BIG);
69 fwrite("RGB\0", 1, 4, file);
70 fprintf(file, "%s", buffer);
71 fwrite("\0", 1, 1, file);
72 }
73 /* It appears that there should be a pad here otherwise it clips into
74 * the color description. */
75 fpad(file, 0, 8);
76 fseek(file, -8, SEEK_END);
77 bytesRemaining = ftell(file);
78 fseek(file, 8, SEEK_SET);
79 binaryWriteUIntBE(file, bytesRemaining);
80 return 1;
81}
