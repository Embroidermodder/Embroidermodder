30EmbColor svgColorToEmbColor(char* colorString)
31{
32 EmbColor c;
33 char* pEnd = 0;
34 char* colorStr = copy_trim(colorString); /* Trim out any junk spaces */
35 int length = strlen(colorStr);
36 int tableColor;
37
38 /* SVGTiny1.2 Spec Section 11.13.1 syntax for color values */
39 if (length == 7 && colorStr[0] == '#') /* Six digit hex — #rrggbb */ {
40 while (*colorStr=='#' && *colorStr!=0) {
41 colorStr++;
42 }
43 c = embColor_fromHexStr(colorStr);
44 }
45 else if (length == 4 && colorStr[0] == '#') /* Three digit hex — #rgb */ {
46 /* Convert the 3 digit hex to a six digit hex */
47 char hex[7];
48 sprintf(hex, "%c%c%c%c%c%c", colorStr[1], colorStr[1], colorStr[2],
49 colorStr[2], colorStr[3], colorStr[3]);
50
51 c = embColor_fromHexStr(hex);
52 }
53 else if (strstr(colorStr, "%")) /* Float functional — rgb(R%, G%, B%) */ {
54 char *s = colorStr;
55 /* replace characters we aren't using with spaces */
56 for (; *s; s++) {
57 if (*s == 'r' || *s == 'g' || *s == 'b' || *s == ','
58 || *s == '(' || *s == ')' || *s == '%') {
59 *s = ' ';
60 }
61 }
62 c.r = (unsigned char)emb_round(255.0/100.0 * strtod(colorStr, &pEnd));
63 c.g = (unsigned char)emb_round(255.0/100.0 * strtod(pEnd, &pEnd));
64 c.b = (unsigned char)emb_round(255.0/100.0 * strtod(pEnd, &pEnd));
65 }
66 else if (length > 3 && colorStr[0] == 'r' && colorStr[1] == 'g' && colorStr[2] == 'b') {
67 /* Integer functional — rgb(rrr, ggg, bbb) */
68 char *s = colorStr;
69 /* replace characters we aren't using with spaces */
70 for (; *s; s++) {
71 if (*s == 'r' || *s == 'g' || *s == 'b' || *s == ','
72 || *s == '(' || *s == ')') {
73 *s = ' ';
74 }
75 }
76 c.r = (unsigned char)strtol(colorStr, &pEnd, 10);
77 c.g = (unsigned char)strtol(pEnd, &pEnd, 10);
78 c.b = (unsigned char)strtol(pEnd, &pEnd, 10);
79 }
80 else /* Color keyword */ {
81 tableColor = threadColor(colorStr, SVG_Colors);
82 if (tableColor < 0) {
83 printf("SVG color string not found: %s.\n", colorStr);
84 c.r = 0;
85 c.g = 0;
86 c.b = 0;
87 }
88 else {
89 c.r = (tableColor >> 16) & 0xFF;
90 c.g = (tableColor >> 8) & 0xFF;
91 c.b = tableColor & 0xFF;
92 }
93 }
94
95 safe_free(colorStr);
96 /* Returns black if all else fails */
97 return c;
98}

163void
164parse_line(EmbPattern *p)
165{
166 char *x1, *x2, *y1, *y2;
167 x1 = svgAttribute_getValue("x1");
168 y1 = svgAttribute_getValue("y1");
169 x2 = svgAttribute_getValue("x2");
170 y2 = svgAttribute_getValue("y2");
171
172 /* If the starting and ending points are the same, it is a point */
173 if (!strcmp(x1, x2) && !strcmp(y1, y2)) {
174 EmbPoint point;
175 point.position.x = atof(x1);
176 point.position.y = atof(y1);
177 embPattern_addPointAbs(p, point);
178 }
179 else {
180 EmbLine line;
181 line.start.x = atof(x1);
182 line.start.y = atof(y1);
183 line.end.x = atof(x2);
184 line.end.y = atof(y2);
185 embPattern_addLineAbs(p, line);
186 }

504

506 if (!polybuff) {
507 printf("ERROR: svgAddToPattern(), cannot allocate memory for polybuff\n");
508 return pointList;
509 }
510 for (i = 0; i < last; i++) {

512 switch (c) {
513 case ' ':

515 break;
516 }

518 pos = 0;
519 /*Compose Point List */
520 if (odd) {
521 odd = 0;
522 xx = atof(polybuff);
523 }

525 EmbPoint a;
526 odd = 1;
527 yy = atof(polybuff);
528
529 if (!pointList) {
530 pointList = embArray_create(EMB_POINT);
531 }
532 a.position.x = xx;
533 a.position.y = yy;
534 embArray_addPoint(pointList, a);
535 }
536
537 break;
538 default:
539 polybuff[pos++] = (char)c;
540 break;
541 }
542 if (pos >= size - 1) {
543 /* increase polybuff length - leave room for 0 */
544 size *= 2;
545 polybuff = (char*)realloc(polybuff, size);
546 if (!polybuff) {
547 printf("ERROR: svgAddToPattern(), cannot re-allocate memory for polybuff\n");
548 return NULL;
549 }
550 }
551 }
552 safe_free(polybuff);
553 return pointList;
554}

647int svgIsElement(const char* buff) {
648 if (stringInArray(buff, svg_element_tokens)) {
649 return SVG_ELEMENT;
650 }
651 /* Attempt to identify the program that created the SVG file.
652 * This should be in a comment at that occurs before the svg element. */
653 else if (!strcmp(buff, "Embroidermodder")) {
654 svgCreator = SVG_CREATOR_EMBROIDERMODDER;
655 }
656 else if (!strcmp(buff, "Illustrator")) {
657 svgCreator = SVG_CREATOR_ILLUSTRATOR;
658 }
659 else if (!strcmp(buff, "Inkscape")) {
660 svgCreator = SVG_CREATOR_INKSCAPE;
661 }
662
663 return SVG_NULL;
664}

684void
685svgProcess(int c, const char* buff)
686{
687 if (svgExpect == SVG_EXPECT_ELEMENT) {
688 char advance = 0;
689 if (buff[0] == '/') {
690 return;
691 }
692 advance = (char)svgIsElement(buff);
693 if (advance) {
694 printf("ELEMENT:\n");
695 svgExpect = SVG_EXPECT_ATTRIBUTE;
696 current_element_id = svg_identify_element((char*)buff);
697 } else {
698 return;
699 }
700 } else if (svgExpect == SVG_EXPECT_ATTRIBUTE) {
701 char advance = 0;
702 switch (current_element_id) {
703 case ELEMENT_A:
704 case ELEMENT_CIRCLE:
705 case ELEMENT_DEFS:
706 case ELEMENT_ELLIPSE:
707 case ELEMENT_FOREIGN_OBJECT:
708 case ELEMENT_G:
709 case ELEMENT_LINE:
710 case ELEMENT_LINEAR_GRADIENT:
711 case ELEMENT_PATH:
712 case ELEMENT_POLYGON:
713 case ELEMENT_POLYLINE:
714 case ELEMENT_RADIAL_GRADIENT:
715 case ELEMENT_RECT:
716 case ELEMENT_SOLID_COLOR:
717 case ELEMENT_STOP:
718 case ELEMENT_SVG:
719 case ELEMENT_SWITCH:
720 case ELEMENT_TEXT:
721 case ELEMENT_TEXT_AREA:
722 case ELEMENT_TSPAN:
723 case ELEMENT_USE:
724 if (stringInArray(buff, svg_property_tokens)) {
725 advance = SVG_PROPERTY;
726 }
727 break;
728 case ELEMENT_ANIMATION:
729 case ELEMENT_AUDIO:
730 case ELEMENT_DESC:
731 case ELEMENT_IMAGE:
732 case ELEMENT_METADATA:
733 case ELEMENT_TITLE:
734 case ELEMENT_VIDEO:
735 if (stringInArray(buff, svg_media_property_tokens)) {
736 advance = SVG_MEDIA_PROPERTY;
737 }
738 break;
739 default: break;
740 }
741 if (!advance) {
742 if (stringInArray(buff, (const char **)svg_attribute_table[current_element_id])) {
743 advance = SVG_ATTRIBUTE;
744 }
745 printf("ERROR %s not found in svg_attribute_table[%d].\n",
746 buff, current_element_id);
747 }
748 if (advance) {
749 printf("ATTRIBUTE:\n");
750 svgExpect = SVG_EXPECT_VALUE;
751 strcpy(currentAttribute, buff);
752 }
753 } else if (svgExpect == SVG_EXPECT_VALUE) {
754 int last = strlen(buff) - 1;
755 printf("VALUE:\n");
756
757 /* single-value */
758 if ((buff[0] == '"' || buff[0] == '\'') && (buff[last] == '/' || buff[last] == '"' || buff[last] == '\'') && !svgMultiValue) {
759 svgExpect = SVG_EXPECT_ATTRIBUTE;
760 strcpy(attributeList[n_attributes].name, currentAttribute);
761 strcpy(attributeList[n_attributes].value, buff);
762 n_attributes++;
763 } else { /* multi-value */
