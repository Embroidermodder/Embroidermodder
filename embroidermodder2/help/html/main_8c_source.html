243void
244parseDIFATSectors(FILE* file, bcf_file* bcfFile)
245{
246 unsigned int difatEntriesToRead = bcfFile->header.numberOfFATSectors - difatEntriesInHeader;
247 unsigned int difatSectorNumber = bcfFile->header.firstDifatSectorLocation;

261int
262bcfFile_read(FILE* file, bcf_file* bcfFile)
263{
264 unsigned int i, numberOfDirectoryEntriesPerSector;
265 unsigned int directorySectorToReadFrom;
266
267 bcfFile->header = bcfFileHeader_read(file);
268 if (memcmp(bcfFile->header.signature, "\xd0\xcf\x11\xe0\xa1\xb1\x1a\xe1", 8) != 0) {
269 printf("bad header signature\n");
270 printf("Failed to parse header\n");
271 return 0;
272 }
273
274 bcfFile->difat = bcf_difat_create(file, bcfFile->header.numberOfFATSectors, sectorSize(bcfFile));
275 if (haveExtraDIFATSectors(bcfFile)) {
276 parseDIFATSectors(file, bcfFile);
277 }
278
279 bcfFile->fat = bcfFileFat_create(sectorSize(bcfFile));
280 for (i = 0; i < bcfFile->header.numberOfFATSectors; ++i) {
281 unsigned int fatSectorNumber = bcfFile->difat->fatSectorEntries[i];
282 seekToSector(bcfFile, file, fatSectorNumber);
283 loadFatFromSector(bcfFile->fat, file);
284 }
285
286 numberOfDirectoryEntriesPerSector = sectorSize(bcfFile) / sizeOfDirectoryEntry;
287 bcfFile->directory = CompoundFileDirectory(numberOfDirectoryEntriesPerSector);
288 directorySectorToReadFrom = bcfFile->header.firstDirectorySectorLocation;
289 while (directorySectorToReadFrom != CompoundFileSector_EndOfChain) {
290 seekToSector(bcfFile, file, directorySectorToReadFrom);
291 readNextSector(file, bcfFile->directory);
292 directorySectorToReadFrom = bcfFile->fat->fatEntries[directorySectorToReadFrom];
293 }
294
295 return 1;
296}

306FILE*
307GetFile(bcf_file* bcfFile, FILE* file, char* fileToFind)
308{
309 int filesize, sectorSize, currentSector;
310 int sizeToWrite, currentSize, totalSectors, i, j;
311 FILE* fileOut = tmpfile();
312 bcf_directory_entry* pointer = bcfFile->directory->dirEntries;
313 while (pointer) {
314 if (strcmp(fileToFind, pointer->directoryEntryName) == 0)
315 break;
316 pointer = pointer->next;
317 }
318 filesize = pointer->streamSize;
319 sectorSize = bcfFile->difat->sectorSize;
320 currentSize = 0;
321 currentSector = pointer->startingSectorLocation;
322 totalSectors = (int)ceil((float)filesize / sectorSize);
323 for (i = 0; i < totalSectors; i++) {
324 seekToSector(bcfFile, file, currentSector);
325 sizeToWrite = filesize - currentSize;
326 if (sectorSize < sizeToWrite) {
327 sizeToWrite = sectorSize;
328 }
329 for (j=0; j<sizeToWrite; j++) {
330 char input;
331 fread(&input, 1, 1, file);
332 fwrite(&input, 1, 1, fileOut);
333 }
334 currentSize += sizeToWrite;
335 currentSector = bcfFile->fat->fatEntries[currentSector];
336 }
337 return fileOut;
338}

362bcf_file_difat*
363bcf_difat_create(FILE* file, unsigned int fatSectors, const unsigned int sectorSize)
364{
365 unsigned int i;
366 bcf_file_difat* difat = 0;
367 unsigned int sectorRef;
368
369 difat = (bcf_file_difat*)malloc(sizeof(bcf_file_difat));
370 if (!difat) {
371 printf("ERROR: compound-file-difat.c bcf_difat_create(), cannot allocate memory for difat\n");
372 return NULL;
373 }
374
375 difat->sectorSize = sectorSize;
376 if (fatSectors > difatEntriesInHeader) {
377 fatSectors = difatEntriesInHeader;
378 }
379
380 for (i = 0; i < fatSectors; ++i) {
381 embInt_read(file, "sectorRef", &sectorRef, EMB_INT32_LITTLE);
382 difat->fatSectorEntries[i] = sectorRef;
383 }
384 difat->fatSectorCount = fatSectors;
385 for (i = fatSectors; i < difatEntriesInHeader; ++i) {
386 embInt_read(file, "sectorRef", &sectorRef, EMB_INT32_LITTLE);
387 if (sectorRef != CompoundFileSector_FreeSector) {
388 printf("ERROR: compound-file-difat.c bcf_difat_create(), Unexpected sector value %x at DIFAT[%d]\n", sectorRef, i);
389 }
390 }
391 return difat;
392}

414unsigned int
415readFullSector(FILE* file, bcf_file_difat* bcfFile,
416 unsigned int* difatEntriesToRead)
417{
418 unsigned int i;
419 unsigned int sectorRef;
420 unsigned int nextDifatSectorInChain;
421 unsigned int entriesToReadInThisSector = 0;
422 if (*difatEntriesToRead > entriesInDifatSector(bcfFile)) {
423 entriesToReadInThisSector = entriesInDifatSector(bcfFile);
424 *difatEntriesToRead -= entriesToReadInThisSector;
425 }
426 else {
427 entriesToReadInThisSector = *difatEntriesToRead;
428 *difatEntriesToRead = 0;
429 }
430
431 for (i = 0; i < entriesToReadInThisSector; ++i) {
432 embInt_read(file, "sectorRef", &sectorRef, EMB_INT32_LITTLE);
433 bcfFile->fatSectorEntries[bcfFile->fatSectorCount] = sectorRef;
434 bcfFile->fatSectorCount++;
435 }
436 for (i = entriesToReadInThisSector; i < entriesInDifatSector(bcfFile); ++i) {
437 embInt_read(file, "sectorRef", &sectorRef, EMB_INT32_LITTLE);
438 if (sectorRef != CompoundFileSector_FreeSector) {
439 printf("ERROR: compound-file-difat.c readFullSector(), ");
440 printf("Unexpected sector value %x at DIFAT[%d]]\n", sectorRef, i);
441 }
442 }
443 embInt_read(file, "nextDifatSectorInChain", &nextDifatSectorInChain, EMB_INT32_LITTLE);
444 return nextDifatSectorInChain;
445}

514bcf_directory_entry*
515CompoundFileDirectoryEntry(FILE* file)
516{
517 const int guidSize = 16;
518 bcf_directory_entry* dir = malloc(sizeof(bcf_directory_entry));
519 if (dir == NULL) {
520 printf("ERROR: compound-file-directory.c CompoundFileDirectoryEntry(), cannot allocate memory for dir\n");
521 return 0;
522 }
523 memset(dir->directoryEntryName, 0, 32);
524 parseDirectoryEntryName(file, dir);
525 dir->next = 0;
526 dir->directoryEntryNameLength = fread_uint16(file);
527 dir->objectType = (unsigned char)fgetc(file);
528 if ((dir->objectType != ObjectTypeStorage) && (dir->objectType != ObjectTypeStream) && (dir->objectType != ObjectTypeRootEntry)) {
529 printf("ERROR: compound-file-directory.c CompoundFileDirectoryEntry()");
530 printf(", unexpected object type: %d\n", dir->objectType);
531 return 0;
532 }
533 dir->colorFlag = (unsigned char)fgetc(file);
534 embInt_read(file, "left sibling id", &(dir->leftSiblingId), EMB_INT32_LITTLE);
535 embInt_read(file, "right sibling id", &(dir->rightSiblingId), EMB_INT32_LITTLE);
536 embInt_read(file, "child id", &(dir->childId), EMB_INT32_LITTLE);
537 fread(dir->CLSID, 1, guidSize, file);
538 embInt_read(file, "state bits", &(dir->stateBits), EMB_INT32_LITTLE);
539 dir->creationTime = parseTime(file);
540 dir->modifiedTime = parseTime(file);
541 embInt_read(file, "starting sector location", &(dir->startingSectorLocation), EMB_INT32_LITTLE);
542 /* StreamSize should really be __int64 or long long,
543 * but for our uses we should never run into an issue */
544 embInt_read(file, "dir->streamSize", &(dir->streamSize), EMB_INT32_LITTLE);
545 /* top portion of int64 */
546 embInt_read(file, "dir->streamSizeHigh", &(dir->streamSizeHigh), EMB_INT32_LITTLE);
547 return dir;
548}

680void
681embSatinOutline_generateSatinOutline(EmbArray *lines, EmbReal thickness, EmbSatinOutline* result)
682{
683 int i;
684 EmbLine line1, line2;
685 EmbSatinOutline outline;
686 EmbVector out;
687 EmbVector v1;
688 EmbVector temp;
689 EmbLine line;
690
691 EmbReal halfThickness = thickness / 2.0;
692 int intermediateOutlineCount = 2 * lines->count - 2;
693 outline.side1 = embArray_create(EMB_VECTOR);
694 if (!outline.side1) {
695 printf("ERROR: emb-satin-line.c embSatinOutline_generateSatinOutline(), cannot allocate memory for outline->side1\n");
696 return;
697 }
698 outline.side2 = embArray_create(EMB_VECTOR);
699 if (!outline.side2) {
700 printf("ERROR: emb-satin-line.c embSatinOutline_generateSatinOutline(), cannot allocate memory for outline->side2\n");
701 return;
702 }
703
704 for (i = 1; i < lines->count; i++) {
705 line.start = lines->geometry[i - 1].object.vector;
706 line.end = lines->geometry[i].object.vector;
707
708 embLine_normalVector(line, &v1, 1);
709
710 embVector_multiply(v1, halfThickness, &temp);
711 temp = embVector_add(temp, lines->geometry[i-1].object.vector);
712 embArray_addVector(outline.side1, temp);
713 temp = embVector_add(temp, lines->geometry[i].object.vector);
714 embArray_addVector(outline.side1, temp);
715
716 embVector_multiply(v1, -halfThickness, &temp);
717 temp = embVector_add(temp, lines->geometry[i - 1].object.vector);
718 embArray_addVector(outline.side2, temp);
719 temp = embVector_add(temp, lines->geometry[i].object.vector);
720 embArray_addVector(outline.side2, temp);
721 }
722
723 if (!result) {
724 printf("ERROR: emb-satin-line.c embSatinOutline_generateSatinOutline(), result argument is null\n");
725 return;
726 }
727 result->side1 = embArray_create(EMB_VECTOR);
728 if (!result->side1) {
729 printf("ERROR: emb-satin-line.c embSatinOutline_generateSatinOutline(), cannot allocate memory for result->side1\n");
730 return;
731 }
732 result->side2 = embArray_create(EMB_VECTOR);
733 if (!result->side2) {
734 printf("ERROR: emb-satin-line.c embSatinOutline_generateSatinOutline(), cannot allocate memory for result->side2\n");
735 return;
736 }
737
738 embArray_addVector(result->side1, outline.side1->geometry[0].object.vector);
739 embArray_addVector(result->side2, outline.side2->geometry[0].object.vector);
740
741 for (i = 3; i < intermediateOutlineCount; i += 2) {
742 line1.start = outline.side1->geometry[i - 3].object.vector;
743 line1.end = outline.side1->geometry[i - 2].object.vector;
744 line2.start = outline.side1->geometry[i - 1].object.vector;
745 line2.end = outline.side1->geometry[i].object.vector;
746 out = embLine_intersectionPoint(line1, line2);
747 if (emb_error) {
748 puts("No intersection point.");
749 }
750 embArray_addVector(result->side1, out);
751
752 line1.start = outline.side2->geometry[i - 3].object.vector;
753 line1.end = outline.side2->geometry[i - 2].object.vector;
754 line2.start = outline.side2->geometry[i - 1].object.vector;
755 line2.end = outline.side2->geometry[i].object.vector;
756 out = embLine_intersectionPoint(line1, line2);
757 if (emb_error) {
758 puts("No intersection point.");
759 }
760 embArray_addVector(result->side2, out);
761 }
762
763 embArray_addVector(result->side1, outline.side1->geometry[2 * lines->count - 3].object.vector);
764 embArray_addVector(result->side2, outline.side2->geometry[2 * lines->count - 3].object.vector);
765 result->length = lines->count;
766}

775EmbArray*
776embSatinOutline_renderStitches(EmbSatinOutline* result, EmbReal density)
777{
778 int i, j;
779 EmbVector currTop, currBottom, topDiff, bottomDiff, midDiff;
780 EmbVector midLeft, midRight, topStep, bottomStep;
781 EmbArray* stitches = 0;
782 int numberOfSteps;
783 EmbReal midLength;
784
785 if (!result) {
786 printf("ERROR: emb-satin-line.c embSatinOutline_renderStitches(), result argument is null\n");
787 return 0;
788 }
789
790 if (result->length > 0) {
791 for (j = 0; j < result->length - 1; j++) {
792 EmbGeometry *g10 = &(result->side1->geometry[j+0]);
793 EmbGeometry *g11 = &(result->side1->geometry[j+1]);
794 EmbGeometry *g20 = &(result->side2->geometry[j+0]);
795 EmbGeometry *g21 = &(result->side2->geometry[j+1]);
796 topDiff = embVector_subtract(g10->object.vector, g11->object.vector);
797 bottomDiff = embVector_subtract(g21->object.vector, g20->object.vector);
798
799 midLeft = embVector_average(g10->object.vector, g20->object.vector);
800 midRight = embVector_average(g11->object.vector, g21->object.vector);
801
802 midDiff = embVector_subtract(midLeft, midRight);
803 midLength = embVector_length(midDiff);
804
805 numberOfSteps = (int)(midLength * density / 200);
806 embVector_multiply(topDiff, 1.0/numberOfSteps, &topStep);
807 embVector_multiply(bottomDiff, 1.0/numberOfSteps, &bottomStep);
808 currTop = g10->object.vector;
809 currBottom = g20->object.vector;
810
811 for (i = 0; i < numberOfSteps; i++) {
812 if (!stitches) {
813 stitches = embArray_create(EMB_VECTOR);
