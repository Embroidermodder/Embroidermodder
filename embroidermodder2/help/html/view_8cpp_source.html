369void View::createGridRect()
370{
371 qreal xSpacing = mainWin->getSettingsGridSpacingX();
372 qreal ySpacing = mainWin->getSettingsGridSpacingY();
373
374 QRectF gr(0, 0, mainWin->getSettingsGridSizeX(), -mainWin->getSettingsGridSizeY());
375 //Ensure the loop will work correctly with negative numbers
376 qreal x1 = qMin(gr.left(), gr.right());
377 qreal y1 = qMin(gr.top(), gr.bottom());
378 qreal x2 = qMax(gr.left(), gr.right());
379 qreal y2 = qMax(gr.top(), gr.bottom());
380
381 gridPath = QPainterPath();
382 gridPath.addRect(gr);
383 for(qreal gx = x1; gx < x2; gx += xSpacing)
384 {
385 for(qreal gy = y1; gy < y2; gy += ySpacing)
386 {
387 gridPath.moveTo(x1,gy);
388 gridPath.lineTo(x2,gy);
389 gridPath.moveTo(gx,y1);
390 gridPath.lineTo(gx,y2);
391 }
392 }
393
394 //Center the Grid
395 QRectF gridRect = gridPath.boundingRect();
396 qreal bx = gridRect.width()/2.0;
397 qreal by = -gridRect.height()/2.0;
398 qreal cx = mainWin->getSettingsGridCenterX();
399 qreal cy = -mainWin->getSettingsGridCenterY();
400 qreal dx = cx - bx;
401 qreal dy = cy - by;
402
403 if(mainWin->getSettingsGridCenterOnOrigin())
404 gridPath.translate(-bx, -by);
405 else
406 gridPath.translate(dx, dy);
407}

602void View::drawForeground(QPainter* painter, const QRectF& rect)
603{
604 //==================================================
605 //Draw grip points for all selected objects
606 //==================================================
607
608 QPen gripPen(QColor::fromRgb(gripColorCool));
609 gripPen.setWidth(2);
610 gripPen.setJoinStyle(Qt::MiterJoin);
611 gripPen.setCosmetic(true);
612 painter->setPen(gripPen);
613 QPoint gripOffset(gripSize, gripSize);
614
615 QList<QPointF> selectedGripPoints;
616 QList<QGraphicsItem*> selectedItemList = gscene->selectedItems();
617 if(selectedItemList.size() <= 100)
618 {
619 foreach(QGraphicsItem* item, selectedItemList)
620 {
621 if(item->type() >= OBJ_TYPE_BASE)
622 {
623 tempBaseObj = static_cast<BaseObject*>(item);
624 if(tempBaseObj) { selectedGripPoints = tempBaseObj->allGripPoints(); }
625
626 foreach(QPointF ssp, selectedGripPoints)
627 {
628 QPoint p1 = mapFromScene(ssp) - gripOffset;
629 QPoint q1 = mapFromScene(ssp) + gripOffset;
630
631 if(ssp == sceneGripPoint)
632 painter->fillRect(QRectF(mapToScene(p1), mapToScene(q1)), QColor::fromRgb(gripColorHot));
633 else
634 painter->drawRect(QRectF(mapToScene(p1), mapToScene(q1)));
635 }
636 }
637 }
638 }
639
640 //==================================================
641 //Draw the closest qsnap point
642 //==================================================
643
644 if(!selectingActive) //TODO: && findClosestSnapPoint == true
645 {
646 QPen qsnapPen(QColor::fromRgb(qsnapLocatorColor));
647 qsnapPen.setWidth(2);
648 qsnapPen.setJoinStyle(Qt::MiterJoin);
649 qsnapPen.setCosmetic(true);
650 painter->setPen(qsnapPen);
651 QPoint qsnapOffset(qsnapLocatorSize, qsnapLocatorSize);
652
653 QList<QPointF> apertureSnapPoints;
654 QList<QGraphicsItem *> apertureItemList = items(viewMousePoint.x()-qsnapApertureSize,
655 viewMousePoint.y()-qsnapApertureSize,
656 qsnapApertureSize*2,
657 qsnapApertureSize*2);
658 foreach(QGraphicsItem* item, apertureItemList)
659 {
660 if(item->type() >= OBJ_TYPE_BASE)
661 {
662 tempBaseObj = static_cast<BaseObject*>(item);
663 if(tempBaseObj) { apertureSnapPoints << tempBaseObj->mouseSnapPoint(sceneMousePoint); }
664 }
665 }
666 //TODO: Check for intersection snap points and add them to the list
667 foreach(QPointF asp, apertureSnapPoints)
668 {
669 QPoint p1 = mapFromScene(asp) - qsnapOffset;
670 QPoint q1 = mapFromScene(asp) + qsnapOffset;
671 painter->drawRect(QRectF(mapToScene(p1), mapToScene(q1)));
672 }
673 }

675 //==================================================
676 //Draw horizontal and vertical rulers
677 //==================================================
678
679 if(gscene->property(ENABLE_RULER).toBool())
680 {
681 bool proceed = true;
682
683 int vw = width(); //View Width
684 int vh = height(); //View Height
685 QPointF origin = mapToScene(0,0);
686 QPointF rulerHoriz = mapToScene(vw,rulerPixelSize);
687 QPointF rulerVert = mapToScene(rulerPixelSize,vh);
688
689 qreal ox = origin.x();
690 qreal oy = origin.y();
691
692 qreal rhx = rulerHoriz.x();
693 qreal rhy = rulerHoriz.y();
694 qreal rhw = rhx - ox;
695 qreal rhh = rhy - oy;
696
697 qreal rvx = rulerVert.x();
698 qreal rvy = rulerVert.y();
699 qreal rvw = rvx - ox;
700 qreal rvh = rvy - oy;
701
702 //NOTE: Drawing ruler if zoomed out too far will cause an assertion failure.
703 // We will limit the maximum size the ruler can be shown at.
704 quint16 maxSize = -1; //Intentional underflow
705 if(rhw >= maxSize || rvh >= maxSize) proceed = false;
706
707 if(proceed)
708 {
709 int distance = mapToScene(rulerPixelSize*3, 0).x() - ox;
710 QString distStr = QString().setNum(distance);
711 int distStrSize = distStr.size();
712 int msd = distStr.at(0).digitValue(); //Most Significant Digit
713
714 if(msd != -1)
715 {
716
717 msd++;
718 if(msd == 10)
719 {
720 msd = 1;
721 distStr.resize(distStrSize+1);
722 distStrSize++;
723 }
724
725 distStr.replace(0, 1, QString().setNum(msd));
726 for(int i = 1; i < distStrSize; ++i)
727 {
728 distStr.replace(i, 1, '0');
729 }
730 int unit = distStr.toInt();

1067int View::roundToMultiple(bool roundUp, int numToRound, int multiple)
1068{
1069 if(multiple == 0)
1070 return numToRound;
1071 int remainder = numToRound % multiple;
1072 if(remainder == 0)
1073 return numToRound;
1074
1075 if(numToRound < 0 && roundUp)
1076 return numToRound - remainder;
1077 if(roundUp)
1078 return numToRound + multiple - remainder;
1079 //else round down
1080 if(numToRound < 0 && !roundUp)
1081 return numToRound - multiple - remainder;
1082 return numToRound - remainder;
1083}

1521void View::mouseMoveEvent(QMouseEvent* event)
1522{
1523 updateMouseCoords(event->x(), event->y());
1524 movePoint = event->pos();
1525 sceneMovePoint = mapToScene(movePoint);
1526
1527 if(mainWin->isCommandActive())
1528 {
1529 if(rapidMoveActive)
1530 {
1531 mainWin->runCommandMove(mainWin->activeCommand(), sceneMovePoint.x(), sceneMovePoint.y());
1532 }
1533 }
1534 if(previewActive)
1535 {
1536 if(previewMode == PREVIEW_MODE_MOVE)
1537 {
1538 previewObjectItemGroup->setPos(sceneMousePoint - previewPoint);
1539 }
1540 else if(previewMode == PREVIEW_MODE_ROTATE)
1541 {
1542 qreal x = previewPoint.x();
1543 qreal y = previewPoint.y();
1544 qreal rot = previewData;
1545
1546 qreal mouseAngle = QLineF(x, y, sceneMousePoint.x(), sceneMousePoint.y()).angle();
1547
1548 qreal rad = radians(rot-mouseAngle);
1549 qreal cosRot = qCos(rad);
1550 qreal sinRot = qSin(rad);
1551 qreal px = 0;
1552 qreal py = 0;
1553 px -= x;
1554 py -= y;
1555 qreal rotX = px*cosRot - py*sinRot;
1556 qreal rotY = px*sinRot + py*cosRot;
1557 rotX += x;
1558 rotY += y;
1559
1560 previewObjectItemGroup->setPos(rotX, rotY);
1561 previewObjectItemGroup->setRotation(rot-mouseAngle);
1562 }
1563 else if(previewMode == PREVIEW_MODE_SCALE)
1564 {
1565 qreal x = previewPoint.x();
1566 qreal y = previewPoint.y();
1567 qreal scaleFactor = previewData;
1568
1569 qreal factor = QLineF(x, y, sceneMousePoint.x(), sceneMousePoint.y()).length()/scaleFactor;
1570
1571 previewObjectItemGroup->setScale(1);
1572 previewObjectItemGroup->setPos(0,0);
1573
1574 if(scaleFactor <= 0.0)
1575 {
1576 QMessageBox::critical(this, QObject::tr("ScaleFactor Error"),
1577 QObject::tr("Hi there. If you are not a developer, report this as a bug. "
1578 "If you are a developer, your code needs examined, and possibly your head too."));
1579 }
1580 else
1581 {
1582 //Calculate the offset
1583 qreal oldX = 0;
1584 qreal oldY = 0;
1585 QLineF scaleLine(x, y, oldX, oldY);
1586 scaleLine.setLength(scaleLine.length()*factor);
1587 qreal newX = scaleLine.x2();
1588 qreal newY = scaleLine.y2();
1589
1590 qreal dx = newX - oldX;
1591 qreal dy = newY - oldY;
1592
1593 previewObjectItemGroup->setScale(previewObjectItemGroup->scale()*factor);
1594 previewObjectItemGroup->moveBy(dx, dy);
1595 }
1596 }
1597 }
1598 if(pastingActive)
1599 {
1600 pasteObjectItemGroup->setPos(sceneMousePoint - pasteDelta);
1601 }
1602 if(movingActive)
1603 {
1604 //Ensure that the preview is only shown if the mouse has moved.
1605 if(!previewActive)
1606 previewOn(PREVIEW_CLONE_SELECTED, PREVIEW_MODE_MOVE, scenePressPoint.x(), scenePressPoint.y(), 0);
1607 }
1608 if(selectingActive)
1609 {
1610 if(sceneMovePoint.x() >= scenePressPoint.x()) { selectBox->setDirection(1); }
1611 else { selectBox->setDirection(0); }
1612 selectBox->setGeometry(QRect(mapFromScene(scenePressPoint), event->pos()).normalized());
1613 event->accept();
1614 }

1616 {
1617 horizontalScrollBar()->setValue(horizontalScrollBar()->value() - (event->x() - panStartX));
1618 verticalScrollBar()->setValue(verticalScrollBar()->value() - (event->y() - panStartY));

1622 }
1623 gscene->update();
1624}
