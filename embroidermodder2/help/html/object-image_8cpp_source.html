174// Returns the closest snap point to the mouse point
175QPointF ImageObject::mouseSnapPoint(const QPointF& mousePoint)
176{
177 QPointF ptl = objectTopLeft(); //Top Left Corner QSnap
178 QPointF ptr = objectTopRight(); //Top Right Corner QSnap
179 QPointF pbl = objectBottomLeft(); //Bottom Left Corner QSnap
180 QPointF pbr = objectBottomRight(); //Bottom Right Corner QSnap
181
182 qreal ptlDist = QLineF(mousePoint, ptl).length();
183 qreal ptrDist = QLineF(mousePoint, ptr).length();
184 qreal pblDist = QLineF(mousePoint, pbl).length();
185 qreal pbrDist = QLineF(mousePoint, pbr).length();
186
187 qreal minDist = qMin(qMin(ptlDist, ptrDist), qMin(pblDist, pbrDist));
188
189 if (minDist == ptlDist) return ptl;
190 else if(minDist == ptrDist) return ptr;
191 else if(minDist == pblDist) return pbl;
192 else if(minDist == pbrDist) return pbr;
193
194 return scenePos();
195}
