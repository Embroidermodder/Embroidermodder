93void
94embGeometry_move(EmbGeometry *obj, EmbVector delta)
95{
96 switch (obj->type) {
97 case EMB_ARC: {
98 EmbArc *arc = &(obj->object.arc);
99 arc->start = embVector_add(arc->start, delta);
100 arc->mid = embVector_add(arc->mid, delta);
101 arc->end = embVector_add(arc->end, delta);
102 return;
103 }
104 case EMB_CIRCLE: {
105 EmbCircle *circle = &(obj->object.circle);
106 circle->center = embVector_add(circle->center, delta);
107 return;
108 }
109 default:
110 break;
111 }
