24EmbPattern*
25embPattern_create(void)
26{
27 EmbPattern* p = (EmbPattern*)malloc(sizeof(EmbPattern));

51void
52embPattern_hideStitchesOverLength(EmbPattern* p, int length)
53{
54 EmbVector prev;
55 int i;
56 prev.x = 0.0;
57 prev.y = 0.0;
58
59 if (!p) {
60 printf("ERROR: emb-pattern.c embPattern_hideStitchesOverLength(), ");
61 printf("p argument is null\n");
62 return;
63 }
64 for (i = 0; i < p->stitch_list->count; i++) {
65 if ((fabs(p->stitch_list->stitch[i].x - prev.x) > length)
66 || (fabs(p->stitch_list->stitch[i].y - prev.y) > length)) {
67 p->stitch_list->stitch[i].flags |= TRIM;
68 p->stitch_list->stitch[i].flags &= ~NORMAL;
69 }
70 prev.x = p->stitch_list->stitch[i].x;
71 prev.y = p->stitch_list->stitch[i].y;
72 }
73}

139void
140embPattern_copystitch_listToPolylines(EmbPattern* p)
141{
142 int breakAtFlags, i;
143 EmbPoint point;
144 EmbColor color;
145
146 if (!p) {
147 printf("ERROR: emb-pattern.c embPattern_copystitch_listToPolylines(), ");
148 printf("p argument is null\n");
149 return;
150 }
151
152 breakAtFlags = (STOP | JUMP | TRIM);
153
154 for (i = 0; i < p->stitch_list->count; i++) {
155 EmbArray *pointList = 0;
156 for (; i < p->stitch_list->count; i++) {
157 EmbStitch st = p->stitch_list->stitch[i];
158 if (st.flags & breakAtFlags) {
159 break;
160 }
161 if (!(st.flags & JUMP)) {
162 if (!pointList) {
163 pointList = embArray_create(EMB_POINT);
164 color = p->thread_list->thread[st.color].color;
165 }
166 point.position.x = st.x;
167 point.position.y = st.y;
168 embArray_addPoint(pointList, point);
169 }
170 }
171
172 /* NOTE: Ensure empty polylines are not created. This is critical. */
173 if (pointList) {
174 EmbPolyline currentPolyline;
175 currentPolyline.pointList = pointList;
176 currentPolyline.color = color;
177 /* TODO: Determine what the correct value should be */
178 currentPolyline.lineType = 1;
179
180 embArray_addPolyline(p->geometry, currentPolyline);
181 }
182 }

186void
187embPattern_copyPolylinesTostitch_list(EmbPattern* p)
188{
189 int firstObject = 1, i, j;
190 /*int currentColor = polyList->polylineObj->color TODO: polyline color */
191
192 if (!p) {
193 printf("ERROR: emb-pattern.c embPattern_copyPolylinesTostitch_list(), ");
194 printf("p argument is null\n");
195 return;
196 }
197 for (i = 0; i < p->geometry->count; i++) {
198 EmbPolyline currentPoly;
199 EmbArray* currentPointList;
200 EmbThread thread;
201
202 if (p->geometry->geometry[i].type != EMB_POLYLINE) {
203 continue;
204 }
205
206 currentPoly = p->geometry->geometry[i].object.polyline;
207 currentPointList = currentPoly.pointList;
208
209 strcpy(thread.catalogNumber, "");
210 thread.color = currentPoly.color;
211 strcpy(thread.description, "");
212 embPattern_addThread(p, thread);
213
214 if (!firstObject) {
215 embPattern_addStitchAbs(p,
216 currentPointList->geometry[0].object.point.position.x,
217 currentPointList->geometry[0].object.point.position.y, TRIM, 1);
218 embPattern_addStitchRel(p, 0.0, 0.0, STOP, 1);
219 }
220
221 embPattern_addStitchAbs(p,
222 currentPointList->geometry[0].object.point.position.x,
223 currentPointList->geometry[0].object.point.position.y,
224 JUMP,
225 1);
226 for (j = 1; j < currentPointList->count; j++) {
227 EmbVector v = currentPointList->geometry[j].object.point.position;
228 embPattern_addStitchAbs(p, v.x, v.y, NORMAL, 1);
229 }
230 firstObject = 0;
231 }
232 embPattern_addStitchRel(p, 0.0, 0.0, END, 1);

261void
262embPattern_addStitchAbs(EmbPattern* p, EmbReal x, EmbReal y,
263 int flags, int isAutoColorIndex)
264{
265 EmbStitch s;
266
267 if (!p) {
268 printf("ERROR: emb-pattern.c embPattern_addStitchAbs(), ");
269 printf("p argument is null\n");
270 return;
271 }
272
273 if (flags & END) {
274 if (p->stitch_list->count == 0) {
275 return;
276 }
277 /* Prevent unnecessary multiple END stitches */
278 if (p->stitch_list->stitch[p->stitch_list->count - 1].flags & END) {
279 printf("ERROR: emb-pattern.c embPattern_addStitchAbs(), found multiple END stitches\n");
280 return;
281 }
282 embPattern_fixColorCount(p);
283 /* HideStitchesOverLength(127); TODO: fix or remove this */
284 }
285
286 if (flags & STOP) {
287 if (p->stitch_list->count == 0) {
288 return;
289 }
290 if (isAutoColorIndex) {
291 p->currentColorIndex++;
292 }
293 }
294
295 /* NOTE: If the stitch_list is empty, we will create it before adding
296 stitches to it. The first coordinate will be the HOME position. */
297 if (p->stitch_list->count == 0) {
298 /* NOTE: Always HOME the machine before starting any stitching */
299 EmbStitch h;
300 h.x = p->home.x;
301 h.y = p->home.y;
302 h.flags = JUMP;
303 h.color = p->currentColorIndex;
304 embArray_addStitch(p->stitch_list, h);
305 }
306 s.x = x;
307 s.y = y;
308 s.flags = flags;
309 s.color = p->currentColorIndex;
310 embArray_addStitch(p->stitch_list, s);
311}

316embPattern_addStitchRel(EmbPattern* p, EmbReal dx, EmbReal dy,
317 int flags, int isAutoColorIndex)
318{
319 EmbReal x, y;
320 if (!p) {
321 printf("ERROR: emb-pattern.c embPattern_addStitchRel(), p argument is null\n");
322 return;
323 }
324 if (p->stitch_list->count > 0) {
325 EmbStitch st = p->stitch_list->stitch[p->stitch_list->count - 1];
326 x = st.x + dx;
327 y = st.y + dy;
328 } else {
329 /* NOTE: The stitch_list is empty, so add it to the HOME position.
330 * The embstitch_list_create function will ensure the first coordinate is at the HOME position. */
331 x = p->home.x + dx;
332 y = p->home.y + dy;
333 }
334 embPattern_addStitchAbs(p, x, y, flags, isAutoColorIndex);
335}

371EmbRect
372embPattern_calcBoundingBox(EmbPattern* p)
373{
374 EmbRect r;
375 EmbStitch pt;
376 int i, j;
377
378 r.left = 0;
379 r.right = 0;
380 r.top = 0;
381 r.bottom = 0;
382
383 if (!p) {
384 printf("ERROR: emb-pattern.c embPattern_calcBoundingBox(), ");
385 printf("p argument is null\n");
386 return r;
387 }
388
389 /* Calculate the bounding rectangle. It's needed for smart repainting. */
390 /* TODO: Come back and optimize this mess so that after going thru all objects
391 and stitches, if the rectangle isn't reasonable, then return a default rect */
392 if ((p->stitch_list->count == 0) && (p->geometry->count == 0)) {
393 r.top = 0.0;
394 r.left = 0.0;
395 r.bottom = 1.0;
396 r.right = 1.0;
397 return r;
398 }
399 r.left = 99999.0;
400 r.top = 99999.0;
401 r.right = -99999.0;
402 r.bottom = -99999.0;
403
404 for (i = 0; i < p->stitch_list->count; i++) {
405 /* If the point lies outside of the accumulated bounding
406 * rectangle, then inflate the bounding rect to include it. */
407 pt = p->stitch_list->stitch[i];
408 if (!(pt.flags & TRIM)) {
409 r.left = EMB_MIN(r.left, pt.x);
410 r.top = EMB_MIN(r.top, pt.y);
411 r.right = EMB_MAX(r.right, pt.x);
412 r.bottom = EMB_MAX(r.bottom, pt.y);
413 }
414 }
415
416 for (i = 0; i < p->geometry->count; i++) {
417 EmbGeometry g = p->geometry->geometry[i];
418 switch (g.type) {
419 case EMB_ARC: {
420 /* TODO: embPattern_calcBoundingBox for arcs,
421 for now just checks the start point */
422 EmbArc arc = g.object.arc;
423 r.left = EMB_MIN(r.left, arc.start.x);
424 r.top = EMB_MIN(r.top, arc.start.y);
425 r.right = EMB_MAX(r.right, arc.start.x);
426 r.bottom = EMB_MAX(r.bottom, arc.start.y);
427 break;
428 }
429 case EMB_CIRCLE: {
430 EmbCircle circle = g.object.circle;
431 r.left = EMB_MIN(r.left, circle.center.x - circle.radius);
432 r.top = EMB_MIN(r.top, circle.center.y - circle.radius);
433 r.right = EMB_MAX(r.right, circle.center.x + circle.radius);
434 r.bottom = EMB_MAX(r.bottom, circle.center.y + circle.radius);
435 break;
436 }
437 case EMB_ELLIPSE: {
438 /* TODO: account for rotation */
439 EmbEllipse ellipse = g.object.ellipse;
440 r.left = EMB_MIN(r.left, ellipse.center.x - ellipse.radius.x);
441 r.top = EMB_MIN(r.top, ellipse.center.y - ellipse.radius.y);
442 r.right = EMB_MAX(r.right, ellipse.center.x + ellipse.radius.x);
443 r.bottom = EMB_MAX(r.bottom, ellipse.center.y + ellipse.radius.y);
444 break;
445 }
446 case EMB_LINE: {
447 EmbLine line = g.object.line;
448 r.left = EMB_MIN(r.left, line.start.x);
449 r.left = EMB_MIN(r.left, line.end.x);
450 r.top = EMB_MIN(r.top, line.start.y);
451 r.top = EMB_MIN(r.top, line.end.y);
452 r.right = EMB_MAX(r.right, line.start.x);
453 r.right = EMB_MAX(r.right, line.end.x);
454 r.bottom = EMB_MAX(r.bottom, line.start.y);
455 r.bottom = EMB_MAX(r.bottom, line.end.y);
456 break;
457 }
458 case EMB_POINT: {
459 EmbVector point = g.object.point.position;
460 r.left = EMB_MIN(r.left, point.x);
461 r.top = EMB_MIN(r.top, point.y);
462 r.right = EMB_MAX(r.right, point.x);
463 r.bottom = EMB_MAX(r.bottom, point.y);
464 break;
465 }
466 case EMB_POLYGON: {
467 EmbArray *polygon = g.object.polygon.pointList;
468 for (j=0; j < polygon->count; j++) {
469 /* TODO: embPattern_calcBoundingBox for polygons */
470 }
471 break;
472 }
473 case EMB_POLYLINE: {
474 EmbArray *polyline = g.object.polyline.pointList;
475 for (j=0; j < polyline->count; j++) {
476 /* TODO: embPattern_calcBoundingBox for polylines */
477 }
478 break;
479 }
480 case EMB_RECT: {
481 EmbRect rect = g.object.rect;
482 r.left = EMB_MIN(r.left, rect.left);
483 r.top = EMB_MIN(r.top, rect.top);
484 r.right = EMB_MAX(r.right, rect.right);
485 r.bottom = EMB_MAX(r.bottom, rect.bottom);
486 break;
487 }
488 case EMB_SPLINE: {
489 /* EmbBezier bezier;
490 bezier = p->splines->spline[i].bezier; */
491 /* TODO: embPattern_calcBoundingBox for splines */
492 break;
493 }
494 default:
495 break;
496 }
497 }
498
499 return r;
500}

655void
656embPattern_combineJumpStitches(EmbPattern* p)
657{
658 int jumpCount = 0, i;
659 EmbArray *newList;
660 EmbStitch j;
661
662 if (!p) {
663 printf("ERROR: emb-pattern.c embPattern_combineJumpStitches(), ");
664 printf("p argument is null\n");
665 return;
666 }
667 newList = embArray_create(EMB_STITCH);
668 for (i = 0; i < p->stitch_list->count; i++) {
669 EmbStitch st = p->stitch_list->stitch[i];
670 if (st.flags & JUMP) {
671 if (jumpCount == 0) {
672 j = st;
673 } else {
674 j.x += st.x;
675 j.y += st.y;
676 }
677 jumpCount++;
678 } else {
679 if (jumpCount > 0) {
680 embArray_addStitch(newList, j);
681 }
682 embArray_addStitch(newList, st);
683 }
684 }
685 embArray_free(p->stitch_list);
686 p->stitch_list = newList;
687}

689/*TODO: The params determine the max XY movement rather than the length.
690 They need renamed or clarified further. */
691void
692embPattern_correctForMaxStitchLength(EmbPattern* p,
693 EmbReal maxStitchLength, EmbReal maxJumpLength)
694{
695 if (!p) {
696 printf("ERROR: emb-pattern.c embPattern_correctForMaxStitchLength(), ");
697 printf("p argument is null\n");
698 return;
699 }
700 if (p->stitch_list->count > 1) {
701 int i, j, splits;
702 EmbReal maxXY, maxLen, addX, addY;
703 EmbArray *newList = embArray_create(EMB_STITCH);
704 for (i=1; i < p->stitch_list->count; i++) {
705 EmbStitch st = p->stitch_list->stitch[i];
706 EmbReal xx = st.x;
707 EmbReal yy = st.y;
708 EmbReal dx = p->stitch_list->stitch[i-1].x - xx;
709 EmbReal dy = p->stitch_list->stitch[i-1].y - yy;
710 if ((fabs(dx) > maxStitchLength) || (fabs(dy) > maxStitchLength)) {
711 maxXY = EMB_MAX(fabs(dx), fabs(dy));
712 if (st.flags & (JUMP | TRIM)) {
713 maxLen = maxJumpLength;
714 } else {
715 maxLen = maxStitchLength;
716 }
717 splits = (int)ceil((EmbReal)maxXY / maxLen);
718
719 if (splits > 1) {
720 addX = (EmbReal)dx / splits;
721 addY = (EmbReal)dy / splits;
722
723 for (j = 1; j < splits; j++) {
724 EmbStitch s;
725 s = st;
726 s.x = xx + addX * j;
727 s.y = yy + addY * j;
728 embArray_addStitch(newList, s);
729 }
730 }
731 }
732 embArray_addStitch(newList, st);
733 }
734 embArray_free(p->stitch_list);
735 p->stitch_list = newList;
736 }
737 embPattern_end(p);
738}
