155int
156embFormat_getExtension(const char *fileName, char *ending)
157{
158 int i;
159 const char *offset;
160
161 if (!fileName) {
162 printf("ERROR: emb-format.c embFormat_getExtension(), fileName argument is null\n");
163 return 0;
164 }
165
166 if (strlen(fileName) == 0) {
167 return 0;
168 }
169
170 offset = strrchr(fileName, '.');
171 if (offset==0) {
172 return 0;
173 }
174
175 i = 0;
176 while (offset[i] != '\0') {
177 ending[i] = tolower(offset[i]);
178 ++i;
179 }
180 ending[i] = 0; /* terminate the string */
181 return 1;
182}

190int
191emb_identify_format(const char *fileName)
192{
193 int i;
194 char ending[5];
195 if (!embFormat_getExtension(fileName, ending)) {
196 return 0;
197 }
198 for (i = 0; i < numberOfFormats; i++) {
199 if (!strcmp(ending, formatTable[i].extension)) {
200 return i;
201 }
202 }
203 return -1;
204}
