14char
15readT09(EmbPattern* pattern, FILE* file)
16{
17 unsigned char b[3];
18
19 fseek(file, 0x0C, SEEK_SET);
20
21 while (fread(b, 1, 3, file) == 3) {
22 int stitchType = NORMAL;
23 int b1 = b[0];
24 int b2 = b[1];
25 unsigned char commandByte = b[2];
26 if (commandByte == 0x00) {
27 break;
28 }
29 if (commandByte & 0x10) {
30 stitchType = STOP;
31 }
32 if (commandByte & 0x20) {
33 b1 = -b1;
34 }
35 if (commandByte & 0x40) {
36 b2 = -b2;
37 }
38 embPattern_addStitchRel(pattern, b2 / 10.0, b1 / 10.0, stitchType, 1);
39 }
40 return 1;
41}

43char
44writeT09(EmbPattern* pattern, FILE* file)
45{
46 int i;
47 EmbVector pos;
48 fpad(file, 0x00, 0x0C);
49
50 pos.x = 0.0;
51 pos.y = 0.0;
52 for (i = 0; i < pattern->stitch_list->count; i++) {
53 unsigned char b[3];
54 int dx, dy;
55 EmbStitch st = pattern->stitch_list->stitch[i];
56 /* convert from mm to 0.1mm for file format */
57 dx = (int)emb_round(10.0*(st.x - pos.x));
58 dy = (int)emb_round(10.0*(st.y - pos.y));
59 pos.x += 0.1*dx;
60 pos.y += 0.1*dy;
61
62 b[0] = dx;
63 b[1] = dy;
64 b[2] = 0;
65 if (st.flags & NORMAL) {
66 /* guessing based on T01 here */
67 b[2] |= 0x03;
68 }
69 /* inferred from readT09 */
70 if (st.flags & STOP) {
71 b[2] |= 0x10;
72 }
73 if (dx < 0) {
74 b[0] = -dx;
75 b[2] |= 0x20;
76 }
77 if (dy < 0) {
78 b[0] = -dy;
79 b[2] |= 0x40;
80 }
81 fwrite(b, 1, 3, file);
82 }
83 return 1;
84}
