15void
16readPecStitches(EmbPattern* pattern, FILE* file)
17{
18 FILE *f = file;
19 unsigned char b[2];
20
21 while (fread(b, 1, 2, f)==2) {
22 int val1 = (int)b[0];
23 int val2 = (int)b[1];
24
25 int stitchType = NORMAL;
26 if (b[0] == 0xFF && b[1] == 0x00) {
27 embPattern_end(pattern);
28 return;
29 }
30 if (b[0] == 0xFE && b[1] == 0xB0) {
31 (void)fgetc(f);
32 embPattern_addStitchRel(pattern, 0.0, 0.0, STOP, 1);
33 continue;
34 }
35 /* High bit set means 12-bit offset, otherwise 7-bit signed delta */
36 if (val1 & 0x80) {
37 if (val1 & 0x20) stitchType = TRIM;
38 if (val1 & 0x10) stitchType = JUMP;
39 val1 = ((val1 & 0x0F) << 8) + val2;
40
41 /* Signed 12-bit arithmetic */
42 if (val1 & 0x800) {
43 val1 -= 0x1000;
44 }
45 }
46 else if (val1 >= 0x40) {
47 val1 -= 0x80;
48 }
49 if (val2 & 0x80) {
50 if (val2 & 0x20) stitchType = TRIM;
51 if (val2 & 0x10) stitchType = JUMP;
52 val2 = ((val2 & 0x0F) << 8) + fgetc(file);
53
54 /* Signed 12-bit arithmetic */
55 if (val2 & 0x800) {
56 val2 -= 0x1000;
57 }
58 }
59 else if (val2 >= 0x40) {
60 val2 -= 0x80;
61 }
62
63 embPattern_addStitchRel(pattern, val1 / 10.0,
64 val2 / 10.0, stitchType, 1);
65 }
66}

68void
69pecEncodeJump(FILE* file, int x, int types)
70{
71 int outputVal = abs(x) & 0x7FF;
72 unsigned int orPart = 0x80;
73 unsigned char toWrite;
74
75 if (!file) {
76 printf("ERROR: format-pec.c pecEncodeJump(), file argument is null\n");
77 return;
78 }
79 if (types & TRIM) {
80 orPart |= 0x20;
81 }
82 if (types & JUMP) {
83 orPart |= 0x10;
84 }
85 if (x < 0) {
86 outputVal = (x + 0x1000) & 0x7FF;
87 outputVal |= 0x800;
88 }
89 toWrite = (unsigned char)(((outputVal >> 8) & 0x0F) | orPart);
90 fwrite(&toWrite, 1, 1, file);
91 toWrite = (unsigned char)(outputVal & 0xFF);
92 fwrite(&toWrite, 1, 1, file);
93}

106char
107readPec(EmbPattern* pattern, const char *fileName, FILE* file)
108{
109 unsigned int graphicsOffset;
110 unsigned char colorChanges;
111 int i;
112
113 if (emb_verbose>1) {
114 printf("fileName: %s\n", fileName);
115 }
116
117 if (!check_header_present(file, 0x20A)) {
118 return 0;
119 }
120
121 fseek(file, 0x38, SEEK_SET);
122 colorChanges = (unsigned char)(char)fgetc(file);
123 for (i = 0; i <= colorChanges; i++) {
124 embPattern_addThread(pattern, pecThreads[(char)fgetc(file) % 65]);
125 }
126
127 /* Get Graphics offset */
128 fseek(file, 0x20A, SEEK_SET);
129
130 graphicsOffset = (unsigned int)(fgetc(file));
131 graphicsOffset |= (fgetc(file) << 8);
132 graphicsOffset |= (fgetc(file) << 16);
133
134 (void)(char)fgetc(file); /* 0x31 */
135 (void)(char)fgetc(file); /* 0xFF */
136 (void)(char)fgetc(file); /* 0xF0 */
137 /* Get X and Y size in .1 mm */
138 /* 0x210 */
139 fread_int16(file); /* x size */
140 fread_int16(file); /* y size */
141
142 fread_int16(file); /* 0x01E0 */
143 fread_int16(file); /* 0x01B0 */
144 fread_int16(file); /* distance left from start */
145 fread_int16(file); /* distance up from start */
146
147 /* Begin Stitch Data */
148 /* 0x21C */
149 /*unsigned int end = graphicsOffset + 0x208; */
150 readPecStitches(pattern, file);
151 embPattern_flipVertical(pattern);
152 return 1;
153}

155void
156pecEncode(FILE* file, EmbPattern* p)
157{
158 EmbReal thisX = 0.0;
159 EmbReal thisY = 0.0;
160 unsigned char stopCode = 2;
161 int i;
162
163 if (!file) {
164 printf("ERROR: format-pec.c pecEncode(), file argument is null\n");
165 return;
166 }
167 if (!p) {
168 printf("ERROR: format-pec.c pecEncode(), p argument is null\n");
169 return;
170 }
171
172 for (i = 0; i < p->stitch_list->count; i++) {
173 int deltaX, deltaY;
174 EmbStitch s = p->stitch_list->stitch[i];
175
176 deltaX = (int)emb_round(s.x - thisX);
177 deltaY = (int)emb_round(s.y - thisY);
178 thisX += (EmbReal)deltaX;
179 thisY += (EmbReal)deltaY;
180
181 if (s.flags & STOP) {
182 pecEncodeStop(file, stopCode);
183 if (stopCode == (unsigned char)2) {
184 stopCode = (unsigned char)1;
185 } else {
186 stopCode = (unsigned char)2;
187 }
188 } else if (s.flags & END) {
189 fwrite("\xFF", 1, 1, file);
190 break;
191 } else if (deltaX < 63 && deltaX > -64 && deltaY < 63 && deltaY > -64 && (!(s.flags & (JUMP | TRIM)))) {
192 unsigned char out[2];
193 if (deltaX < 0) {
194 out[0] = (unsigned char)(deltaX + 0x80);
195 }
196 else {
197 out[0] = (unsigned char)deltaX;
198 }
199 if (deltaY < 0) {
200 out[1] = (unsigned char)(deltaY + 0x80);
201 }
202 else {
203 out[1] = (unsigned char)deltaY;
204 }
205 fwrite(out, 1, 2, file);
206 }
207 else {
208 pecEncodeJump(file, deltaX, s.flags);
209 pecEncodeJump(file, deltaY, s.flags);
210 }
211 }
212}

216void
217writePecStitches(EmbPattern* pattern, FILE* file, const char *fileName)
218{
219 EmbRect bounds;
220 unsigned char image[38][48], toWrite;
221 int i, j, flen, graphicsOffsetLocation;
222 int graphicsOffsetValue, height, width;
223 EmbReal xFactor, yFactor;
224 const char* forwardSlashPos = strrchr(fileName, '/');
225 const char* backSlashPos = strrchr(fileName, '\\');
226 const char* dotPos = strrchr(fileName, '.');
227 const char* start = 0;
228
229 start = fileName;
230 if (forwardSlashPos) {
231 start = forwardSlashPos + 1;
232 }
233 if (backSlashPos && backSlashPos > start) {
234 start = backSlashPos + 1;
235 }
236 fwrite("LA:", 1, 3, file);
237 flen = (int)(dotPos - start);
238
239 while (start < dotPos) {
240 fwrite(start, 1, 1, file);
241 start++;
242 }
243 fpad(file, 0x20, 16-flen);
244 fwrite("\x0D", 1, 1, file);
245 fpad(file, 0x20, 12);
246 fwrite("\xff\x00\x06\x26", 1, 4, file);
247
248 fpad(file, 0x20, 12);
249 toWrite = (unsigned char)(pattern->thread_list->count-1);
250 fwrite(&toWrite, 1, 1, file);
251
252 for (i = 0; i < pattern->thread_list->count; i++) {
253 EmbColor thr = pattern->thread_list->thread[i].color;
254 unsigned char color = (unsigned char)
255 embThread_findNearestThread(thr,
256 (EmbThread*)pecThreads, pecThreadCount);
257 fwrite(&color, 1, 1, file);
258 }
259 fpad(file, 0x20, (int)(0x1CF - pattern->thread_list->count));
260 fpad(file, 0x00, 2);
261
262 graphicsOffsetLocation = ftell(file);
263 /* placeholder bytes to be overwritten */
264 fpad(file, 0x00, 3);
265
266 fwrite("\x31\xff\xf0", 1, 3, file);
267
268 bounds = embPattern_calcBoundingBox(pattern);
269
270 height = (int)emb_round(bounds.bottom - bounds.top);
271 width = (int)emb_round(bounds.right - bounds.left);
272 /* write 2 byte x size */
273 binaryWriteShort(file, (short)width);
274 /* write 2 byte y size */
275 binaryWriteShort(file, (short)height);
276
277 /* Write 4 miscellaneous int16's */
278 fwrite("\x01\xe0\x01\xb0", 1, 4, file);
279
280 binaryWriteUShortBE(file, (unsigned short)(0x9000 | -(int)emb_round(bounds.left)));
281 binaryWriteUShortBE(file, (unsigned short)(0x9000 | -(int)emb_round(bounds.top)));
282
283 pecEncode(file, pattern);
284 graphicsOffsetValue = ftell(file) - graphicsOffsetLocation + 2;
285 fseek(file, graphicsOffsetLocation, SEEK_SET);
286
287 fputc((unsigned char)(graphicsOffsetValue & 0xFF), file);
288 fputc((unsigned char)((graphicsOffsetValue >> 8) & 0xFF), file);
289 fputc((unsigned char)((graphicsOffsetValue >> 16) & 0xFF), file);
290
291 fseek(file, 0x00, SEEK_END);
292
293 /* Writing all colors */
294 memcpy(image, imageWithFrame, 48*38);
295
296 yFactor = 32.0 / height;
297 xFactor = 42.0 / width;
298 for (i = 0; i < pattern->stitch_list->count; i++) {
299 EmbStitch st = pattern->stitch_list->stitch[i];
300 int x = (int)emb_round((st.x - bounds.left) * xFactor) + 3;
301 int y = (int)emb_round((st.y - bounds.top) * yFactor) + 3;
302 if (x<=0 || x>48) continue;
303 if (y<=0 || y>38) continue;
304 image[y][x] = 1;
305 }
306 writeImage(file, image);
307
308 /* Writing each individual color */
309 j = 0;
310 for (i = 0; i < pattern->thread_list->count; i++) {
311 memcpy(image, imageWithFrame, 48*38);
312 for (; j < pattern->stitch_list->count; j++) {
313 EmbStitch st = pattern->stitch_list->stitch[j];
314 int x = (int)emb_round((st.x - bounds.left) * xFactor) + 3;
315 int y = (int)emb_round((st.y - bounds.top) * yFactor) + 3;
316 if (x<=0 || x>48) continue;
317 if (y<=0 || y>38) continue;
318 if (st.flags & STOP) {
319 break;
320 }
321 image[y][x] = 1;
322 }
323 writeImage(file, image);
324 }
325}

327char
328writePec(EmbPattern* pattern, const char* fileName, FILE* file)
329{
330 /* TODO: There needs to be a matching flipVertical() call after the write
331 to ensure multiple writes from the same pattern work properly */

333 embPattern_fixColorCount(pattern);
334 embPattern_correctForMaxStitchLength(pattern, 12.7, 204.7);

337 writePecStitches(pattern, file, fileName);
