75void PolylineObject::updateRubber(QPainter* painter)
76{
77 int rubberMode = objectRubberMode();
78 if(rubberMode == OBJ_RUBBER_POLYLINE)
79 {
80 setObjectPos(objectRubberPoint("POLYLINE_POINT_0"));
81
82 QLineF rubberLine(normalPath.currentPosition(), mapFromScene(objectRubberPoint(QString())));
83 if(painter) drawRubberLine(rubberLine, painter, VIEW_COLOR_CROSSHAIR);
84
85 bool ok = false;
86 QString numStr = objectRubberText("POLYLINE_NUM_POINTS");
87 if(numStr.isNull()) return;
88 int num = numStr.toInt(&ok);
89 if(!ok) return;
90
91 QString appendStr;
92 QPainterPath rubberPath;
93 for(int i = 1; i <= num; i++)
94 {
95 appendStr = "POLYLINE_POINT_" + QString().setNum(i);
96 QPointF appendPoint = mapFromScene(objectRubberPoint(appendStr));
97 rubberPath.lineTo(appendPoint);
98 }
99 updatePath(rubberPath);
100
101 //Ensure the path isn't updated until the number of points is changed again
102 setObjectRubberText("POLYLINE_NUM_POINTS", QString());
103 }
104 else if(rubberMode == OBJ_RUBBER_GRIP)
105 {
106 if(painter)
107 {
108 int elemCount = normalPath.elementCount();
109 QPointF gripPoint = objectRubberPoint("GRIP_POINT");
110 if(gripIndex == -1) gripIndex = findIndex(gripPoint);
111 if(gripIndex == -1) return;
112
113 if(!gripIndex) //First
114 {
115 QPainterPath::Element ef = normalPath.elementAt(1);
116 QPointF efPoint = QPointF(ef.x, ef.y);
117 painter->drawLine(efPoint, mapFromScene(objectRubberPoint(QString())));
118 }
119 else if(gripIndex == elemCount-1) //Last
120 {
121 QPainterPath::Element el = normalPath.elementAt(gripIndex-1);
122 QPointF elPoint = QPointF(el.x, el.y);
123 painter->drawLine(elPoint, mapFromScene(objectRubberPoint(QString())));
124 }
125 else //Middle
126 {
127 QPainterPath::Element em = normalPath.elementAt(gripIndex-1);
128 QPainterPath::Element en = normalPath.elementAt(gripIndex+1);
129 QPointF emPoint = QPointF(em.x, em.y);
130 QPointF enPoint = QPointF(en.x, en.y);
131 painter->drawLine(emPoint, mapFromScene(objectRubberPoint(QString())));
132 painter->drawLine(enPoint, mapFromScene(objectRubberPoint(QString())));
133 }
134
135 QLineF rubLine(mapFromScene(gripPoint), mapFromScene(objectRubberPoint(QString())));
136 drawRubberLine(rubLine, painter, VIEW_COLOR_CROSSHAIR);
137 }

153QPointF PolylineObject::mouseSnapPoint(const QPointF& mousePoint)
154{
155 QPainterPath::Element element = normalPath.elementAt(0);
156 QPointF closestPoint = mapToScene(QPointF(element.x, element.y));
157 qreal closestDist = QLineF(mousePoint, closestPoint).length();
158 int elemCount = normalPath.elementCount();
159 for(int i = 0; i < elemCount; ++i)
160 {
161 element = normalPath.elementAt(i);
162 QPointF elemPoint = mapToScene(element.x, element.y);
163 qreal elemDist = QLineF(mousePoint, elemPoint).length();
164 if(elemDist < closestDist)
165 {
166 closestPoint = elemPoint;
167 closestDist = elemDist;
168 }
169 }
170 return closestPoint;
171}

186{
187 int elemCount = normalPath.elementCount();
188 //NOTE: Points here are in item coordinates

190 for(int i = 0; i < elemCount; i++)
191 {
192 QPainterPath::Element e = normalPath.elementAt(i);
193 QPointF elemPoint = QPointF(e.x, e.y);
