1113/* Using stdarg we can pack arguments into the context using the above functions.
1114 *
1115 * https://pubs.opengroup.org/onlinepubs/009695399/basedefs/stdarg.h.html
1116 */
1117ScriptEnv *
1118pack(ScriptEnv *context, const char *fmt, ...)
1119{
1120 va_list a;
1121 int argno;
1122 context->argumentCount = 0;
1123 va_start(a, fmt);
1124 for (argno = 0; fmt[argno]; argno++) {
1125 switch (fmt[argno]) {
1126 case 's': {
1127 const char *s = va_arg(a, const char *);
1128 strncpy(context->argument[context->argumentCount].s, s, MAX_STRING_LENGTH);
1129 context->argument[context->argumentCount].type = SCRIPT_STRING;
1130 break;
1131 }
1132 case 'i': {
1133 int i = va_arg(a, int);
1134 context->argument[context->argumentCount].i = i;
1135 context->argument[context->argumentCount].type = SCRIPT_INT;
1136 break;
1137 }
1138 case 'r': {
1139 EmbReal r = va_arg(a, double);
1140 context->argument[context->argumentCount].r = r;
1141 context->argument[context->argumentCount].type = SCRIPT_REAL;
1142 break;
1143 }
1144 default:
1145 break;
1146 }
1147 context->argumentCount++;
1148 }
1149 va_end(a);
1150 return context;
1151}

1169 for (argno = 0; fmt[argno]; argno++) {
1170 switch (fmt[argno]) {
1171 case 's': {
1172 const char *s = va_arg(a, const char *);
1173 strncpy(context->argument[context->argumentCount].s, s, MAX_STRING_LENGTH);
1174 context->argument[context->argumentCount].type = SCRIPT_STRING;
1175 break;
1176 }
1177 case 'i': {
1178 int i = va_arg(a, int);
1179 context->argument[context->argumentCount].i = i;
1180 context->argument[context->argumentCount].type = SCRIPT_INT;

1265/* Rather than loading necessary configuration data from file at load, it is
1266 * compiled into the program. However, the ability to change the UI as a
1267 * user, without re-compiling the program, can be preserved by overriding the string
1268 * tables from a custom "overrides.ini" file in the configuration directory.
1269 */
1270int
1271load_data(void)
1272{
1273 /* Load config_table into config ScriptValues. */
1274 for (n_variables=0; ; n_variables++) {
1275 if (string_equal(config_table[n_variables], END_SYMBOL)) {
1276 break;
1277 }
1278 }
1279 return 1;
1280 config = (ScriptValue*)malloc(n_variables * sizeof(ScriptValue));
1281 for (int line_no=0; line_no<n_variables; line_no++) {
1282 char line[200];
1283 strcpy(line, config_table[line_no]);
1284 int eq_pos = 0;
1285 for (int i=0; i<200; i++) {
1286 if (line[i] == '=') {
1287 eq_pos = i;
1288 break;
1289 }
1290 }
1291 line[eq_pos] = 0;
1292 strcpy(config[line_no].label, line);
1293 strcpy(config[line_no].s, line+eq_pos+1);
1294 }
1295
1296 /* Identify the labels of commands. */
1297 for (int i=0; i<n_variables; i++) {
1298 int offset = strlen(config[i].label) - 4;
1299 if (string_equal(config[i].label + offset, "type")) {
1300 if (string_equal(config[i].s, "command")) {
1301 char label[MAX_STRING_LENGTH];
1302 snprintf(label, offset, "%s", config[i].label);
1303 load_command_data(label);
1304 }
1305 }
1306 }
1307
1308 /* If the overrides file exists, load it on top of the default data. */
1309 FILE *file = fopen("overrides.ini", "r");
1310 if (file) {
1311 fclose(file);
1312 char *data = malloc(n_variables*100);
1313 data = load_file("overrides.ini");
1314 free(data);
1315 }
1316 return 1;
1317}
