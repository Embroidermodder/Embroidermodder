14char
15readSst(EmbPattern* pattern, FILE* file)
16{
17 int fileLength;
18
19 fseek(file, 0, SEEK_END);
20 fileLength = ftell(file);
21 fseek(file, 0xA0, SEEK_SET); /* skip the all zero header */
22 while (ftell(file) < fileLength) {
23 int stitchType = NORMAL;
24
25 int b1 = fgetc(file);
26 int b2 = fgetc(file);
27 unsigned char commandByte = (unsigned char)fgetc(file);
28
29 if (commandByte == 0x04) {
30 embPattern_addStitchRel(pattern, 0, 0, END, 1);
31 break;
32 }
33
34 if ((commandByte & 0x01) == 0x01)
35 stitchType = STOP;
36 if ((commandByte & 0x02) == 0x02)
37 stitchType = JUMP;
38 if ((commandByte & 0x10) != 0x10)
39 b2 = -b2;
40 if ((commandByte & 0x40) == 0x40)
41 b1 = -b1;
42 embPattern_addStitchRel(pattern, b1 / 10.0, b2 / 10.0, stitchType, 1);
43 }
44
45 return 1; /*TODO: finish readSst */
46}
