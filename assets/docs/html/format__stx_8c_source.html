116 /* public Bitmap Image; */

157 embInt_read(file, ".", &left, EMB_INT16_LITTLE);
158 embInt_read(file, ".", &bottom, EMB_INT16_LITTLE);
159 embInt_read(file, ".", &top, EMB_INT16_LITTLE);
160 if (emb_verbose>1) {
161 printf("paletteLength: %d\n", paletteLength);
162 printf("imageLength: %d\n", imageLength);
163 printf("something1: %d\n", something1);
164 printf("stitchDataOffset: %d\n", stitchDataOffset);

177 printf("ERROR: format-stx.c readStx(), unable to allocate memory for gif\n");

181 /*Stream s2 = new MemoryStream(gif); TODO: review */
182 /*Image = new Bitmap(s2); TODO: review */

185 stxThreads = (StxThread*)malloc(sizeof(StxThread) * threadCount);
186 if (!stxThreads) {
187 printf("ERROR: format-stx.c readStx(), unable ");
188 printf("to allocate memory for stxThreads\n");
189 return 0;
190 }
191 for (i = 0; i < threadCount; i++) {
192 EmbThread t;
193 StxThread st;
194 stxReadThread(&st, file);
195
196 t.color = st.stxColor;
197 strcpy(t.description, st.colorName);
198 strcpy(t.catalogNumber, st.colorCode);
199 embPattern_addThread(pattern, t);
200 stxThreads[i] = st;
201 }
202
203 fseek(file, 15, SEEK_CUR);
204
205 for (i = 0; i < 12; i++) {
206 embInt_read(file, ".", val+i, EMB_INT16_LITTLE);
207 if (emb_verbose>1) {
208 printf("identify val[%d] = %d", i, val[i]);
209 }
210 }
211 if (emb_verbose>1) {
212 puts("val[4] == val[5] == 0");
213 puts("val[10] == val[11] == 0");
214 }
215 fseek(file, 8, SEEK_CUR); /* 0 0 */
216 /* br.BaseStream.Position = stitchDataOffset; TODO: review */
217 for (i = 1; i < stitchCount; ) {
218 char b0 = (char)fgetc(file);
219 char b1 = (char)fgetc(file);
220 if (b0 == -128) {
221 switch (b1) {
222 case 1:
223 b0 = (char)fgetc(file);
224 b1 = (char)fgetc(file);
225 /*embPattern_addStitchRel(b0, b1, STOP); TODO: review */
226
227 i++;
228 break;
229 case 2:
230 b0 = (char)fgetc(file);
231 b1 = (char)fgetc(file);
232 embPattern_addStitchRel(pattern, b0 / 10.0,
233 b1 / 10.0, JUMP, 1);
234 i++;
235 break;
236 case -94:
237 /* TODO: Is this a synchronize?
238 If so document it in the comments. */
239 break;
240 default:
241 /*Debugger.Break(); TODO: review */
242 break;
243 }
244 } else {
245 embPattern_addStitchRel(pattern, b0 / 10.0, b1 / 10.0, NORMAL, 1);
246 i++;
247 }
248 }
249 embPattern_flipVertical(pattern);
250 return 1;
251}
