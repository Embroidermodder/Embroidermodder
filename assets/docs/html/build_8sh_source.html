78function code_style () {
79
80 # Remove all trailing blank space.
81 sed -i -r "s/[[:blank:]]*$//g" src/main.c
82
83 # Tabs to 4 space indentation. No tab symbol should appear in the code.
84 sed -i -r "s/\t/ /g" src/main.c
85
86 # Do not use
87 # ----------
88 # Internally we use snake_case rather than camelCase for most tokens,
89 # however this will be difficult to detect which non-conforming tokens have
90 # to be present.
91 # s/ \‍([a-z0-9]\‍)\‍([A-Z]\‍)/ \1_\L\2/g
92
93 # Use unix \n line endings not Windows \r\n line endings for source code.
94 # All other uses of \r should also be removed.
95 sed -i -r "s/\r//g" src/main.c
96
97 # Remove all non printing characters in C source code.
98 # (Unicode only appears in the configuration files.)
99 sed -i -r "s/[^[:print:]]//g" src/main.c
100
101}

200function debian_setup () {
201
202cat <<EOF
203Version Info
204------------
205
206* MAJOR.txt - Contains the major release version number.
207This will be updated to 2 when Embroidermodder 2 is final.
208Until then, it is set to 1 while Embroidermodder 2 is in development.
209
210* MINOR.txt - Contains the minor release version number.
211This file will be updated frequently when minor releases happen.
212It will be set to zero when Embroidermodder 2 is final.
213Until then, it is set to 90+ while Embroidermodder 2 is in development.
214The number shall be 2 digits, padded with a zero for single digit numbers.
215
216* PATCH.txt - Contains the patch release version number.
217This file will always be zero. Nightly builds will not use this file and
218will use the build unix date as such: ```date +%Y%m%d%H%M%S```
219This ensures that the latest build is always the highest number.
220EOF
221
222 mkdir version
223 echo 1 > version/MAJOR.txt
224 echo 90 > version/MINOR.txt
225 echo 0 > version/PATCH.txt
226
227 mkdir DEBIAN
228 cat <<EOF > DEBIAN/control
229Package: embroidermodder
