15char
16readU00(EmbPattern* pattern, FILE* file)
17{
18 int i;
19 char dx = 0, dy = 0;
20 int flags = NORMAL;
21 unsigned char b[3];
22
23 if (!check_header_present(file, 0x100)) {
24 return 0;
25 }
26
27 /* 16 3byte RGB's start @ 0x08 followed by 14 bytes between
28 0 and 15 with index of color for each color change */
29 fseek(file, 0x08, SEEK_SET);
30 for (i = 0; i < 16; i++) {
31 EmbThread t;
32 embColor_read(file, &(t.color), 3);
33 strcpy(t.catalogNumber, "");
34 strcpy(t.description, "");
35 embPattern_addThread(pattern, t);
36 }
37
38 fseek(file, 0x100, SEEK_SET);
39 while (fread(b, 1, 3, file) == 3) {
40 char negativeX , negativeY;
41
42 if (b[0] == 0xF8 || b[0] == 0x87 || b[0] == 0x91) {
43 break;
44 }
45 if ((b[0] & 0x0F) == 0) {
46 flags = NORMAL;
47 } else if ((b[0] & 0x1f) == 1) {
48 flags = JUMP;
49 } else if ((b[0] & 0x0F) > 0) {
50 flags = STOP;
51 }
52 negativeX = ((b[0] & 0x20) > 0);
53 negativeY = ((b[0] & 0x40) > 0);
54
55 dx = (char) b[2];
56 dy = (char) b[1];
57 if (negativeX) {
58 dx = (char) -dx;
59 }
60 if (negativeY) {
61 dy = (char) -dy;
62 }
63 embPattern_addStitchRel(pattern, dx / 10.0, dy / 10.0, flags, 1);
64 }
65 return 1;
66}
