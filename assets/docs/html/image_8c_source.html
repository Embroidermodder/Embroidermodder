31void
32writeImage(FILE* file, unsigned char image[][48])
33{
34 int i, j;
35
36 if (!file) {
37 printf("ERROR: format-pec.c writeImage(), file argument is null\n");
38 return;
39 }
40 for (i = 0; i < 38; i++) {
41 for (j = 0; j < 6; j++) {
42 int offset = j * 8;
43 unsigned char output = 0;
44 output |= (unsigned char)(image[i][offset] != 0);
45 output |= (unsigned char)(image[i][offset + 1] != (unsigned char)0) << 1;
46 output |= (unsigned char)(image[i][offset + 2] != (unsigned char)0) << 2;
47 output |= (unsigned char)(image[i][offset + 3] != (unsigned char)0) << 3;
48 output |= (unsigned char)(image[i][offset + 4] != (unsigned char)0) << 4;
49 output |= (unsigned char)(image[i][offset + 5] != (unsigned char)0) << 5;
50 output |= (unsigned char)(image[i][offset + 6] != (unsigned char)0) << 6;
51 output |= (unsigned char)(image[i][offset + 7] != (unsigned char)0) << 7;
52 fwrite(&output, 1, 1, file);
53 }
54 }
55}
