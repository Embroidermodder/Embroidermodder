19int husDecodeStitchType(unsigned char b)
20{
21 switch(b) {
22 case 0x80:
23 return NORMAL;
24 case 0x81:
25 return JUMP;
26 case 0x84:
27 return STOP;
28 case 0x90:
29 return END;
30 default:
31 return NORMAL;
32 }
33}

35unsigned char*
36husDecompressData(unsigned char* input, int compressedInputLength, int decompressedContentLength)
37{
38 char* decompressedData = (char*)malloc(sizeof(char)*decompressedContentLength);
39 if (!decompressedData) { printf("ERROR: husDecompressData(), cannot allocate memory for decompressedData\n"); return 0; }
40 hus_decompress((char*)input, compressedInputLength, decompressedData, &decompressedContentLength);
41 return (unsigned char *)decompressedData;
42}

67{

70 return (0x80);
71 case JUMP:
72 case TRIM:
73 return (0x81);
74 case STOP:

83char
84readHus(EmbPattern* pattern, FILE* file)
85{
86 int fileLength;
87 int magicCode, numberOfStitches, numberOfColors;
88 int positiveXHoopSize, positiveYHoopSize, negativeXHoopSize, negativeYHoopSize;
89 int attributeOffset, xOffset, yOffset;
90 unsigned char *attributeData, *xData , *yData;
91 unsigned char* attributeDataDecompressed = 0;
92 unsigned char* xDecompressed = 0;
93 unsigned char* yDecompressed = 0;
94 unsigned char* stringVal = 0;
95
96 int unknown, i = 0;
97
98 puts("ERROR: Overridden. readHus is not implemented for now.");
99 return 0;
100
101 fseek(file, 0x00, SEEK_END);
102 fileLength = ftell(file);
103 fseek(file, 0x00, SEEK_SET);
104
105 embInt_read(file, "magicCode", &magicCode, EMB_INT32_LITTLE);
106 embInt_read(file, "numberOfStitchs", &numberOfStitches, EMB_INT32_LITTLE);
107 embInt_read(file, "numberOfColors", &numberOfColors, EMB_INT32_LITTLE);
108
109 /* These should be put in the EmbPattern. */
110 embInt_read(file, "+X HoopSize", &positiveXHoopSize, EMB_INT16_LITTLE);
111 embInt_read(file, "+Y HoopSize", &positiveYHoopSize, EMB_INT16_LITTLE);
112 embInt_read(file, "-X HoopSize", &negativeXHoopSize, EMB_INT16_LITTLE);
113 embInt_read(file, "-Y HoopSize", &negativeYHoopSize, EMB_INT16_LITTLE);
114
115 embInt_read(file, "attributeOffset", &attributeOffset, EMB_INT32_LITTLE);
116 embInt_read(file, "xOffset", &xOffset, EMB_INT32_LITTLE);
117 embInt_read(file, "yOffset", &yOffset, EMB_INT32_LITTLE);
118
119 stringVal = (unsigned char*)malloc(sizeof(unsigned char)*8);
120 if (!stringVal) {
121 printf("ERROR: format-hus.c readHus(), cannot allocate memory for stringVal\n");
122 return 0;
123 }
124 fread(stringVal, 1, 8, file); /* TODO: check return value */
125
126 embInt_read(file, "unknown", &unknown, EMB_INT16_LITTLE);
127 for (i = 0; i < numberOfColors; i++) {
128 short pos;
129 embInt_read(file, "pos", &pos, EMB_INT16_LITTLE);
130 embPattern_addThread(pattern, husThreads[pos]);
131 }
132
133 attributeData = (unsigned char*)malloc(sizeof(unsigned char)*(xOffset - attributeOffset + 1));
134 if (!attributeData) {
135 printf("ERROR: format-hus.c readHus(), cannot allocate memory for attributeData\n");
136 return 0;
137 }
138 fread(attributeData, 1, xOffset - attributeOffset, file); /* TODO: check return value */
139 attributeDataDecompressed = husDecompressData(attributeData, xOffset - attributeOffset, numberOfStitches + 1);
140
141 xData = (unsigned char*)malloc(sizeof(unsigned char)*(yOffset - xOffset + 1));
142 if (!xData) {
143 printf("ERROR: format-hus.c readHus(), cannot allocate memory for xData\n");
144 return 0;
145 }
146 fread(xData, 1, yOffset - xOffset, file); /* TODO: check return value */
147 xDecompressed = husDecompressData(xData, yOffset - xOffset, numberOfStitches);
148
149 yData = (unsigned char*)malloc(sizeof(unsigned char)*(fileLength - yOffset + 1));
150 if (!yData) {
151 printf("ERROR: format-hus.c readHus(), cannot allocate memory for yData\n");
152 return 0;
153 }
154 fread(yData, 1, fileLength - yOffset, file); /* TODO: check return value */
155 yDecompressed = husDecompressData(yData, fileLength - yOffset, numberOfStitches);
156
157 for (i = 0; i < numberOfStitches; i++) {
158 int flag;
159 EmbVector v;
160 v.x = husDecodeByte(xDecompressed[i]) / 10.0;
161 v.y = husDecodeByte(yDecompressed[i]) / 10.0;
162 flag = husDecodeStitchType(attributeDataDecompressed[i]);
163 embPattern_addStitchRel(pattern, v.x, v.y, flag, 1);
164 }
165
166 safe_free(stringVal);
167 safe_free(xData);
168 safe_free(xDecompressed);
169 safe_free(yData);
170 safe_free(yDecompressed);
171 safe_free(attributeData);
172 safe_free(attributeDataDecompressed);
173
174 return 1;
175}

177char
178writeHus(EmbPattern* pattern, FILE* file)
179{
180 EmbRect boundingRect;
181 int stitchCount, minColors, patternColor, attributeSize, xCompressedSize, yCompressedSize, i;
182 EmbReal previousX, previousY;
183 short right, top, bottom, left;
184 unsigned int code, colors, offset1, offset2;
185 unsigned char *xValues = 0, *yValues = 0, *attributeValues = 0,
186 *attributeCompressed = 0, *xCompressed = 0, *yCompressed = 0;
187
188 stitchCount = pattern->stitch_list->count;
189 /* embPattern_correctForMaxStitchLength(pattern, 0x7F, 0x7F); */
190 minColors = pattern->thread_list->count;
191 patternColor = minColors;
192 if (minColors > 24) minColors = 24;
193 code = 0x00C8AF5B;
194 embInt_write(file, "code", &code, EMB_INT32_LITTLE);
195 embInt_write(file, "stitchCount", &stitchCount, EMB_INT32_LITTLE);
196 embInt_write(file, "minColors", &minColors, EMB_INT32_LITTLE);
197
198 boundingRect = embPattern_calcBoundingBox(pattern);
199 right = (short) emb_round(boundingRect.right * 10.0);
200 top = (short) -emb_round(boundingRect.top * 10.0);
201 left = (short) emb_round(boundingRect.left * 10.0);
202 bottom = (short) -emb_round(boundingRect.bottom * 10.0);
203 embInt_write(file, "right", &right, EMB_INT16_LITTLE);
204 embInt_write(file, "top", &top, EMB_INT16_LITTLE);
205 embInt_write(file, "left", &left, EMB_INT16_LITTLE);
206 embInt_write(file, "bottom", &bottom, EMB_INT16_LITTLE);
207
208 colors = 0x2A + 2 * minColors;
209 embInt_write(file, ".", &colors, EMB_INT32_LITTLE);
210
211 xValues = (unsigned char*)malloc(sizeof(unsigned char)*(stitchCount));
212 if (!xValues) {
213 printf("ERROR: format-hus.c writeHus(), cannot allocate memory for xValues\n");
214 return 0;
215 }
216 yValues = (unsigned char*)malloc(sizeof(unsigned char)*(stitchCount));
217 if (!yValues) {
218 printf("ERROR: format-hus.c writeHus(), cannot allocate memory for yValues\n");
219 return 0;
220 }
221 attributeValues = (unsigned char*)malloc(sizeof(unsigned char)*(stitchCount));
222 if (!attributeValues) {
223 printf("ERROR: format-hus.c writeHus(), cannot allocate memory for attributeValues\n");
224 return 0;
225 }
226
227 previousX = 0.0;
228 previousY = 0.0;
229 for (i = 0; i < pattern->stitch_list->count; i++) {
230 EmbStitch st;
231 st = pattern->stitch_list->stitch[i];
232 xValues[i] = husEncodeByte((st.x - previousX) * 10.0);
233 previousX = st.x;
234 yValues[i] = husEncodeByte((st.y - previousY) * 10.0);
235 previousY = st.y;
236 attributeValues[i] = husEncodeStitchType(st.flags);
237 }
238 attributeCompressed = husCompressData(attributeValues, stitchCount, &attributeSize);
239 xCompressed = husCompressData(xValues, stitchCount, &xCompressedSize);
240 yCompressed = husCompressData(yValues, stitchCount, &yCompressedSize);
241 /* TODO: error if husCompressData returns zero? */
242
243 offset1 = (unsigned int) (0x2A + 2 * patternColor + attributeSize);
244 offset2 = (unsigned int) (0x2A + 2 * patternColor + attributeSize + xCompressedSize);
245 embInt_write(file, "offset1", &offset1, EMB_INT32_LITTLE);
246 embInt_write(file, "offset2", &offset2, EMB_INT32_LITTLE);
247 fpad(file, 0, 10);
248
249 for (i = 0; i < patternColor; i++) {
250 short color_index = (short)embThread_findNearestThread(pattern->thread_list->thread[i].color, (EmbThread*)husThreads, 29);
251 embInt_write(file, ".", &color_index, EMB_INT16_LITTLE);
252 }
253
254 fwrite(attributeCompressed, 1, attributeSize, file);
255 fwrite(xCompressed, 1, xCompressedSize, file);
256 fwrite(yCompressed, 1, yCompressedSize, file);
257
258 safe_free(xValues);
259 safe_free(xCompressed);
260 safe_free(yValues);
261 safe_free(yCompressed);
262 safe_free(attributeValues);
263 safe_free(attributeCompressed);
264
265 return 1;
266}
