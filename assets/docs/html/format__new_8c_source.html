15readNew(EmbPattern* pattern, FILE* file)
16{
17 unsigned int stitchCount;
18 unsigned char data[3];
19
20 embInt_read(file, "stitchCount", &stitchCount, EMB_INT16_LITTLE);
21 while (fread(data, 1, 3, file) == 3) {
22 int x = decodeNewStitch(data[0]);
23 int y = decodeNewStitch(data[1]);
24 int flag = NORMAL;
25 char val = data[2];
26 if (data[2] & 0x40) {
27 x = -x;
28 }
29 if (data[2] & 0x20) {
30 y = -y;
31 }
32 if (data[2] & 0x10) {
33 flag = TRIM;
34 }
35 if (data[2] & 0x01) {
36 flag = JUMP;
37 }
38 if ((val & 0x1E) == 0x02) {
39 flag = STOP;
40 }
41 /* Unknown values, possibly TRIM
42 155 = 1001 1011 = 0x9B
43 145 = 1001 0001 = 0x91
44 */
45 /*val = (data[2] & 0x1C);
46 if (val != 0 && data[2] != 0x9B && data[2] != 0x91) {
47 int z = 1;
48 }*/
49 embPattern_addStitchRel(pattern, x / 10.0, y / 10.0, flag, 1);
50 }
51
52 return 1;
53}
