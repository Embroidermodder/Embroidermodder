20char
21readExp(EmbPattern* pattern, FILE* file)
22{
23 unsigned char b[2];
24
25 while (fread(b, 1, 2, file) == 2) {
26 char dx = 0, dy = 0;
27 int flags = NORMAL;
28 if (b[0] == 0x80) {
29 if (b[1] == 0x01) {
30 if (fread(b, 1, 2, file) != 2) break;
31 /* b0=0x00 and b1=0x00, but accept any,
32 not worth crashing over. */
33 flags = STOP;
34 } else if (b[1] == 0x04) {
35 if (fread(b, 1, 2, file) != 2) {
36 break;
37 }
38 flags = JUMP;
39 } else if (b[1] == 0x80) {
40 if (fread(b, 1, 2, file) != 2) {
41 break;
42 }
43 /* b0=0x07 and b1=0x00, but accept any,
44 not worth crashing over. */
45 flags = TRIM;
46 }
47 }
48 dx = expDecode(b[0]);
49 dy = expDecode(b[1]);
50 embPattern_addStitchRel(pattern, dx / 10.0, dy / 10.0, flags, 1);
51 }
52 return 1;
53}

55char
56writeExp(EmbPattern* pattern, FILE* file) {
57 EmbVector pos;
58 int i;
59
60 /* write stitches */
61 pos.x = 0.0;
62 pos.y = 0.0;
63 for (i = 0; i < pattern->stitch_list->count; i++) {
64 char b[4];
65 char dx, dy;
66 EmbStitch st = pattern->stitch_list->stitch[i];
67 dx = (char)emb_round(10.0*(st.x - pos.x));
68 dy = (char)emb_round(10.0*(st.y - pos.y));
69 pos.x += 0.1*dx;
70 pos.y += 0.1*dy;
71 switch (st.flags) {
72 case STOP:
73 b[0] = (char)(0x80);
74 b[1] = 0x01;
75 b[2] = 0x00;
76 b[3] = 0x00;
77 fwrite(b, 1, 4, file);
78 break;
79 case JUMP:
80 b[0] = (char)(0x80);
81 b[1] = 0x04;
82 b[2] = dx;
83 b[3] = dy;
84 fwrite(b, 1, 4, file);
85 break;
86 case TRIM:
87 b[0] = (char)(0x80);
88 b[1] = (char)(0x80);
89 b[2] = 0x07;
90 b[3] = 0x00;
91 fwrite(b, 1, 4, file);
92 break;
93 default: /* STITCH */
94 b[0] = dx;
95 b[1] = dy;
96 fwrite(b, 1, 2, file);
97 break;
98 }
99 }
100 fprintf(file, "\x1a");
101 return 1;
102}
