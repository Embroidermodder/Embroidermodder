14char
15readZsk(EmbPattern* pattern, FILE* file)
16{
17 char b[3];
18 int stitchType;
19 unsigned char colorNumber;
20
21 fseek(file, 0, SEEK_END);
22 if (ftell(file) < 0x230) {
23 return 0;
24 }
25
26 fseek(file, 0x230, SEEK_SET);
27 colorNumber = fgetc(file);
28 while (colorNumber != 0) {
29 EmbThread t;
30 embColor_read(file, &(t.color), 3);
31 strcpy(t.catalogNumber, "");
32 strcpy(t.description, "");
33 embPattern_addThread(pattern, t);
34 fseek(file, 0x48, SEEK_CUR);
35 colorNumber = fgetc(file);
36 }
37 fseek(file, 0x2E, SEEK_CUR);
38
39 while (fread(b, 1, 3, file) == 3) {
40 stitchType = NORMAL;
41 if (b[0] & 0x04) {
42 b[2] = -b[2];
43 }
44 if (b[0] & 0x08) {
45 b[1] = -b[1];
46 }
47 if (b[0] & 0x02) {
48 stitchType = JUMP;
49 }
50 if (b[0] & 0x20) {
51 if (b[1] == 2) {
52 stitchType = TRIM;
53 } else if (b[1] == -1) {
54 break;
55 } else {
56 if (b[2] != 0) {
57 colorNumber = b[2];
58 }
59 /* TODO: need to determine what b[1] is used for.*/
60 stitchType = STOP;
61 embPattern_changeColor(pattern, colorNumber - 1);
62 }
63 b[1] = 0;
64 b[2] = 0;
65 }
66 embPattern_addStitchRel(pattern, b[1] / 10.0,
67 b[2] / 10.0, stitchType, 0);
68 }
69
70 return 1;
71}

74char
75writeZsk(EmbPattern* pattern, FILE* file)
76{
77 int i;
78
79 fpad(file, 0x00, 0x230);
80
81 fprintf(file, "%c", pattern->thread_list->count);
82 for (i=pattern->thread_list->count; i>0; i--) {
83 EmbThread t = pattern->thread_list->thread[i-1];
84 embColor_write(file, t.color, 3);
85 fpad(file, 0x00, 0x48);
86 fprintf(file, "%c", i-1);
87 }
88
89 fpad(file, 0x00, 0x2E);
90
91 for (i=0; i<pattern->stitch_list->count; i++) {
92 EmbStitch st;
93 unsigned char b[3];
94 st = pattern->stitch_list->stitch[i];
95 b[0] = 0;
96 b[1] = st.x;
97 b[2] = st.y;
98 if (st.flags & JUMP) {
99 b[0] |= 0x02;
100 }
101 if (st.x < 0) {
102 b[0] |= 0x08;
103 b[1] = -st.x;
104 }
105 if (st.y < 0) {
106 b[0] |= 0x04;
107 b[2] = -st.y;
108 }
109 if (st.flags & TRIM) {
110 b[0] |= 0x20;
111 b[1] = 0x00;
112 b[2] = 0x00;
113 }
114 if (st.flags & STOP) {
115 b[0] |= 0x20;
116 b[1] = 0x00;
117 b[2] = st.color;
118 }
119 if (st.flags & END) {
120 b[0] |= 0x20;
121 b[1] = 0x80;
122 b[2] = 0x00;
123 fwrite(b, 1, 3, file);
124 break;
125 }
126 fwrite(b, 1, 3, file);
127 }
128 return 1;
129}
