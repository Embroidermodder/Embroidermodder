14char
15shvDecode(unsigned char inputByte)
16{
17 if (inputByte >= 0x80) {
18 return (char)-((unsigned char)((~inputByte) + 1));
19 }
20 return ((char)inputByte);
21}

23short shvDecodeShort(unsigned short inputByte)
24{
25 if (inputByte > 0x8000) {
26 return (short)-((unsigned short)((~inputByte) + 1));
27 }
28 return ((short)inputByte);

31char
32readShv(EmbPattern* pattern, FILE* file)
33{
34 int i;
35 char inJump = 0;
36 unsigned char fileNameLength, designWidth, designHeight;
37 char halfDesignWidth, halfDesignHeight, halfDesignWidth2, halfDesignHeight2;
38 char* headerText = "Embroidery disk created using software licensed from Viking Sewing Machines AB, Sweden";
39 char dx = 0, dy = 0;
40 char numberOfColors;
41 unsigned short magicCode;
42 int something;
43 short left,top,right,bottom;
44 char something2, numberOfSections, something3;
45 int stitchesPerColor[256];
46 int stitchesSinceChange = 0;
47 int currColorIndex = 0;
48 unsigned short sx, sy;
49
50 if (!check_header_present(file, 25)) {
51 return 0;
52 }
53
54 fseek(file, strlen(headerText), SEEK_SET);
55 fileNameLength = fgetc(file);
56 fseek(file, fileNameLength, SEEK_CUR);
57 designWidth = fgetc(file);
58 designHeight = fgetc(file);
59 halfDesignWidth = fgetc(file);
60 halfDesignHeight = fgetc(file);
61 halfDesignWidth2 = fgetc(file);
62 halfDesignHeight2 = fgetc(file);
63 if ((designHeight % 2) == 1) {
64 fseek(file, ((designHeight + 1)*designWidth)/2, SEEK_CUR);
65 }
66 else {
67 fseek(file, (designHeight*designWidth)/2, SEEK_CUR);
68 }
69 numberOfColors = fgetc(file);
70 magicCode = fread_uint16(file);
71 fseek(file, 1, SEEK_CUR);
72 embInt_read(file, "something", &something, EMB_INT32_LITTLE);
73 left = fread_int16(file);
74 top = fread_int16(file);
75 right = fread_int16(file);
76 bottom = fread_int16(file);
77
78 something2 = (char)fgetc(file);
79 numberOfSections = fgetc(file);
80 something3 = (char)fgetc(file);
81
82 if (emb_verbose>1) {
83 printf("magicCode: %d\n", magicCode);
84 printf("halfDesignWidth: %d\n", halfDesignWidth);
85 printf("halfDesignHeight: %d\n", halfDesignHeight);
86 printf("halfDesignWidth2: %d\n", halfDesignWidth2);
87 printf("halfDesignHeight2: %d\n", halfDesignHeight2);
88 printf("left: %d\n", left);
89 printf("top: %d\n", top);
90 printf("right: %d\n", right);
91 printf("bottom: %d\n", bottom);
92 printf("something2: %d\n", something2);
93 printf("number of sections: %d\n", numberOfSections);
94 printf("something3: %d\n", something3);
95 }
96
97 for (i = 0; i < numberOfColors; i++) {
98 unsigned int stitchCount, colorNumber;
99 embInt_read(file, "stitchCount", &stitchCount, EMB_INT32_BIG);
100 colorNumber = fgetc(file);
101 embPattern_addThread(pattern, shvThreads[colorNumber % 43]);
102 stitchesPerColor[i] = stitchCount;
103 fseek(file, 9, SEEK_CUR);
104 }
105
106 fseek(file, -2, SEEK_CUR);
107
108 for (i = 0; !feof(file); i++) {
109 unsigned char b0, b1;
110 int flags;
111 flags = NORMAL;
112 if (inJump) {
113 flags = JUMP;
114 }
115 b0 = fgetc(file);
116 b1 = fgetc(file);
117 if (stitchesSinceChange >= stitchesPerColor[currColorIndex]) {
118 embPattern_addStitchRel(pattern, 0, 0, STOP, 1);
119 currColorIndex++;
120 stitchesSinceChange = 0;
121 }
122 if (b0 == 0x80) {
123 stitchesSinceChange++;
124 if (b1 == 3) {
125 continue;
126 }
127 else if (b1 == 0x02) {
128 inJump = 0;
129 continue;
130 }
131 else if (b1 == 0x01) {
132 stitchesSinceChange += 2;
133 sx = fgetc(file);
134 sx = (unsigned short)(sx << 8 | fgetc(file));
135 sy = fgetc(file);
136 sy = (unsigned short)(sy << 8 | fgetc(file));
137 flags = TRIM;
138 inJump = 1;
139 embPattern_addStitchRel(pattern, shvDecodeShort(sx) / 10.0, shvDecodeShort(sy) / 10.0, flags, 1);
140 continue;
141 }
142 }
143 dx = shvDecode(b0);
144 dy = shvDecode(b1);
145 stitchesSinceChange++;
146 embPattern_addStitchRel(pattern, dx / 10.0, dy / 10.0, flags, 1);
147 }
148 embPattern_flipVertical(pattern);
149
150 return 1;
151}
