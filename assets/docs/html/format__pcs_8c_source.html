15char
16readPcs(EmbPattern* pattern, const char* fileName, FILE* file)
17{
18 char allZeroColor = 1;
19 int i = 0;
20 unsigned char b[9];
21 EmbReal dx = 0, dy = 0;
22 int flags = 0, st = 0;
23 unsigned char version, hoopSize;
24 unsigned short colorCount;
25
26 version = (char)fgetc(file);
27 hoopSize = (char)fgetc(file); /* 0 for PCD, 1 for PCQ (MAXI), 2 for PCS with small hoop(80x80), */
28 /* and 3 for PCS with large hoop (115x120) */
29
30 switch(hoopSize) {
31 case 2:
32 pattern->hoop_width = 80.0;
33 pattern->hoop_height = 80.0;
34 break;
35 case 3:
36 pattern->hoop_width = 115.0;
37 pattern->hoop_height = 120.0;
38 break;
39 }
40
41 colorCount = fread_uint16(file);
42 if (emb_verbose>1) {
43 printf("version: %d\n", version);
44 printf("hoop size: %d\n", hoopSize);
45 printf("color count: %d\n", colorCount);
46 }
47
48 for (i = 0; i < colorCount; i++) {
49 EmbThread t;
50 embColor_read(file, &(t.color), 4);
51 strcpy(t.catalogNumber, "");
52 strcpy(t.description, "");
53 if (t.color.r || t.color.g || t.color.b) {
54 allZeroColor = 0;
55 }
56 embPattern_addThread(pattern, t);
57 }
58 if (allZeroColor) {
59 embPattern_loadExternalColorFile(pattern, fileName);
60 }
61 st = fread_uint16(file);
62 /* READ STITCH RECORDS */
63 for (i = 0; i < st; i++) {
64 flags = NORMAL;
65 if (fread(b, 1, 9, file) != 9)
66 break;
67
68 if (b[8] & 0x01) {
69 flags = STOP;
70 }
71 else if (b[8] & 0x04) {
72 flags = TRIM;
73 }
74 else if (b[8] != 0) {
75 /* TODO: ONLY INTERESTED IN THIS CASE TO LEARN MORE ABOUT THE FORMAT */
76 }
77 dx = pfaffDecode(b[1], b[2], b[3]);
78 dy = pfaffDecode(b[5], b[6], b[7]);
79 embPattern_addStitchAbs(pattern, dx / 10.0, dy / 10.0, flags, 1);
80 }
81
82 return 1;
83}

85char
86writePcs(EmbPattern* pattern, FILE* file)
87{
88 int i;
89
90 /* TODO: select hoop size defaulting to Large PCS hoop */
91 fwrite("2\x03", 1, 2, file);
92 binaryWriteUShort(file, (unsigned short)pattern->thread_list->count);
93 for (i = 0; i < pattern->thread_list->count; i++) {
94 EmbColor color = pattern->thread_list->thread[i].color;
95 embColor_write(file, color, 4);
96 }
97
98 /* write remaining colors to reach 16 */
99 /* fpad(file, 0, 4*(16-i)); */
100
101 binaryWriteUShort(file, (unsigned short)pattern->stitch_list->count);
102 /* write stitches */
103 for (i = 0; i < pattern->stitch_list->count; i++) {
104 EmbStitch st = pattern->stitch_list->stitch[i];
105 pfaffEncode(file, (int)emb_round(st.x * 10.0), (int)emb_round(st.y * 10.0), st.flags);
106 }
107 return 1;
108}
