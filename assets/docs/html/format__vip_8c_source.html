55 case 0x81:

78char
79readVip(EmbPattern* pattern, FILE* file)
80{
81 int fileLength;
82 int i;
83 unsigned char prevByte = 0;
84 unsigned char *attributeData = 0, *decodedColors = 0;
85 unsigned char *attributeDataDecompressed = 0;
86 unsigned char *xData = 0, *xDecompressed = 0, *yData = 0, *yDecompressed = 0;
87 VipHeader header;
88
89 fileLength = check_header_present(file, 32);
90 if (!fileLength) {
91 printf("ERROR: file shorter than header.");
92 return 0;
93 }
94
95 embInt_read(file, "magicCode", &(header.magicCode), EMB_INT32_LITTLE);
96 embInt_read(file, "numberOfStitches", &(header.numberOfStitches), EMB_INT32_LITTLE);
97 embInt_read(file, ".", &(header.numberOfColors), EMB_INT32_LITTLE);
98
99 embInt_read(file, ".", &(header.postitiveXHoopSize), EMB_INT16_LITTLE);
100 embInt_read(file, ".", &(header.postitiveYHoopSize), EMB_INT16_LITTLE);
101 embInt_read(file, ".", &(header.negativeXHoopSize), EMB_INT16_LITTLE);
102 embInt_read(file, ".", &(header.negativeYHoopSize), EMB_INT16_LITTLE);
103 embInt_read(file, ".", &(header.attributeOffset), EMB_INT32_LITTLE);
104 embInt_read(file, ".", &(header.xOffset), EMB_INT32_LITTLE);
105 embInt_read(file, ".", &(header.yOffset), EMB_INT32_LITTLE);
106
107 fread(header.stringVal, 1, 8, file); /* TODO: check return value */
108
109 embInt_read(file, ".", &(header.unknown), EMB_INT16_LITTLE);
110
111 embInt_read(file, ".", &(header.colorLength), EMB_INT32_LITTLE);
112 decodedColors = (unsigned char*)malloc(header.numberOfColors*4);
113 if (!decodedColors) {
114 printf("ERROR: format-vip.c readVip(), ");
115 printf("cannot allocate memory for decodedColors\n");
116 return 0;
117 }
118 for (i = 0; i < header.numberOfColors*4; ++i) {
119 unsigned char inputByte = (char)fgetc(file);
120 unsigned char tmpByte = (unsigned char) (inputByte ^ vipDecodingTable[i]);
121 decodedColors[i] = (unsigned char) (tmpByte ^ prevByte);
122 prevByte = inputByte;
123 }
124 for (i = 0; i < header.numberOfColors; i++) {
125 EmbThread thread;
126 int startIndex = i << 2;
127 thread.color.r = decodedColors[startIndex];
128 thread.color.g = decodedColors[startIndex + 1];
129 thread.color.b = decodedColors[startIndex + 2];
130 /* printf("%d\n", decodedColors[startIndex + 3]); */
131 embPattern_addThread(pattern, thread);
132 }
133 fseek(file, header.attributeOffset, SEEK_SET);
134 attributeData = (unsigned char*)malloc(header.xOffset - header.attributeOffset);
135 if (!attributeData) {
136 printf("ERROR: format-vip.c readVip(), cannot allocate memory for attributeData\n");
137 return 0;
138 }
139 fread(attributeData, 1, header.xOffset - header.attributeOffset, file); /* TODO: check return value */
140 attributeDataDecompressed = vipDecompressData(attributeData, header.xOffset - header.attributeOffset, header.numberOfStitches);
141
142 fseek(file, header.xOffset, SEEK_SET);
143 xData = (unsigned char*)malloc(header.yOffset - header.xOffset);
144 if (!xData) {
145 printf("ERROR: format-vip.c readVip(), cannot allocate memory for xData\n");
146 return 0;
147 }
148 fread(xData, 1, header.yOffset - header.xOffset, file); /* TODO: check return value */
149 xDecompressed = vipDecompressData(xData, header.yOffset - header.xOffset, header.numberOfStitches);
150
151 fseek(file, header.yOffset, SEEK_SET);
152 yData = (unsigned char*)malloc(fileLength - header.yOffset);
153 if (!yData) { printf("ERROR: format-vip.c readVip(), cannot allocate memory for yData\n"); return 0; }
154 fread(yData, 1, fileLength - header.yOffset, file); /* TODO: check return value */
155 yDecompressed = vipDecompressData(yData, fileLength - header.yOffset, header.numberOfStitches);
156
157 for (i = 0; i < header.numberOfStitches; i++) {
158 embPattern_addStitchRel(pattern,
159 vipDecodeByte(xDecompressed[i]) / 10.0,
160 vipDecodeByte(yDecompressed[i]) / 10.0,
161 vipDecodeStitchType(attributeDataDecompressed[i]), 1);
162 }
163 embPattern_addStitchRel(pattern, 0, 0, END, 1);
164
165 safe_free(attributeData);
166 safe_free(xData);
167 safe_free(yData);
168 safe_free(attributeDataDecompressed);
169 safe_free(xDecompressed);
170 safe_free(yDecompressed);
171 return 1;
172}

210char
211writeVip(EmbPattern* pattern, FILE* file)
212{
213 EmbRect boundingRect;
214 int stitchCount, minColors, patternColor;
215 int attributeSize = 0;
216 int xCompressedSize = 0;
217 int yCompressedSize = 0;
218 EmbReal previousX = 0;
219 EmbReal previousY = 0;
220 unsigned char* xValues = 0, *yValues = 0, *attributeValues = 0;
221 EmbReal xx = 0.0;
222 EmbReal yy = 0.0;
223 int flags = 0;
224 int i = 0;
225 unsigned char* attributeCompressed = 0, *xCompressed = 0;
226 unsigned char* yCompressed = 0, *decodedColors = 0, *encodedColors = 0;
227 unsigned char prevByte = 0;
228
229 puts("VIP not yet implemented.");
230 return 0;
231
232 stitchCount = pattern->stitch_list->count;
233 minColors = pattern->thread_list->count;
234 decodedColors = (unsigned char*)malloc(minColors << 2);
235 if (!decodedColors) {
236 return 0;
237 }
238 encodedColors = (unsigned char*)malloc(minColors << 2);
239 if (!encodedColors) {
240 safe_free(decodedColors);
241 return 0;
242 }
243 /* embPattern_correctForMaxStitchLength(pattern, 0x7F, 0x7F); */
244
245 patternColor = minColors;
246 if (emb_verbose>1) {
247 printf("patternColor: %d\n", patternColor);
248 }
249 if (minColors > 24) {
250 minColors = 24;
251 }
252 binaryWriteUInt(file, 0x0190FC5D);
253 binaryWriteUInt(file, stitchCount);
254 binaryWriteUInt(file, minColors);
255
256 boundingRect = embPattern_calcBoundingBox(pattern);
257 binaryWriteShort(file, (short) emb_round(boundingRect.right * 10.0));
258 binaryWriteShort(file, (short) -emb_round(boundingRect.top * 10.0 - 1.0));
259 binaryWriteShort(file, (short) emb_round(boundingRect.left * 10.0));
260 binaryWriteShort(file, (short) -emb_round(boundingRect.bottom * 10.0 - 1.0));
261
262 binaryWriteUInt(file, 0x38 + (minColors << 3));
263
264 xValues = (unsigned char*)malloc(sizeof(unsigned char)*(stitchCount));
265 yValues = (unsigned char*)malloc(sizeof(unsigned char)*(stitchCount));
266 attributeValues = (unsigned char*)malloc(sizeof(unsigned char)*(stitchCount));
267 if (xValues && yValues && attributeValues) {
268 for (i = 0; i < pattern->stitch_list->count; i++) {
269 EmbStitch st;
270 st = pattern->stitch_list->stitch[i];
271 xx = st.x;
272 yy = st.y;
273 flags = st.flags;
274 xValues[i] = vipEncodeByte((xx - previousX) * 10.0);
275 previousX = xx;
276 yValues[i] = vipEncodeByte((yy - previousY) * 10.0);
277 previousY = yy;
278 attributeValues[i] = vipEncodeStitchType(flags);
279 }
280 attributeCompressed = vipCompressData(attributeValues, stitchCount, &attributeSize);
281 xCompressed = vipCompressData(xValues, stitchCount, &xCompressedSize);
282 yCompressed = vipCompressData(yValues, stitchCount, &yCompressedSize);
283
284 binaryWriteUInt(file, (unsigned int) (0x38 + (minColors << 3) + attributeSize));
285 binaryWriteUInt(file, (unsigned int) (0x38 + (minColors << 3) + attributeSize + xCompressedSize));
286 binaryWriteUInt(file, 0x00000000);
287 binaryWriteUInt(file, 0x00000000);
288 binaryWriteUShort(file, 0x0000);
289
290 binaryWriteInt(file, minColors << 2);
291
292 for (i = 0; i < minColors; i++) {
293 int byteChunk = i << 2;
294 EmbColor currentColor = pattern->thread_list->thread[i].color;
295 decodedColors[byteChunk] = currentColor.r;
296 decodedColors[byteChunk + 1] = currentColor.g;
297 decodedColors[byteChunk + 2] = currentColor.b;
298 decodedColors[byteChunk + 3] = 0x01;
299 }
300
301 for (i = 0; i < minColors << 2; ++i) {
302 unsigned char tmpByte = (unsigned char) (decodedColors[i] ^ vipDecodingTable[i]);
303 prevByte = (unsigned char) (tmpByte ^ prevByte);
304 fputc(prevByte, file);
305 }
306 for (i = 0; i <= minColors; i++) {
307 binaryWriteInt(file, 1);
308 }
309 binaryWriteUInt(file, 0); /* string length */
310 binaryWriteShort(file, 0);
311 fwrite((char*) attributeCompressed, 1, attributeSize, file);
312 fwrite((char*) xCompressed, 1, xCompressedSize, file);
313 fwrite((char*) yCompressed, 1, yCompressedSize, file);
314 }
315
316 safe_free(attributeCompressed);
317 safe_free(xCompressed);
318 safe_free(yCompressed);
319 safe_free(attributeValues);
320 safe_free(xValues);
321 safe_free(yValues);
322 safe_free(decodedColors);
323 safe_free(encodedColors);
324 return 1;
325}
